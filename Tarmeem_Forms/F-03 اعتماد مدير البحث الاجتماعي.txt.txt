```react
import React, { useState } from 'react';
import { UserCog, CheckCircle2, RefreshCcw, FileText } from 'lucide-react';

const ReadOnlyField = ({ label, value }) => (
  <div className="flex flex-col gap-1">
    <label className="text-xs font-semibold text-gray-500">{label}</label>
    <div className="px-3 py-2 text-sm rounded-lg bg-gray-100 border border-gray-200 text-gray-700 font-medium">{value || '-'}</div>
  </div>
);

export default function App({ onSubmit, f02Data = {} }) {
  const [isSubmitting, setIsSubmitting] = useState(false);
  const [eligibility, setEligibility] = useState('');
  const [managerNotes, setManagerNotes] = useState('');

  const data = Object.keys(f02Data).length > 0 ? f02Data : {
    caseRef: 'CS-1234', personal: { fullName: 'محمد عبدالله', idNumber: '1029384756', city: 'الدمام' },
    housing: { type: 'شعبي قديم', need: 'ترميم كلي' }, researcher: { opinion: 'الأسرة محتاجة جداً والمنزل متهالك' }
  };

  const handleAction = async (actionType) => {
    setIsSubmitting(true);
    if (onSubmit) {
      await onSubmit({ formId: 'F-03', action: actionType, payload: { eligibility, managerNotes } });
    }
  };

  if (isSubmitting && !onSubmit) return (
     <div className="min-h-screen bg-gray-50 flex items-center justify-center p-4" dir="rtl">
       <div className="bg-teal-500 text-white rounded-xl p-8 shadow-lg text-center max-w-md w-full">
          <CheckCircle2 className="w-16 h-16 mx-auto mb-4 text-teal-100" />
          <h2 className="text-2xl font-bold mb-2">تم تحديث حالة الملف!</h2>
       </div>
     </div>
  );

  return (
    <div className="min-h-screen bg-gray-50 font-sans text-gray-800 pb-16" dir="rtl">
      <header className="bg-gradient-to-r from-purple-900 to-purple-700 text-white shadow-md pt-6 pb-12 px-4 text-center rounded-b-[2rem] relative z-10">
        <h1 className="text-2xl md:text-3xl font-extrabold mb-1">اعتماد مدير البحث (F-03)</h1>
      </header>

      <main className="container mx-auto px-4 -mt-8 relative z-20 max-w-3xl">
        <div className="bg-white rounded-xl shadow-sm border border-gray-200 mb-4 overflow-hidden">
          <div className="bg-gray-100 px-4 py-2.5 flex items-center gap-2 border-b border-gray-200">
            <FileText className="w-5 h-5 text-gray-500" />
            <h3 className="text-base font-bold text-gray-700">بيانات دراسة الحالة (للقراءة فقط)</h3>
          </div>
          <div className="p-4 grid grid-cols-1 md:grid-cols-3 gap-4">
            <ReadOnlyField label="رقم الحالة" value={data.caseRef} />
            <ReadOnlyField label="اسم المستفيد" value={data.personal?.fullName} />
            <ReadOnlyField label="المدينة" value={data.personal?.city} />
            <div className="md:col-span-3">
              <ReadOnlyField label="رأي الباحث الميداني" value={data.researcher?.opinion} />
            </div>
            <div className="md:col-span-3">
              <button className="text-sm font-bold text-purple-600 underline">عرض جميع بيانات F-02 والمرفقات الكاملة</button>
            </div>
          </div>
        </div>

        <div className="bg-white rounded-xl shadow-sm border border-gray-200 p-5">
          <div className="flex items-center gap-2 mb-4 text-purple-800">
            <UserCog className="w-6 h-6" /><h3 className="text-lg font-bold">قرار مدير البحث الاجتماعي</h3>
          </div>
          
          <label className="block text-sm font-bold text-gray-800 mb-3">الاستحقاق المبدئي:</label>
          <div className="flex gap-4 mb-5">
            <label className="flex items-center gap-2 cursor-pointer">
              <input type="radio" checked={eligibility === 'مستحق'} onChange={() => setEligibility('مستحق')} className="w-4 h-4 text-purple-600" />
              <span className="font-bold text-sm">مستحق</span>
            </label>
            <label className="flex items-center gap-2 cursor-pointer">
              <input type="radio" checked={eligibility === 'غير مستحق'} onChange={() => setEligibility('غير مستحق')} className="w-4 h-4 text-purple-600" />
              <span className="font-bold text-sm">غير مستحق</span>
            </label>
          </div>
          
          <div className="flex flex-col gap-1 mb-6">
            <label className="text-xs font-semibold text-gray-700">توصية المدير</label>
            <textarea rows={3} value={managerNotes} onChange={e => setManagerNotes(e.target.value)} placeholder="اكتب توصيتك أو أسباب الإرجاع هنا..." className="px-3 py-2 text-sm rounded-lg border border-gray-300 focus:border-purple-500 outline-none bg-white resize-none" />
          </div>

          <div className="flex gap-3 border-t border-gray-100 pt-4">
            <button onClick={() => handleAction('إعتماد')} disabled={!eligibility || isSubmitting} className="flex-1 bg-teal-600 hover:bg-teal-700 text-white py-3 rounded-lg font-bold text-sm flex justify-center items-center gap-2">
              <CheckCircle2 size={18} /> إعتماد ورفع للمدير التنفيذي
            </button>
            <button onClick={() => handleAction('إرجاع')} disabled={!eligibility || isSubmitting} className="flex-1 bg-yellow-500 hover:bg-yellow-600 text-white py-3 rounded-lg font-bold text-sm flex justify-center items-center gap-2">
              <RefreshCcw size={18} /> إرجاع للباحث الاجتماعي
            </button>
          </div>
        </div>
      </main>
    </div>
  );
}

```
