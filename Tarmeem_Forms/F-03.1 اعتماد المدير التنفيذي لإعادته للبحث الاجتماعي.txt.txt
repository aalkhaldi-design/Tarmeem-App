```react
import React, { useState } from 'react';
import { UserCheck, CheckCircle2, XCircle, FileText, UserCog } from 'lucide-react';

const ReadOnlyField = ({ label, value }) => (
  <div className="flex flex-col gap-1">
    <label className="text-xs font-semibold text-gray-500">{label}</label>
    <div className="px-3 py-2 text-sm rounded-lg bg-gray-100 border border-gray-200 text-gray-700 font-medium">{value || '-'}</div>
  </div>
);

export default function App({ onSubmit, f02Data = {}, f03Data = {} }) {
  const [isSubmitting, setIsSubmitting] = useState(false);
  const [executiveNotes, setExecutiveNotes] = useState('');

  const caseData = Object.keys(f02Data).length > 0 ? f02Data : { caseRef: 'CS-1234', personal: { fullName: 'محمد عبدالله' } };
  const managerData = Object.keys(f03Data).length > 0 ? f03Data : { eligibility: 'مستحق', managerNotes: 'يحتاج للترميم العاجل نظراً لسوء حالة السقف.' };

  const handleAction = async (actionType) => {
    setIsSubmitting(true);
    if (onSubmit) {
      await onSubmit({ formId: 'F-03.1', action: actionType, payload: { executiveNotes, decision: actionType } });
    }
  };

  if (isSubmitting && !onSubmit) return (
     <div className="min-h-screen bg-gray-50 flex items-center justify-center p-4" dir="rtl">
       <div className="bg-teal-500 text-white rounded-xl p-8 shadow-lg text-center max-w-md w-full">
          <CheckCircle2 className="w-16 h-16 mx-auto mb-4 text-teal-100" />
          <h2 className="text-2xl font-bold mb-2">تم تسجيل القرار!</h2>
       </div>
     </div>
  );

  return (
    <div className="min-h-screen bg-gray-50 font-sans text-gray-800 pb-16" dir="rtl">
      <header className="bg-gradient-to-r from-purple-900 to-purple-700 text-white shadow-md pt-6 pb-12 px-4 text-center rounded-b-[2rem] relative z-10">
        <h1 className="text-2xl md:text-3xl font-extrabold mb-1">اعتماد المدير التنفيذي (F-03.1)</h1>
      </header>

      <main className="container mx-auto px-4 -mt-8 relative z-20 max-w-3xl">
        <div className="bg-white rounded-xl shadow-sm border border-gray-200 mb-4 overflow-hidden p-4 grid grid-cols-1 md:grid-cols-2 gap-4">
           <ReadOnlyField label="رقم الحالة (من F-02)" value={caseData.caseRef} />
           <ReadOnlyField label="اسم المستفيد (من F-02)" value={caseData.personal?.fullName} />
        </div>

        <div className="bg-gray-100 rounded-xl shadow-sm border border-gray-300 mb-4 p-4">
          <div className="flex items-center gap-2 mb-3 text-gray-600"><UserCog className="w-5 h-5" /><h3 className="font-bold">رأي مدير البحث الاجتماعي السابقة</h3></div>
          <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
            <ReadOnlyField label="الاستحقاق" value={managerData.eligibility} />
            <ReadOnlyField label="الإجراء السابق" value="إعتماد للرفع" />
            <div className="md:col-span-2">
              <ReadOnlyField label="توصية مدير البحث" value={managerData.managerNotes} />
            </div>
          </div>
        </div>

        <div className="bg-white rounded-xl shadow-sm border border-purple-200 p-5">
          <div className="flex items-center gap-2 mb-4 text-purple-800"><UserCheck className="w-6 h-6" /><h3 className="text-lg font-bold">قرار المدير التنفيذي</h3></div>
          <div className="flex flex-col gap-1 mb-6">
            <label className="text-xs font-semibold text-gray-700">توصية المدير التنفيذي</label>
            <textarea rows={3} value={executiveNotes} onChange={e => setExecutiveNotes(e.target.value)} placeholder="اكتب ملاحظاتك وقرارك هنا..." className="px-3 py-2 text-sm rounded-lg border border-gray-300 focus:border-purple-500 outline-none resize-none" />
          </div>
          <div className="flex gap-3">
            <button onClick={() => handleAction('موافقة')} disabled={isSubmitting} className="flex-1 bg-green-600 hover:bg-green-700 text-white py-3 rounded-lg font-bold text-sm flex justify-center items-center gap-2">
              <CheckCircle2 size={18} /> موافقة (Approve)
            </button>
            <button onClick={() => handleAction('رفض')} disabled={isSubmitting} className="flex-1 bg-red-600 hover:bg-red-700 text-white py-3 rounded-lg font-bold text-sm flex justify-center items-center gap-2">
              <XCircle size={18} /> رفض (Disapprove)
            </button>
          </div>
          <p className="text-xs text-gray-500 text-center mt-3">كلا القرارين سيعيدان الملف لمدير البحث الاجتماعي لإكمال الإجراءات.</p>
        </div>
      </main>
    </div>
  );
}

```
