```react
import React, { useState } from 'react';
import { 
  Home, Calendar, ShieldCheck, CheckCircle2, UserCheck, UploadCloud, X, Plus, Trash2, Camera
} from 'lucide-react';

const Card = ({ title, icon: Icon, children, className = '' }) => (
  <div className={`bg-white rounded-xl shadow-sm border border-gray-200 overflow-hidden mb-4 transition-all duration-300 ${className}`}>
    <div className="bg-gradient-to-r from-purple-800 to-purple-600 px-4 py-2.5 flex items-center gap-2 text-white">
      {Icon && <Icon className="w-5 h-5 text-teal-200" />}
      <h3 className="text-base font-bold">{title}</h3>
    </div>
    <div className="p-4 md:p-5">{children}</div>
  </div>
);

const ReadOnlyField = ({ label, value }) => (
  <div className="flex flex-col gap-1">
    <label className="text-xs font-semibold text-gray-500">{label}</label>
    <div className="px-3 py-2 text-sm rounded-lg bg-gray-100 border border-gray-200 text-gray-700 font-medium">
      {value || '-'}
    </div>
  </div>
);

const Input = ({ label, type = 'text', value, onChange, placeholder, className = '', required = false }) => (
  <div className={`flex flex-col gap-1 ${className}`}>
    <label className="text-xs font-semibold text-gray-700">{label} {required && <span className="text-red-500">*</span>}</label>
    <input type={type} value={value} onChange={onChange} placeholder={placeholder} required={required} className="px-3 py-1.5 text-sm rounded-lg border border-gray-200 focus:border-teal-500 focus:ring-1 focus:ring-teal-200 outline-none text-gray-800 bg-gray-50" />
  </div>
);

export default function F07_HomeHandover({ onSubmit, f02Data = {}, f18Data = {}, projectData = {} }) {
  const [isSubmitting, setIsSubmitting] = useState(false);
  const [certificateFile, setCertificateFile] = useState(null);
  
  // Safe extraction from previous forms (F-02, F-18, Projects)
  const caseData = Object.keys(f02Data).length > 0 ? f02Data : {
    caseRef: 'CS-9082',
    personal: { fullName: 'عبدالله محمد السالم', city: 'الدمام', neighborhood: 'حي النور' },
    partnershipInfo: { partnerEntity: 'جمعية البر' }
  };

  const [data, setData] = useState({
    streetName: '',
    supervisingEngineer: projectData.engineerName || '',
    contractorName: projectData.contractorName || '',
    projectType: 'ترميم', // ترميم أو صيانة
    actualStartDate: f18Data.renovationStartDate || '',
    actualEndDate: '',
    insulationGuarantee: '',
    mediaRequested: false, // Triggers F-52
    employeePledge: false
  });

  const [guarantees, setGuarantees] = useState([
    { id: Date.now(), item: '', qty: '', supplier: '', invoiceNo: '', duration: '' }
  ]);

  const addGuaranteeRow = () => {
    setGuarantees([...guarantees, { id: Date.now(), item: '', qty: '', supplier: '', invoiceNo: '', duration: '' }]);
  };

  const removeGuaranteeRow = (id) => {
    setGuarantees(guarantees.filter(g => g.id !== id));
  };

  const updateGuarantee = (id, field, value) => {
    setGuarantees(guarantees.map(g => g.id === id ? { ...g, [field]: value } : g));
  };

  const handleFileChange = (e) => {
    if (e.target.files && e.target.files[0]) {
      setCertificateFile({
        file: e.target.files[0],
        name: e.target.files[0].name,
        previewUrl: URL.createObjectURL(e.target.files[0])
      });
    }
  };

  const removeFile = () => {
    if (certificateFile) URL.revokeObjectURL(certificateFile.previewUrl);
    setCertificateFile(null);
  };

  const handleSubmit = async (e) => {
    e.preventDefault();
    if (!data.employeePledge || !certificateFile) return;
    setIsSubmitting(true);
    
    if (onSubmit) {
      await onSubmit({ 
        formId: 'F-07', 
        payload: { ...data, guarantees, signedCertificate: certificateFile } 
      });
    }
  };

  if (isSubmitting && !onSubmit) {
    return (
       <div className="min-h-screen bg-gray-50 flex items-center justify-center p-4" dir="rtl">
         <div className="bg-teal-500 text-white rounded-xl p-8 shadow-lg text-center max-w-md w-full">
            <CheckCircle2 className="w-16 h-16 mx-auto mb-4 text-teal-100" />
            <h2 className="text-2xl font-bold mb-2">تم إصدار شهادة التسليم!</h2>
            <p className="text-teal-100 text-sm mb-4">تم حفظ بيانات الضمانات وتوثيق تسليم المنزل بنجاح.</p>
            {data.mediaRequested && (
              <div className="bg-white/20 p-3 rounded-lg border border-white/30 text-sm font-bold flex items-center justify-center gap-2">
                <Camera size={18} />
                تم إرسال طلب التغطية الإعلامية (F-52)
              </div>
            )}
         </div>
       </div>
    );
  }

  return (
    <div className="min-h-screen bg-gray-50 font-sans text-gray-800 pb-16" dir="rtl">
      <header className="bg-gradient-to-r from-purple-900 to-purple-700 text-white shadow-md pt-6 pb-12 px-4 text-center rounded-b-[2rem] relative z-10">
        <h1 className="text-2xl md:text-3xl font-extrabold mb-1">شهادة تسليم منزل (F-07)</h1>
        <h2 className="text-sm md:text-base text-teal-300 font-medium">الاستلام النهائي، توثيق الضمانات، وإغلاق المشروع</h2>
      </header>

      <main className="container mx-auto px-4 -mt-8 relative z-20 max-w-4xl">
        <form onSubmit={handleSubmit}>
          
          {/* Project & Beneficiary Info */}
          <Card title="بيانات المشروع" icon={Home}>
            <div className="grid grid-cols-2 md:grid-cols-4 gap-4 mb-4">
              <ReadOnlyField label="اسم المنشأة" value="جمعية ترميم" />
              <ReadOnlyField label="اسم المستفيد" value={caseData.personal?.fullName} />
              <ReadOnlyField label="المدينة" value={caseData.personal?.city} />
              <ReadOnlyField label="الحي" value={caseData.personal?.neighborhood} />
            </div>
            <div className="grid grid-cols-1 md:grid-cols-3 gap-4">
              <Input label="الشارع" required value={data.streetName} onChange={e => setData({...data, streetName: e.target.value})} />
              <ReadOnlyField label="الجهات الشريكة" value={caseData.partnershipInfo?.partnerEntity} />
              <Input label="المهندس المشرف" required value={data.supervisingEngineer} onChange={e => setData({...data, supervisingEngineer: e.target.value})} />
              <Input label="المقاول المنفذ" required className="md:col-span-3" value={data.contractorName} onChange={e => setData({...data, contractorName: e.target.value})} />
            </div>
          </Card>

          {/* Execution Period */}
          <Card title="فترة التنفيذ ونوع المشروع" icon={Calendar}>
            <div className="flex items-center gap-4 mb-5 bg-gray-50 p-3 rounded-lg border border-gray-200">
              <span className="text-sm font-bold text-gray-700">تم الانتهاء من:</span>
              <label className="flex items-center gap-2 cursor-pointer">
                <input type="radio" checked={data.projectType === 'ترميم'} onChange={() => setData({...data, projectType: 'ترميم'})} className="w-4 h-4 text-purple-600" />
                <span className="font-bold text-sm">ترميم</span>
              </label>
              <label className="flex items-center gap-2 cursor-pointer">
                <input type="radio" checked={data.projectType === 'صيانة'} onChange={() => setData({...data, projectType: 'صيانة'})} className="w-4 h-4 text-purple-600" />
                <span className="font-bold text-sm">صيانة</span>
              </label>
              <span className="text-sm text-gray-600 mr-2">المنزل وإنجاز جميع بنود الأعمال المعتمدة.</span>
            </div>

            <div className="grid grid-cols-1 md:grid-cols-2 gap-5">
              <Input label="بدء المشروع في (التاريخ الفعلي)" type="date" required value={data.actualStartDate} onChange={e => setData({...data, actualStartDate: e.target.value})} />
              <Input label="وتم الانتهاء في (تاريخ التسليم)" type="date" required value={data.actualEndDate} onChange={e => setData({...data, actualEndDate: e.target.value})} />
            </div>
          </Card>

          {/* Guarantees Table */}
          <Card title="الضمانات المتوفرة" icon={ShieldCheck}>
            <div className="overflow-x-auto mb-4">
              <table className="w-full text-right border-collapse min-w-[600px]">
                <thead>
                  <tr className="bg-teal-600 text-white text-sm">
                    <th className="p-2 border border-teal-700 font-bold rounded-tr-lg w-1/4">الصنف</th>
                    <th className="p-2 border border-teal-700 font-bold w-16 text-center">العدد</th>
                    <th className="p-2 border border-teal-700 font-bold">التنفيذ والتوريد (الجهة)</th>
                    <th className="p-2 border border-teal-700 font-bold">رقم الفاتورة</th>
                    <th className="p-2 border border-teal-700 font-bold w-1/5">مدة الضمان</th>
                    <th className="p-2 border border-teal-700 font-bold rounded-tl-lg w-10"></th>
                  </tr>
                </thead>
                <tbody>
                  {guarantees.map((row) => (
                    <tr key={row.id} className="bg-gray-50 border-b border-gray-200">
                      <td className="p-1 border-x border-gray-200"><input type="text" required value={row.item} onChange={e => updateGuarantee(row.id, 'item', e.target.value)} className="w-full px-2 py-1 text-sm rounded border-gray-300 outline-none focus:border-teal-500" placeholder="مثال: مكيف سبلت" /></td>
                      <td className="p-1 border-x border-gray-200"><input type="number" value={row.qty} onChange={e => updateGuarantee(row.id, 'qty', e.target.value)} className="w-full px-2 py-1 text-sm rounded border-gray-300 outline-none text-center" /></td>
                      <td className="p-1 border-x border-gray-200"><input type="text" value={row.supplier} onChange={e => updateGuarantee(row.id, 'supplier', e.target.value)} className="w-full px-2 py-1 text-sm rounded border-gray-300 outline-none focus:border-teal-500" /></td>
                      <td className="p-1 border-x border-gray-200"><input type="text" value={row.invoiceNo} onChange={e => updateGuarantee(row.id, 'invoiceNo', e.target.value)} className="w-full px-2 py-1 text-sm rounded border-gray-300 outline-none focus:border-teal-500" /></td>
                      <td className="p-1 border-x border-gray-200"><input type="text" value={row.duration} onChange={e => updateGuarantee(row.id, 'duration', e.target.value)} className="w-full px-2 py-1 text-sm rounded border-gray-300 outline-none focus:border-teal-500" placeholder="سنتين / 5 سنوات..." /></td>
                      <td className="p-1 text-center border-x border-gray-200">
                        {guarantees.length > 1 && (
                          <button type="button" onClick={() => removeGuaranteeRow(row.id)} className="text-red-500 hover:bg-red-100 p-1 rounded"><Trash2 size={16} /></button>
                        )}
                      </td>
                    </tr>
                  ))}
                </tbody>
              </table>
            </div>
            
            <button type="button" onClick={addGuaranteeRow} className="bg-teal-50 text-teal-700 border border-dashed border-teal-300 px-4 py-2 rounded-lg text-sm font-bold flex items-center gap-2 hover:bg-teal-100 transition-colors mb-5">
              <Plus size={16} /> إضافة صنف ضمان جديد
            </button>

            <div className="w-1/2">
              <Input label="مدة ضمان العازل (إن وجد)" value={data.insulationGuarantee} onChange={e => setData({...data, insulationGuarantee: e.target.value})} placeholder="مثال: 10 سنوات" />
            </div>
          </Card>

          {/* Media Request Trigger */}
          <div className={`mb-4 p-5 rounded-xl border-2 transition-all ${data.mediaRequested ? 'bg-purple-50 border-purple-500' : 'bg-white border-gray-200'}`}>
            <div className="flex items-start gap-4">
              <input 
                type="checkbox" 
                id="mediaRequested"
                checked={data.mediaRequested}
                onChange={(e) => setData({...data, mediaRequested: e.target.checked})}
                className="w-6 h-6 mt-1 rounded text-purple-600 focus:ring-purple-500 cursor-pointer"
              />
              <label htmlFor="mediaRequested" className="cursor-pointer select-none">
                <h4 className="text-base font-bold text-gray-900 mb-1 flex items-center gap-2">
                  <Camera className="w-5 h-5 text-purple-600" />
                  طلب تغطية إعلامية وتوثيق مرئي לתسليم (يُفعل F-52)
                </h4>
                <p className="text-sm text-gray-600 mt-1 font-medium">
                  تحديد هذا الخيار سيقوم النظام من خلاله تلقائياً بإرسال طلب (نموذج F-52) إلى مسؤول العلاقات العامة والإعلام (PR/Media Officer) لتوثيق التسليم للجهة الداعمة.
                </p>
              </label>
            </div>
          </div>

          {/* File Upload for the signed paper */}
          <Card title="إرفاق الشهادة الموقعة" icon={UploadCloud}>
            <div className="bg-purple-50 p-6 rounded-xl border-2 border-dashed border-purple-200 text-center hover:bg-purple-100 transition-colors">
              <UploadCloud className="w-10 h-10 text-purple-500 mx-auto mb-2" />
              <p className="text-sm font-bold text-purple-900 mb-1">إرفاق نسخة الشهادة الموقعة من الأطراف *</p>
              <p className="text-xs text-purple-600 mb-4">الشهادة التي تحمل تواقيع (المستفيد، المهندس، مدير المشاريع، والبحث الاجتماعي)</p>
              <label className="cursor-pointer bg-purple-600 text-white px-6 py-2.5 rounded-lg text-sm font-bold hover:bg-purple-700 shadow-sm">
                تصفح الملفات
                <input type="file" required={!certificateFile} accept="image/*,.pdf" className="hidden" onChange={handleFileChange} />
              </label>
            </div>
            {certificateFile && (
              <div className="mt-4 flex justify-between items-center bg-white p-3 rounded-lg border border-teal-200 text-sm shadow-sm bg-teal-50">
                <div className="flex items-center gap-2 text-teal-800 font-bold">
                  <CheckCircle2 size={18} />
                  <span className="truncate max-w-[200px]">{certificateFile.name}</span>
                </div>
                <button type="button" onClick={removeFile} className="text-red-500 hover:bg-red-50 p-1 rounded"><X size={18}/></button>
              </div>
            )}
          </Card>

          {/* Final Pledge */}
          <Card title="توثيق الاعتماد للإغلاق" icon={UserCheck}>
            <div className={`p-4 rounded-xl border transition-all ${data.employeePledge ? 'bg-teal-50 border-teal-500' : 'bg-gray-50 border-gray-200'}`}>
              <div className="flex items-start gap-4">
                <input 
                  type="checkbox" 
                  id="employeePledge"
                  required
                  checked={data.employeePledge}
                  onChange={(e) => setData({...data, employeePledge: e.target.checked})}
                  className="w-6 h-6 mt-1 rounded text-teal-600 focus:ring-teal-500 cursor-pointer"
                />
                <label htmlFor="employeePledge" className="cursor-pointer select-none">
                  <h4 className="text-sm font-bold text-gray-900 mb-2">اعتماد تسليم المنزل</h4>
                  <p className="text-sm text-gray-700 leading-relaxed font-medium">
                    أقر أنا المهندس المشرف، باستكمال جميع بنود الأعمال المتفق عليها، وأن المستفيد قد استلم المنزل مع كافة الضمانات المرفقة أعلاه، وبذلك يُعتبر المشروع منتهياً فنياً.
                  </p>
                </label>
              </div>
            </div>
          </Card>

          {/* Submit */}
          <button 
            type="submit"
            disabled={!data.employeePledge || !certificateFile || isSubmitting}
            className={`w-full py-4 rounded-xl text-base font-bold flex items-center justify-center gap-2 transition-all shadow-md
              ${(data.employeePledge && certificateFile)
                ? 'bg-gradient-to-r from-teal-600 to-teal-500 hover:from-teal-700 text-white cursor-pointer' 
                : 'bg-gray-200 text-gray-400 cursor-not-allowed'}`}
          >
            <ShieldCheck className="w-5 h-5" /> 
            {isSubmitting ? 'جاري إغلاق المشروع...' : 'اعتماد شهادة التسليم وإغلاق المشروع'}
          </button>
        </form>
      </main>
    </div>
  );
}

```
