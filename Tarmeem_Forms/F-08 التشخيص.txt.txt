```react
import React, { useState } from 'react';
import { 
  Plus, Trash2, Image as ImageIcon, 
  CheckCircle2, Home, Users, ClipboardList, Sofa, 
  FileSignature, X, ChevronDown, ChevronUp, ChevronLeft, ChevronRight, AlertTriangle, PenTool
} from 'lucide-react';

// --- Reusable UI Components ---
const Card = ({ title, icon: Icon, children, className = '' }) => (
  <div className={`bg-white rounded-xl shadow-sm border border-gray-200 overflow-hidden mb-4 transition-all duration-300 ${className}`}>
    <div className="bg-gradient-to-r from-purple-800 to-purple-600 px-4 py-2.5 flex items-center gap-2 text-white">
      {Icon && <Icon className="w-5 h-5 text-teal-200" />}
      <h3 className="text-base font-bold">{title}</h3>
    </div>
    <div className="p-4 md:p-5">{children}</div>
  </div>
);

const Input = ({ label, type = 'text', value, onChange, placeholder, readOnly = false, className = '' }) => (
  <div className={`flex flex-col gap-1 ${className}`}>
    <label className="text-xs font-semibold text-gray-700">{label}</label>
    <input
      type={type}
      value={value}
      onChange={onChange}
      placeholder={placeholder}
      readOnly={readOnly}
      className={`px-3 py-1.5 text-sm rounded-lg border focus:outline-none transition-all text-gray-800 
      ${readOnly ? 'bg-gray-100 border-gray-200 cursor-not-allowed' : 'bg-gray-50 border-gray-200 focus:border-teal-500 focus:ring-1 focus:ring-teal-200'}`}
    />
  </div>
);

const TextArea = ({ label, value, onChange, placeholder, rows = 2, className = '' }) => (
  <div className={`flex flex-col gap-1 ${className}`}>
    <label className="text-xs font-semibold text-gray-700">{label}</label>
    <textarea
      value={value}
      onChange={onChange}
      placeholder={placeholder}
      rows={rows}
      className="px-3 py-2 text-sm rounded-lg border border-gray-200 focus:border-teal-500 focus:ring-1 focus:ring-teal-200 outline-none transition-all text-gray-800 bg-gray-50/50 resize-none"
    />
  </div>
);

const NumberCounter = ({ label, value, onChange }) => (
  <div className="flex items-center justify-between p-2 bg-gray-50 rounded-lg border border-gray-100">
    <span className="text-xs font-semibold text-gray-700 truncate mr-2">{label}</span>
    <div className="flex items-center gap-2 bg-white rounded-md border shadow-sm p-0.5">
      <button 
        type="button"
        onClick={() => onChange(Math.max(0, (Number(value) || 0) - 1))}
        className="w-6 h-6 flex items-center justify-center rounded bg-gray-100 hover:bg-red-100 hover:text-red-600 transition-colors text-gray-600 font-bold text-sm"
      >
        -
      </button>
      <span className="w-5 text-center text-sm font-bold text-purple-900">{value || 0}</span>
      <button 
        type="button"
        onClick={() => onChange((Number(value) || 0) + 1)}
        className="w-6 h-6 flex items-center justify-center rounded bg-gray-100 hover:bg-teal-100 hover:text-teal-600 transition-colors text-gray-600 font-bold text-sm"
      >
        +
      </button>
    </div>
  </div>
);

export default function F08_DiagnosisReport({ onSubmit, f02Data = {} }) {
  const [currentStep, setCurrentStep] = useState(0);
  const [isSubmitting, setIsSubmitting] = useState(false);
  
  // Data State bounded by Workflow Context
  const [data, setData] = useState({
    general: {
      caseRef: f02Data.caseRef || '', 
      projectNumber: f02Data.projectNumber || '', 
      familyName: f02Data.familyName || '', 
      contactNumber: f02Data.contactNumber || '', 
      cityNeighborhood: f02Data.cityNeighborhood || '', 
      partnerEntity: f02Data.partnerEntity || '', 
      partnerRep: f02Data.partnerRep || '', 
      repContact: f02Data.repContact || '',
      safetyHazard: false // Trigger for F-18 & F-22
    },
    visit: {
      date: new Date().toISOString().split('T')[0], type: 'منزل', area: '', age: '', team: '', diagnosis: 'قابل للترميم', summary: ''
    },
    works: [],
    furnitureInfo: { type: 'منزل', males: 0, females: 0, condition: 'ترميم' },
    furnitureItems: { bed15: 0, mattress15: 0, bed1: 0, mattress1: 0, bedDouble: 0, mattressDouble: 0, carpet: 0, sofaSeats: 0, floorSeating: 0, nightstand: 0, wardrobe2: 0, dresser: 0, wardrobe3: 0, wardrobe4: 0, sofaMeters: 0 },
    appliances: { acSplit1: 0, acWindow15: 0, acSplit15: 0, washer: 0, vacuum: 0, waterCooler: 0, fridge: 0, stove: 0 },
    diagnosisNotes: '', // Optional notes just in case
    pledge: false
  });

  const steps = [
    { id: 0, title: 'البيانات الأساسية', icon: Home },
    { id: 1, title: 'حصر الأعمال', icon: ClipboardList },
    { id: 2, title: 'الأثاث والأجهزة', icon: Sofa },
    { id: 3, title: 'الاعتماد والرفع', icon: FileSignature }
  ];

  const updateNestedData = (section, field, value) => {
    setData(prev => ({ ...prev, [section]: { ...prev[section], [field]: value } }));
  };

  const addWorkSpace = () => {
    const newSpace = {
      id: Date.now(), name: '', isExpanded: true, images: [],
      civil: { concrete: '', roof: '', insulation: '', shinko: '', ceramic: '', paint: '', plaster: '', wood: '', aluminum: '', steel: '', notes: '' },
      electrical: { panel: 0, ceilingLight: 0, concreteLight: 0, spotlight: 0, sockets: 0, doubleSwitch: 0, acSwitch: 0, heaterSocket: 0 },
      plumbing: { toiletFr: 0, toiletAr: 0, heater: 0, bidet: 0, showerMixer: 0, sink: 0, sinkMixer: 0, exhaust: 0 }
    };
    setData(prev => ({ ...prev, works: [...prev.works, newSpace] }));
  };

  const toggleExpand = (id) => setData(prev => ({ ...prev, works: prev.works.map(w => w.id === id ? { ...w, isExpanded: !w.isExpanded } : w) }));
  const removeWorkSpace = (id) => setData(prev => ({ ...prev, works: prev.works.filter(w => w.id !== id) }));
  const updateWorkSpace = (id, category, field, value) => {
    setData(prev => ({ ...prev, works: prev.works.map(w => {
        if (w.id === id) {
          if (category) return { ...w, [category]: { ...w[category], [field]: value } };
          return { ...w, [field]: value };
        }
        return w;
      })
    }));
  };

  // Image Upload optimized for Cloud Storage Integration (Supports Unlimited Images robustly)
  const handleImageUpload = (spaceId, e) => {
    const files = Array.from(e.target.files);
    if (!files.length) return;

    // Create Object URLs for UI preview, keeping the actual File object for Firebase
    const newImages = files.map(file => ({
      file: file,
      previewUrl: URL.createObjectURL(file)
    }));

    setData(prev => ({
      ...prev, works: prev.works.map(w => w.id === spaceId ? { ...w, images: [...w.images, ...newImages] } : w)
    }));
    
    // Reset input value to allow selecting the same file again if needed
    e.target.value = null; 
  };

  const removeImage = (spaceId, imageIndex) => {
    setData(prev => ({
      ...prev, works: prev.works.map(w => {
        if (w.id === spaceId) {
          const newImages = [...w.images];
          URL.revokeObjectURL(newImages[imageIndex].previewUrl); // Prevent memory leaks
          newImages.splice(imageIndex, 1);
          return { ...w, images: newImages };
        }
        return w;
      })
    }));
  };

  const handleSubmit = async () => {
    if (!data.pledge) return;
    setIsSubmitting(true);
    
    // Passing the complete payload to Claude's unified Forms Engine
    if (onSubmit) {
      await onSubmit({ formId: 'F-08', payload: data });
    }
    window.scrollTo({ top: 0, behavior: 'smooth' });
  };

  if (isSubmitting && !onSubmit) {
    return (
       <div className="min-h-screen bg-gray-50 flex items-center justify-center p-4">
         <div className="bg-teal-500 text-white rounded-xl p-8 shadow-lg text-center max-w-md w-full">
            <CheckCircle2 className="w-16 h-16 mx-auto mb-4 text-teal-100" />
            <h2 className="text-2xl font-bold mb-2">تم رفع كراسة التشخيص!</h2>
            <p className="text-teal-100 text-sm">تم تغذية النظام بالكميات والأثاث واعتماد التقرير.</p>
         </div>
       </div>
    );
  }

  return (
    <div className="min-h-screen bg-gray-50 font-sans text-gray-800 pb-16" dir="rtl">
      
      {/* Header */}
      <header className="bg-gradient-to-r from-purple-900 to-purple-700 text-white shadow-md pt-6 pb-12 px-4 text-center rounded-b-[2rem] relative z-10">
        <h1 className="text-2xl md:text-3xl font-extrabold mb-1">كراسة تشخيص مبنى (F-08)</h1>
        <h2 className="text-sm md:text-base text-teal-300 font-medium">وفق الدليل الإجرائي المعتمد لإدارة المشاريع</h2>
      </header>

      <main className="container mx-auto px-4 -mt-8 relative z-20 max-w-4xl">
        
        {/* Stepper */}
        <div className="flex overflow-x-auto gap-2 mb-4 bg-white p-1.5 rounded-xl shadow-sm border border-gray-100 custom-scrollbar">
          {steps.map((step, idx) => (
            <button
              key={step.id}
              onClick={() => setCurrentStep(step.id)}
              className={`flex-1 flex min-w-[120px] items-center justify-center gap-2 py-2 px-2 rounded-lg text-xs font-bold transition-all ${currentStep === step.id ? 'bg-purple-100 text-purple-800 shadow-sm border border-purple-200' : 'text-gray-500 hover:bg-gray-50'}`}
            >
              <div className={`w-5 h-5 rounded-full flex items-center justify-center text-[10px] ${currentStep === step.id ? 'bg-purple-600 text-white' : 'bg-gray-200 text-gray-600'}`}>
                {idx + 1}
              </div>
              <span className="whitespace-nowrap">{step.title}</span>
            </button>
          ))}
        </div>

        {/* Step 0: General Info */}
        <div className={`${currentStep === 0 ? 'block animate-fade-in' : 'hidden'}`}>
          <Card title="بيانات الأسرة والمنزل (تُجلب من F-02)" icon={Home}>
            <div className="grid grid-cols-2 md:grid-cols-4 gap-3 mb-5">
              <Input label="اسم الأسرة" value={data.general.familyName} onChange={e => updateNestedData('general', 'familyName', e.target.value)} readOnly={!!f02Data.familyName} />
              <Input label="رقم التواصل" value={data.general.contactNumber} onChange={e => updateNestedData('general', 'contactNumber', e.target.value)} readOnly={!!f02Data.contactNumber} />
              <Input label="رقم الحالة المرجعي" value={data.general.caseRef} onChange={e => updateNestedData('general', 'caseRef', e.target.value)} readOnly={!!f02Data.caseRef} />
              <Input label="رقم المشروع" value={data.general.projectNumber} onChange={e => updateNestedData('general', 'projectNumber', e.target.value)} />
              <Input label="المدينة – الحي" value={data.general.cityNeighborhood} onChange={e => updateNestedData('general', 'cityNeighborhood', e.target.value)} />
              <Input label="الجهة الشريكة" value={data.general.partnerEntity} onChange={e => updateNestedData('general', 'partnerEntity', e.target.value)} />
              <Input label="ممثل الجهة الشريكة" value={data.general.partnerRep} onChange={e => updateNestedData('general', 'partnerRep', e.target.value)} />
              <Input label="رقم تواصل الممثل" value={data.general.repContact} onChange={e => updateNestedData('general', 'repContact', e.target.value)} />
            </div>

            <hr className="my-4 border-gray-100" />
            
            <div className="flex items-center gap-2 mb-3">
              <Users className="w-4 h-4 text-purple-700" />
              <h4 className="font-bold text-sm text-purple-800">زيارة التشخيص</h4>
            </div>
            
            <div className="grid grid-cols-2 md:grid-cols-4 gap-3 mb-4">
              <Input label="تاريخ الزيارة" type="date" value={data.visit.date} onChange={e => updateNestedData('visit', 'date', e.target.value)} />
              <div className="flex flex-col gap-1">
                <label className="text-xs font-semibold text-gray-700">نوع المبنى</label>
                <select className="px-3 py-1.5 text-sm rounded-lg border border-gray-200 bg-gray-50 focus:border-teal-500 focus:ring-1 focus:ring-teal-200 outline-none" value={data.visit.type} onChange={e => updateNestedData('visit', 'type', e.target.value)}>
                  <option>منزل</option>
                  <option>شقة</option>
                  <option>فيلا</option>
                  <option>شعبي</option>
                </select>
              </div>
              <Input label="المساحة التقريبية" value={data.visit.area} onChange={e => updateNestedData('visit', 'area', e.target.value)} />
              <Input label="العمر التقديري" value={data.visit.age} onChange={e => updateNestedData('visit', 'age', e.target.value)} />
              <Input label="فريق التشخيص" className="col-span-2 md:col-span-4" value={data.visit.team} onChange={e => updateNestedData('visit', 'team', e.target.value)} />
            </div>

            <div className="bg-gray-50 p-3 rounded-lg border border-gray-100 mb-4">
              <label className="block text-xs font-semibold text-gray-700 mb-2">النتيجة المبدئية للزيارة</label>
              <div className="flex flex-wrap gap-3">
                {['المبنى قابل للترميم', 'المبنى لا يحتاج للترميم', 'المبنى آيل للسقوط', 'يحول صيانة'].map(opt => (
                  <label key={opt} className="flex items-center gap-1.5 cursor-pointer bg-white px-3 py-1.5 rounded-md border shadow-sm hover:border-teal-400 transition-colors">
                    <input type="radio" value={opt} checked={data.visit.diagnosis === opt} onChange={e => updateNestedData('visit', 'diagnosis', e.target.value)} className="text-teal-600 focus:ring-teal-500 w-3 h-3" />
                    <span className="text-xs font-medium">{opt}</span>
                  </label>
                ))}
              </div>
            </div>

            {/* Safety Hazard Trigger - Crucial for F-18 and F-22 */}
            <div className={`mb-4 p-4 rounded-xl flex items-start gap-3 transition-all border-2 ${data.general.safetyHazard ? 'bg-red-50 border-red-400' : 'bg-white border-gray-100'}`}>
              <div className="mt-1">
                <input
                  type="checkbox"
                  id="safetyHazard"
                  checked={data.general.safetyHazard}
                  onChange={(e) => updateNestedData('general', 'safetyHazard', e.target.checked)}
                  className="w-5 h-5 text-red-600 rounded focus:ring-red-500 cursor-pointer"
                />
              </div>
              <label htmlFor="safetyHazard" className="cursor-pointer select-none">
                <div className="flex items-center gap-2 text-sm font-bold text-red-600">
                  <AlertTriangle size={18} />
                  تنبيه سلامة: المبنى يشكل خطراً إنشائياً (يفعل طلب الإخلاء F-18)
                </div>
                <p className="text-xs text-gray-600 mt-1">تحديد هذا الخيار سيقوم النظام تلقائياً بإنشاء (F-18 طلب إخلاء) و (F-22 إسكان بديل) وتحويلها في سلسلة الاعتمادات.</p>
              </label>
            </div>

            <TextArea label="ملخص تقييم المبنى (أسباب قابلية أو عدم قابلية الترميم)" rows={3} value={data.visit.summary} onChange={e => updateNestedData('visit', 'summary', e.target.value)} />
          </Card>
        </div>

        {/* Step 1: Works Quantities */}
        <div className={`${currentStep === 1 ? 'block animate-fade-in' : 'hidden'}`}>
          <Card title="حصر كميات الأعمال (البنود المبدئية)" icon={ClipboardList}>
            <div className="space-y-4">
              {data.works.length === 0 ? (
                <div className="bg-gray-50 p-8 rounded-xl text-center border-2 border-dashed border-gray-200">
                  <ClipboardList className="w-10 h-10 text-gray-400 mx-auto mb-3" />
                  <h3 className="text-sm font-bold text-gray-600 mb-1">لا توجد مساحات مضافة</h3>
                  <button onClick={addWorkSpace} className="mt-2 bg-teal-600 hover:bg-teal-700 text-white px-4 py-2 rounded-lg text-sm font-bold transition-colors inline-flex items-center gap-2">
                    <Plus className="w-4 h-4" /> إضافة مساحة (غرفة، صالة، مطبخ، الخ)
                  </button>
                </div>
              ) : (
                data.works.map((space, index) => (
                  <div key={space.id} className="bg-white rounded-xl border border-gray-200 overflow-hidden">
                    <div className="bg-gray-50 border-b border-gray-100 p-2.5 flex items-center justify-between gap-3">
                      <div className="flex-1 flex items-center gap-2">
                        <button onClick={() => toggleExpand(space.id)} className="p-1.5 text-gray-500 hover:bg-gray-200 rounded-md">
                          {space.isExpanded ? <ChevronUp className="w-4 h-4" /> : <ChevronDown className="w-4 h-4" />}
                        </button>
                        <span className="font-bold text-gray-500 text-sm">{index + 1}.</span>
                        <input 
                          type="text" 
                          placeholder="اسم المساحة (مثال: غرفة النوم 1)" 
                          value={space.name}
                          onChange={(e) => updateWorkSpace(space.id, null, 'name', e.target.value)}
                          className="w-full text-sm font-bold bg-white border border-gray-200 rounded px-2.5 py-1.5 focus:ring-1 focus:ring-purple-500 outline-none"
                        />
                      </div>
                      <button onClick={() => removeWorkSpace(space.id)} className="text-red-500 hover:bg-red-50 p-1.5 rounded flex items-center">
                        <Trash2 className="w-4 h-4" />
                      </button>
                    </div>

                    <div className={`${space.isExpanded ? 'block' : 'hidden'} p-4 space-y-5`}>
                      
                      {/* Robust Image Upload (Unlimited) */}
                      <div className="bg-purple-50/40 p-3 rounded-lg border border-purple-100">
                        <div className="flex justify-between items-center mb-2">
                          <label className="text-xs font-bold text-purple-800 flex items-center gap-1.5">
                            <ImageIcon className="w-4 h-4" /> صور الموقع / المساحة (لامحدود)
                          </label>
                          <label className="cursor-pointer bg-white border border-purple-200 text-purple-700 hover:bg-purple-100 px-3 py-1.5 rounded-md text-xs font-bold transition-colors flex items-center gap-1">
                            <Plus className="w-3 h-3" /> إرفاق صور
                            <input type="file" multiple accept="image/*" className="hidden" onChange={(e) => handleImageUpload(space.id, e)} />
                          </label>
                        </div>
                        {space.images.length > 0 ? (
                          <div className="flex gap-2 overflow-x-auto pb-1 custom-scrollbar mt-3">
                            {space.images.map((imgObj, i) => (
                              <div key={i} className="relative flex-shrink-0 group">
                                <img src={imgObj.previewUrl} alt="Preview" className="w-16 h-16 object-cover rounded border border-gray-200" />
                                <button onClick={() => removeImage(space.id, i)} className="absolute -top-1 -right-1 bg-red-500 text-white rounded-full p-0.5 opacity-0 group-hover:opacity-100 transition-opacity">
                                  <X className="w-3 h-3" />
                                </button>
                              </div>
                            ))}
                          </div>
                        ) : (
                          <p className="text-[11px] text-gray-500 mt-1">يمكنك إرفاق أي عدد من الصور للمساحة لتوثيق الحالة.</p>
                        )}
                      </div>

                      {/* Quantities Grids */}
                      <div>
                        <h4 className="text-xs font-bold text-teal-700 mb-2 border-b pb-1">الأعمال المدنية والتشطيبات</h4>
                        <div className="grid grid-cols-2 md:grid-cols-4 lg:grid-cols-5 gap-2.5">
                          <Input label="المعالجة الخرسانية" value={space.civil.concrete} onChange={e => updateWorkSpace(space.id, 'civil', 'concrete', e.target.value)} />
                          <Input label="أعمال الأسقف" value={space.civil.roof} onChange={e => updateWorkSpace(space.id, 'civil', 'roof', e.target.value)} />
                          <Input label="أعمال العزل" value={space.civil.insulation} onChange={e => updateWorkSpace(space.id, 'civil', 'insulation', e.target.value)} />
                          <Input label="أعمال الشينكو" value={space.civil.shinko} onChange={e => updateWorkSpace(space.id, 'civil', 'shinko', e.target.value)} />
                          <Input label="سيراميك (أرض/جدار)" value={space.civil.ceramic} onChange={e => updateWorkSpace(space.id, 'civil', 'ceramic', e.target.value)} />
                          <Input label="دهانات (داخلي/خارجي)" value={space.civil.paint} onChange={e => updateWorkSpace(space.id, 'civil', 'paint', e.target.value)} />
                          <Input label="أعمال المساح" value={space.civil.plaster} onChange={e => updateWorkSpace(space.id, 'civil', 'plaster', e.target.value)} />
                          <Input label="نجارة (أبواب)" value={space.civil.wood} onChange={e => updateWorkSpace(space.id, 'civil', 'wood', e.target.value)} />
                          <Input label="ألمنيوم (نوافذ)" value={space.civil.aluminum} onChange={e => updateWorkSpace(space.id, 'civil', 'aluminum', e.target.value)} />
                          <Input label="أعمال الحدادة" value={space.civil.steel} onChange={e => updateWorkSpace(space.id, 'civil', 'steel', e.target.value)} />
                        </div>
                        <div className="mt-2.5">
                          <TextArea label="ملاحظات" rows={1} value={space.civil.notes} onChange={e => updateWorkSpace(space.id, 'civil', 'notes', e.target.value)} />
                        </div>
                      </div>

                      <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
                        <div>
                          <h4 className="text-xs font-bold text-teal-700 mb-2 border-b pb-1">أعمال الكهرباء</h4>
                          <div className="grid grid-cols-2 gap-2">
                            <NumberCounter label="لوح/بريكر" value={space.electrical.panel} onChange={v => updateWorkSpace(space.id, 'electrical', 'panel', v)} />
                            <NumberCounter label="مصباح مستعار" value={space.electrical.ceilingLight} onChange={v => updateWorkSpace(space.id, 'electrical', 'ceilingLight', v)} />
                            <NumberCounter label="مصباح خرساني" value={space.electrical.concreteLight} onChange={v => updateWorkSpace(space.id, 'electrical', 'concreteLight', v)} />
                            <NumberCounter label="سبوت لايت" value={space.electrical.spotlight} onChange={v => updateWorkSpace(space.id, 'electrical', 'spotlight', v)} />
                            <NumberCounter label="أفياش عادية" value={space.electrical.sockets} onChange={v => updateWorkSpace(space.id, 'electrical', 'sockets', v)} />
                            <NumberCounter label="مفتاح مزدوج" value={space.electrical.doubleSwitch} onChange={v => updateWorkSpace(space.id, 'electrical', 'doubleSwitch', v)} />
                            <NumberCounter label="مفتاح مكيف" value={space.electrical.acSwitch} onChange={v => updateWorkSpace(space.id, 'electrical', 'acSwitch', v)} />
                            <NumberCounter label="فيش سخان" value={space.electrical.heaterSocket} onChange={v => updateWorkSpace(space.id, 'electrical', 'heaterSocket', v)} />
                          </div>
                        </div>

                        <div>
                          <h4 className="text-xs font-bold text-teal-700 mb-2 border-b pb-1">أعمال السباكة</h4>
                          <div className="grid grid-cols-2 gap-2">
                            <NumberCounter label="كرسي إفرنجي" value={space.plumbing.toiletFr} onChange={v => updateWorkSpace(space.id, 'plumbing', 'toiletFr', v)} />
                            <NumberCounter label="كرسي عربي" value={space.plumbing.toiletAr} onChange={v => updateWorkSpace(space.id, 'plumbing', 'toiletAr', v)} />

                <div>
                  <h4 className="text-xs font-bold text-teal-700 mb-3 pb-1 border-b">الأثاث المطلوب</h4>
                  <div className="grid grid-cols-1 sm:grid-cols-2 gap-2">
                    <NumberCounter label="سرير نفر ونص مع مرتبة" value={data.furnitureItems.bed15} onChange={v => updateNestedData('furnitureItems', 'bed15', v)} />

                    <NumberCounter label="مراتب نفر فقط" value={data.furnitureItems.mattress1} onChange={v => updateNestedData('furnitureItems', 'mattress1', v)} />

                    <NumberCounter label="كومدينة درجين" value={data.furnitureItems.nightstand} onChange={v => updateNestedData('furnitureItems', 'nightstand', v)} />

                    <NumberCounter label="دولاب 3 أبواب" value={data.furnitureItems.wardrobe3} onChange={v => updateNestedData('furnitureItems', 'wardrobe3', v)} />

        <div className="flex justify-between items-center mt-4 pt-4 border-t border-gray-200">
           <button onClick={() => setCurrentStep(prev => Math.max(0, prev - 1))} disabled={currentStep === 0} className={`flex items-center gap-1 px-4 py-2 rounded-lg text-sm font-bold transition-colors ${currentStep === 0 ? 'text-gray-300 cursor-not-allowed' : 'text-purple-700 hover:bg-purple-100'}`}>

           {currentStep < steps.length - 1 && (

           )}
        </div>

      </main>
      
      <style dangerouslySetInnerHTML={{__html: `

        @keyframes fadeIn { from { opacity: 0; transform: translateY(5px); } to { opacity: 1; transform: translateY(0); } }
      `}} />
    </div>
  );
}
```
