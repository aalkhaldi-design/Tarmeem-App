```react
import React, { useState } from 'react';
import { 
  Building, CheckCircle2, UserCheck, History, CheckSquare, 
  Percent, FileText, ChevronDown, ChevronUp, Camera, 
  ShieldAlert, HardHat, Plus, Trash2 
} from 'lucide-react';

const Card = ({ title, icon: Icon, children, className = '' }) => (
  <div className={`bg-white rounded-xl shadow-sm border border-gray-200 overflow-hidden mb-4 transition-all duration-300 ${className}`}>
    <div className="bg-gradient-to-r from-purple-800 to-purple-600 px-4 py-2.5 flex items-center gap-2 text-white">
      {Icon && <Icon className="w-5 h-5 text-teal-200" />}
      <h3 className="text-base font-bold">{title}</h3>
    </div>
    <div className="p-4 md:p-5">{children}</div>
  </div>
);

const ReadOnlyField = ({ label, value }) => (
  <div className="flex flex-col gap-1">
    <label className="text-xs font-semibold text-gray-500">{label}</label>
    <div className="px-3 py-2 text-sm rounded-lg bg-gray-100 border border-gray-200 text-gray-700 font-medium">
      {value || '-'}
    </div>
  </div>
);

const Input = ({ label, type = 'text', value, onChange, placeholder, className = '', required = false }) => (
  <div className={`flex flex-col gap-1 ${className}`}>
    <label className="text-xs font-semibold text-gray-700">{label} {required && <span className="text-red-500">*</span>}</label>
    <input type={type} value={value} onChange={onChange} placeholder={placeholder} required={required} className="px-3 py-1.5 text-sm rounded-lg border border-gray-200 focus:border-teal-500 outline-none text-gray-800 bg-gray-50" />
  </div>
);

export default function F14_SupervisionReport({ onSubmit, f08Data = {}, previousVisits = [], projectData = {} }) {
  const [isSubmitting, setIsSubmitting] = useState(false);
  
  // Safe extraction
  const caseData = Object.keys(f08Data).length > 0 ? f08Data : {
    general: { caseRef: 'CS-9082', familyName: 'عبدالله محمد السالم', cityNeighborhood: 'الدمام - حي النور' },
    projectNumber: 'PRJ-2024-884',
    works: [
      { id: 1, name: 'الصالة الرئيسية', civil: { paint: 'دهان كامل' }, electrical: { sockets: 4 } }
    ]
  };

  const pData = Object.keys(projectData).length > 0 ? projectData : { contractor: 'مؤسسة البناء الحديث' };

  // --- F-14 Normal State ---
  const [visitData, setVisitData] = useState({
    visitDate: new Date().toISOString().split('T')[0],
    engineerName: '',
    progress: {},
    milestoneReached: '',
    recommendation: '',
    pledge: false
  });

  const [expandedSpaces, setExpandedSpaces] = useState(caseData.works?.map(w => w.id) || []);
  const [showHistory, setShowHistory] = useState(false);

  // --- F-23 Integrated State (Scope Change) ---
  const [requestScopeChange, setRequestScopeChange] = useState(false);
  const [additionalWorks, setAdditionalWorks] = useState([
    { id: 1, description: '', dimensions: '', contractorName: pData.contractor, price: '' }
  ]);
  const [scopeChangeNotes, setScopeChangeNotes] = useState('');

  // Helpers for F-14 Progress
  const toggleSpace = (id) => setExpandedSpaces(prev => prev.includes(id) ? prev.filter(s => s !== id) : [...prev, id]);
  const handlePercentageChange = (spaceId, value) => {
    setVisitData(prev => ({ ...prev, progress: { ...prev.progress, [spaceId]: { ...(prev.progress[spaceId] || { completedItems: [] }), percentage: value } }}));
  };
  const toggleItemCompletion = (spaceId, itemKey) => {
    setVisitData(prev => {
      const spaceProgress = prev.progress[spaceId] || { percentage: 0, completedItems: [] };
      const items = spaceProgress.completedItems.includes(itemKey) ? spaceProgress.completedItems.filter(i => i !== itemKey) : [...spaceProgress.completedItems, itemKey];
      return { ...prev, progress: { ...prev.progress, [spaceId]: { ...spaceProgress, completedItems: items } } };
    });
  };

  // Helpers for F-23 Scope Change
  const addWorkItem = () => setAdditionalWorks([...additionalWorks, { id: Date.now(), description: '', dimensions: '', contractorName: pData.contractor, price: '' }]);
  const removeWorkItem = (id) => setAdditionalWorks(additionalWorks.filter(w => w.id !== id));
  const updateWorkItem = (id, field, value) => setAdditionalWorks(additionalWorks.map(w => w.id === id ? { ...w, [field]: value } : w));
  const getTotalPrice = () => additionalWorks.reduce((sum, item) => sum + (Number(item.price) || 0), 0);

  const handleSubmit = async (e) => {
    e.preventDefault();
    if (!visitData.pledge || isSubmitting) return;
    setIsSubmitting(true);
    
    // Construct the combined payload
    const payload = {
      ...visitData,
      // If the engineer toggled the F-23 section, attach it to the payload for the engine to catch
      scopeChangeRequest: requestScopeChange ? { additionalWorks, engineerNotes: scopeChangeNotes } : null
    };

    if (onSubmit) await onSubmit({ formId: 'F-14', payload });
  };

  const renderDiagnosisItems = (space, category, categoryLabel) => {
    if (!space[category]) return null;
    const activeItems = Object.entries(space[category]).filter(([k, v]) => v !== '' && v !== 0 && v !== '0' && k !== 'notes');
    if (activeItems.length === 0) return null;
    const spaceProgress = visitData.progress[space.id] || { completedItems: [] };

    return (
      <div className="mb-4">
        <h5 className="text-xs font-bold text-teal-700 mb-2 border-b border-gray-200 pb-1">{categoryLabel}</h5>
        <div className="grid grid-cols-1 md:grid-cols-2 gap-3">
          {activeItems.map(([key, value]) => {
            const itemKey = `${category}_${key}`;
            const isCompleted = spaceProgress.completedItems.includes(itemKey);
            return (
              <label key={key} className={`flex items-start gap-3 p-3 rounded-lg border transition-all cursor-pointer ${isCompleted ? 'bg-teal-50 border-teal-300' : 'bg-gray-50 border-gray-200 hover:border-teal-200'}`}>
                <input type="checkbox" checked={isCompleted} onChange={() => toggleItemCompletion(space.id, itemKey)} className="w-5 h-5 mt-0.5 rounded text-teal-600" />
                <div className="flex-1">
                  <div className="text-xs font-semibold text-gray-500 mb-1">{key}</div>
                  <div className="text-sm font-bold text-gray-800 bg-gray-100 px-2 py-1 rounded inline-block">{value}</div>
                </div>
              </label>
            );
          })}
        </div>
      </div>
    );
  };

  if (isSubmitting && !onSubmit) {
    return (
       <div className="min-h-screen bg-gray-50 flex items-center justify-center p-4" dir="rtl">
         <div className="bg-teal-500 text-white rounded-xl p-8 shadow-lg text-center max-w-md w-full">
            <CheckCircle2 className="w-16 h-16 mx-auto mb-4 text-teal-100" />
            <h2 className="text-2xl font-bold mb-2">تم رفع التقرير الميداني!</h2>
            <p className="text-teal-100 text-sm leading-relaxed mb-3">تم حفظ سجل الزيارة بنجاح.</p>
            {visitData.milestoneReached && <div className="bg-white/20 p-2 rounded mb-2 text-sm">✓ تم إشعار المالية لطلب الدفعة (F-15)</div>}
            {requestScopeChange && <div className="bg-white/20 p-2 rounded text-sm">✓ تم توجيه طلب الأعمال الإضافية لمدير المشاريع (F-23)</div>}
         </div>
       </div>
    );
  }

  return (
    <div className="min-h-screen bg-gray-50 font-sans text-gray-800 pb-16" dir="rtl">
      <header className="bg-gradient-to-r from-purple-900 to-purple-700 text-white shadow-md pt-6 pb-12 px-4 text-center rounded-b-[2rem] relative z-10">
        <h1 className="text-2xl md:text-3xl font-extrabold mb-1">تقرير الزيارة الميدانية (F-14)</h1>
        <h2 className="text-sm md:text-base text-teal-300 font-medium">متابعة الإنجاز ومطابقة التشخيص</h2>
      </header>

      <main className="container mx-auto px-4 -mt-8 relative z-20 max-w-4xl">
        <form onSubmit={handleSubmit}>
          
          <Card title="بيانات المشروع" icon={Building}>
            <div className="grid grid-cols-1 md:grid-cols-4 gap-4">
              <ReadOnlyField label="رقم المشروع" value={caseData.projectNumber} />
              <ReadOnlyField label="رقم الحالة" value={caseData.general?.caseRef} />
              <ReadOnlyField label="اسم المستفيد" value={caseData.general?.familyName} />
              <ReadOnlyField label="الموقع" value={caseData.general?.cityNeighborhood} />
            </div>
          </Card>

          {/* Historical Visits Accordion */}
          <div className="mb-4">
            <button type="button" onClick={() => setShowHistory(!showHistory)} className="w-full bg-indigo-50 border border-indigo-200 text-indigo-800 px-4 py-3 rounded-xl font-bold flex items-center justify-between hover:bg-indigo-100">
              <div className="flex items-center gap-2"><History size={20}/> السجل التاريخي للزيارات السابقة</div>
              {showHistory ? <ChevronUp size={20}/> : <ChevronDown size={20}/>}
            </button>
            {showHistory && (
              <div className="mt-2 space-y-3 animate-fade-in">
                {previousVisits.length > 0 ? previousVisits.map((h, i) => (
                  <div key={i} className="bg-white p-4 rounded-xl border border-gray-200 shadow-sm flex flex-col md:flex-row gap-4 justify-between">
                    <div>
                      <div className="text-xs text-gray-500 font-bold mb-1">تاريخ الزيارة: <span className="text-gray-800">{h.date}</span></div>
                      <div className="text-xs text-gray-500 font-bold mb-2">المهندس: <span className="text-gray-800">{h.engineer}</span></div>
                      <div className="text-sm text-gray-700 bg-gray-50 p-2 rounded border border-gray-100">{h.recommendation}</div>
                    </div>
                    <div className="flex flex-col items-center justify-center bg-indigo-50 px-4 py-2 rounded-lg border border-indigo-100 min-w-[100px]">
                      <span className="text-xs font-bold text-indigo-600">الإنجاز</span>
                      <span className="text-xl font-black text-indigo-900">{h.overallProgress}%</span>
                    </div>
                  </div>
                )) : <div className="text-center text-sm text-gray-500 py-4">لا توجد زيارات سابقة مسجلة.</div>}
              </div>
            )}
          </div>

          <Card title="بيانات الزيارة الحالية" icon={CheckSquare}>
            <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
              <Input label="تاريخ الزيارة *" type="date" required value={visitData.visitDate} onChange={e => setVisitData({...visitData, visitDate: e.target.value})} />
              <Input label="اسم المهندس المشرف *" required value={visitData.engineerName} onChange={e => setVisitData({...visitData, engineerName: e.target.value})} />
            </div>
          </Card>

          {/* Works / Spaces Evaluation */}
          <div className="mb-6">
            <h3 className="text-lg font-extrabold text-gray-800 mb-3 flex items-center gap-2">
              <Percent className="text-teal-600" /> تقييم إنجاز المساحات (حسب F-08)
            </h3>
            {caseData.works?.map((space, index) => {
              const isExpanded = expandedSpaces.includes(space.id);
              const spacePercentage = visitData.progress[space.id]?.percentage || 0;
              return (
                <div key={space.id} className="bg-white rounded-xl border border-gray-200 overflow-hidden mb-3 shadow-sm">
                  <div className="bg-gray-50 border-b border-gray-200 p-3 flex items-center justify-between">
                    <button type="button" onClick={() => toggleSpace(space.id)} className="flex items-center gap-3 font-bold text-gray-800">
                      {isExpanded ? <ChevronUp size={20} className="text-gray-500"/> : <ChevronDown size={20} className="text-gray-500"/>}
                      <span>{index + 1}. {space.name}</span>
                    </button>
                    <div className="flex items-center gap-3 bg-white px-3 py-1.5 rounded-lg border border-gray-200 shadow-sm w-1/2 md:w-1/3">
                      <span className="text-xs font-bold text-teal-700 w-10">{spacePercentage}%</span>
                      <input type="range" min="0" max="100" step="5" value={spacePercentage} onChange={(e) => handlePercentageChange(space.id, e.target.value)} className="w-full accent-teal-600 h-2 bg-gray-200 rounded-lg appearance-none cursor-pointer" />
                    </div>
                  </div>
                  {isExpanded && (
                    <div className="p-4 bg-white">
                      {renderDiagnosisItems(space, 'civil', 'الأعمال المدنية والتشطيبات')}
                      {renderDiagnosisItems(space, 'electrical', 'أعمال الكهرباء')}
                      {renderDiagnosisItems(space, 'plumbing', 'أعمال السباكة')}
                      <div className="mt-4 border-t border-gray-100 pt-4">
                         <label className="cursor-pointer bg-gray-50 border border-dashed border-gray-300 text-gray-600 px-4 py-3 rounded-lg text-sm font-bold flex items-center justify-center gap-2 hover:bg-gray-100">
                           <Camera size={18} /> التقاط / إرفاق صور إنجاز ({space.name}) <input type="file" multiple accept="image/*" className="hidden" />
                         </label>
                      </div>
                    </div>
                  )}
                </div>
              );
            })}
          </div>

          {/* ----- F-23 INTEGRATION (SCOPE CHANGE ACCORDION) ----- */}
          <div className="mb-6 border-2 border-teal-500 rounded-xl overflow-hidden shadow-sm">
            <div className={`p-4 flex items-center gap-3 ${requestScopeChange ? 'bg-teal-50 border-b border-teal-200' : 'bg-white'}`}>
              <input 
                type="checkbox" 
                id="scopeChange" 
                checked={requestScopeChange} 
                onChange={e => setRequestScopeChange(e.target.checked)} 
                className="w-5 h-5 text-teal-600 rounded cursor-pointer" 
              />
              <label htmlFor="scopeChange" className="cursor-pointer font-bold text-teal-800 flex items-center gap-2">
                <HardHat size={20} />
                تفعيل طلب اعتماد بنود أعمال إضافية للمقاول (يُولد نموذج F-23 آلياً)
              </label>
            </div>
            
            {requestScopeChange && (
              <div className="p-4 bg-white animate-fade-in">
                <div className="bg-blue-50 border border-blue-200 p-3 rounded-lg text-sm text-blue-800 mb-4 font-medium">
                  بتفعيلك لهذا القسم، سيقوم النظام تلقائياً بإنشاء طلب استثنائي (F-23) وإرساله لمدير المشاريع للموافقة عليه فور رفعك لهذا التقرير.
                </div>
                
                <div className="space-y-4 mb-4">
                  {additionalWorks.map((work, index) => (
                    <div key={work.id} className="p-4 bg-gray-50 border border-gray-200 rounded-xl relative">
                      {additionalWorks.length > 1 && (
                        <button type="button" onClick={() => removeWorkItem(work.id)} className="absolute top-3 left-3 text-red-500 hover:bg-red-100 p-1.5 rounded-lg"><Trash2 size={18}/></button>
                      )}
                      <h4 className="text-xs font-bold text-teal-700 mb-3">البند الإضافي ({index + 1})</h4>
                      <div className="grid grid-cols-1 md:grid-cols-2 gap-3 mb-3">
                        <Input label="وصف العمل المطلوب" required value={work.description} onChange={e => updateWorkItem(work.id, 'description', e.target.value)} className="md:col-span-2" />
                        <Input label="المساحة / الأبعاد" required value={work.dimensions} onChange={e => updateWorkItem(work.id, 'dimensions', e.target.value)} />
                        <Input label="اسم المقاول (الافتراضي هو مقاول المشروع)" required value={work.contractorName} onChange={e => updateWorkItem(work.id, 'contractorName', e.target.value)} />
                      </div>
                      <div className="w-full md:w-1/2">
                        <Input label="السعر التقديري (ر.س)" type="number" required value={work.price} onChange={e => updateWorkItem(work.id, 'price', e.target.value)} />
                      </div>
                    </div>
                  ))}
                </div>
                
                <div className="flex flex-col md:flex-row items-center justify-between gap-4 border-t border-gray-100 pt-4 mb-4">
                  <button type="button" onClick={addWorkItem} className="bg-teal-50 text-teal-700 border border-dashed border-teal-300 px-4 py-2.5 rounded-lg text-sm font-bold flex items-center gap-2 hover:bg-teal-100 w-full md:w-auto">
                    <Plus size={18} /> إضافة بند عمل آخر
                  </button>
                  <div className="text-lg font-bold text-gray-800 bg-gray-100 px-4 py-2 rounded-lg border border-gray-200">
                    إجمالي التكلفة الإضافية: <span className="text-teal-700 font-black">{getTotalPrice().toLocaleString()} ر.س</span>
                  </div>
                </div>

                <div className="flex flex-col gap-1">
                  <label className="text-xs font-semibold text-gray-700">المبررات الفنية لطلب هذه الأعمال الإضافية *</label>
                  <textarea required={requestScopeChange} rows={3} value={scopeChangeNotes} onChange={e => setScopeChangeNotes(e.target.value)} placeholder="اكتب الأسباب الفنية التي استدعت طلب هذه الزيادة..." className="px-3 py-2 text-sm rounded-lg border border-gray-300 focus:border-teal-500 outline-none resize-none bg-white" />
                </div>
              </div>
            )}
          </div>
          {/* ----- END F-23 INTEGRATION ----- */}

          <Card title="استحقاق الدفعات والاعتماد" icon={FileText}>
            <div className="bg-yellow-50 border border-yellow-200 p-4 rounded-xl mb-5">
              <label className="block text-sm font-bold text-yellow-900 mb-3">هل بلغت نسبة الإنجاز مرحلة استحقاق دفعة للمقاول؟</label>
              <select value={visitData.milestoneReached} onChange={e => setVisitData({...visitData, milestoneReached: e.target.value})} className="w-full p-3 rounded-lg border border-yellow-300 focus:border-yellow-600 outline-none text-yellow-900 font-bold bg-white">
                <option value="">لا، مجرد زيارة تفقدية</option>
                <option value="40%">نعم، تم إنجاز 40% (توليد F-15 للدفعة الثانية)</option>
                <option value="80%">نعم، تم إنجاز 80% (توليد F-15 للدفعة الثالثة)</option>
                <option value="100%">نعم، تم إنجاز 100% (تجهيز للدفعة الأخيرة)</option>
              </select>
            </div>

            <div className="flex flex-col gap-1 mb-4">
              <label className="text-xs font-semibold text-gray-700">توصية المهندس المشرف / الملاحظات الميدانية *</label>
              <textarea required rows={4} value={visitData.recommendation} onChange={e => setVisitData({...visitData, recommendation: e.target.value})} className="px-3 py-2 text-sm rounded-lg border border-gray-300 focus:border-teal-500 outline-none resize-none bg-white" />
            </div>

            <div className={`p-4 rounded-xl border transition-all ${visitData.pledge ? 'bg-teal-50 border-teal-500' : 'bg-gray-50 border-gray-200'}`}>
              <div className="flex items-start gap-4">
                <input type="checkbox" id="pledge" required checked={visitData.pledge} onChange={(e) => setVisitData({...visitData, pledge: e.target.checked})} className="w-5 h-5 mt-1 rounded text-teal-600 cursor-pointer" />
                <label htmlFor="pledge" className="cursor-pointer">
                  <h4 className="text-sm font-bold text-gray-900 mb-1">توثيق التقرير الميداني</h4>
                  <p className="text-xs text-gray-600 font-medium">أقر بصفتي المهندس المشرف بمطابقة النسب أعلاه للواقع الميداني الفعلي.</p>
                </label>
              </div>
            </div>
          </Card>

          <button type="submit" disabled={!visitData.pledge || isSubmitting} className={`w-full py-4 rounded-xl text-base font-bold flex items-center justify-center gap-2 transition-all shadow-md ${visitData.pledge ? 'bg-gradient-to-r from-teal-600 to-teal-500 text-white' : 'bg-gray-200 text-gray-400'}`}>
            <ShieldAlert className="w-5 h-5" /> {isSubmitting ? 'جاري الرفع...' : 'حفظ التقرير وإرساله'}
          </button>
        </form>
      </main>
      <style dangerouslySetInnerHTML={{__html: `.animate-fade-in{animation:fadeIn .4s ease-in-out}@keyframes fadeIn{from{opacity:0;transform:translateY(10px)}to{opacity:1;transform:translateY(0)}}`}} />
    </div>
  );
}
```
