```react
import React, { useState } from 'react';
import { 
  Home, Calendar, AlertTriangle, FileSignature, CheckCircle2, UserCheck, ShieldAlert, UploadCloud, X
} from 'lucide-react';

const Card = ({ title, icon: Icon, children, className = '' }) => (
  <div className={`bg-white rounded-xl shadow-sm border border-gray-200 overflow-hidden mb-4 transition-all duration-300 ${className}`}>
    <div className="bg-gradient-to-r from-purple-800 to-purple-600 px-4 py-2.5 flex items-center gap-2 text-white">
      {Icon && <Icon className="w-5 h-5 text-teal-200" />}
      <h3 className="text-base font-bold">{title}</h3>
    </div>
    <div className="p-4 md:p-5">{children}</div>
  </div>
);

const ReadOnlyField = ({ label, value }) => (
  <div className="flex flex-col gap-1">
    <label className="text-xs font-semibold text-gray-500">{label}</label>
    <div className="px-3 py-2 text-sm rounded-lg bg-gray-100 border border-gray-200 text-gray-700 font-medium">
      {value || '-'}
    </div>
  </div>
);

const Input = ({ label, type = 'text', value, onChange, placeholder, className = '', required = false }) => (
  <div className={`flex flex-col gap-1 ${className}`}>
    <label className="text-xs font-semibold text-gray-700">{label} {required && <span className="text-red-500">*</span>}</label>
    <input type={type} value={value} onChange={onChange} placeholder={placeholder} required={required} className="px-3 py-1.5 text-sm rounded-lg border border-gray-200 focus:border-teal-500 focus:ring-1 focus:ring-teal-200 outline-none text-gray-800 bg-gray-50" />
  </div>
);

export default function F18_EvacuationPledge({ onSubmit, initialData = {} }) {
  const [isSubmitting, setIsSubmitting] = useState(false);
  const [pledgeFile, setPledgeFile] = useState(null);
  
  // Safe extraction from previous forms
  const caseData = Object.keys(initialData).length > 0 ? initialData : {
    caseRef: 'CS-9082',
    personal: { fullName: 'عبدالله محمد السالم', city: 'الدمام', neighborhood: 'حي النور' }
  };

  const [data, setData] = useState({
    renovationStartDate: '',
    renovationEndDate: '',
    evacuationDate: '',
    researcherName: '',
    familyRepName: caseData.personal?.fullName || '',
    employeePledge: false
  });

  const handleFileChange = (e) => {
    if (e.target.files && e.target.files[0]) {
      setPledgeFile({
        file: e.target.files[0],
        name: e.target.files[0].name,
        previewUrl: URL.createObjectURL(e.target.files[0])
      });
    }
  };

  const removeFile = () => {
    if (pledgeFile) URL.revokeObjectURL(pledgeFile.previewUrl);
    setPledgeFile(null);
  };

  const handleSubmit = async (e) => {
    e.preventDefault();
    if (!data.employeePledge || !pledgeFile) return;
    setIsSubmitting(true);
    
    if (onSubmit) {
      await onSubmit({ 
        formId: 'F-18', 
        payload: { ...data, uploadedDocument: pledgeFile } 
      });
    }
  };

  if (isSubmitting && !onSubmit) {
    return (
       <div className="min-h-screen bg-gray-50 flex items-center justify-center p-4" dir="rtl">
         <div className="bg-teal-500 text-white rounded-xl p-8 shadow-lg text-center max-w-md w-full">
            <CheckCircle2 className="w-16 h-16 mx-auto mb-4 text-teal-100" />
            <h2 className="text-2xl font-bold mb-2">تم أتمتة التعهد بنجاح!</h2>
            <p className="text-teal-100 text-sm">تم رفع المستند الورقي وتوثيق بيانات الإخلاء في النظام، ويمكن الآن البدء بإجراءات تسليم المنزل للمقاول.</p>
         </div>
       </div>
    );
  }

  return (
    <div className="min-h-screen bg-gray-50 font-sans text-gray-800 pb-16" dir="rtl">
      <header className="bg-gradient-to-r from-purple-900 to-purple-700 text-white shadow-md pt-6 pb-12 px-4 text-center rounded-b-[2rem] relative z-10">
        <h1 className="text-2xl md:text-3xl font-extrabold mb-1">أتمتة تعهد إخلاء المنزل (F-18)</h1>
        <h2 className="text-sm md:text-base text-teal-300 font-medium">يُعبأ بواسطة الموظف نيابة عن الأسرة بعد التوقيع الورقي</h2>
      </header>

      <main className="container mx-auto px-4 -mt-8 relative z-20 max-w-3xl">
        <form onSubmit={handleSubmit}>
          
          {/* Read Only Data from System */}
          <Card title="بيانات المستفيد المرجعية" icon={Home}>
            <div className="grid grid-cols-1 md:grid-cols-3 gap-4">
              <ReadOnlyField label="رقم المستفيد (الحالة)" value={caseData.caseRef} />
              <ReadOnlyField label="المدينة" value={caseData.personal?.city} />
              <ReadOnlyField label="الحي" value={caseData.personal?.neighborhood} />
            </div>
          </Card>

          {/* Dates & Timeline */}
          <Card title="تواريخ التنفيذ والإخلاء المُتفق عليها" icon={Calendar}>
            <div className="bg-blue-50 border border-blue-100 p-4 rounded-xl mb-5 text-sm text-blue-800 leading-relaxed font-medium text-justify">
              تُسجل هنا التواريخ المطابقة للنموذج الورقي الذي تم إبلاغ الأسرة به، بأن المقاول سيبدأ أعمال الترميم 
              وسيتم الانتهاء في التاريخ المحدد، وأنه يجب إخلاء المنزل من الأثاث والأجهزة.
            </div>

            <div className="grid grid-cols-1 md:grid-cols-2 gap-5">
              <Input label="تاريخ بدء أعمال الترميم المتوقع" type="date" required value={data.renovationStartDate} onChange={e => setData({...data, renovationStartDate: e.target.value})} />
              <Input label="تاريخ الانتهاء المتوقع" type="date" required value={data.renovationEndDate} onChange={e => setData({...data, renovationEndDate: e.target.value})} />
              <div className="md:col-span-2">
                <Input label="تاريخ إخلاء المنزل الفعلي (المُلزم للأسرة)" type="date" required value={data.evacuationDate} onChange={e => setData({...data, evacuationDate: e.target.value})} />
              </div>
            </div>
          </Card>

          {/* File Upload for the signed paper */}
          <Card title="إرفاق التعهد الورقي" icon={UploadCloud}>
            <div className="bg-purple-50 p-6 rounded-xl border-2 border-dashed border-purple-200 text-center hover:bg-purple-100 transition-colors">
              <UploadCloud className="w-10 h-10 text-purple-500 mx-auto mb-2" />
              <p className="text-sm font-bold text-purple-900 mb-1">إرفاق نسخة التعهد الموقعة *</p>
              <p className="text-xs text-purple-600 mb-4">يجب إرفاق صورة أو ملف PDF للنموذج الورقي الموقع من ممثل الأسرة</p>
              <label className="cursor-pointer bg-purple-600 text-white px-6 py-2.5 rounded-lg text-sm font-bold hover:bg-purple-700 shadow-sm">
                تصفح الملفات
                <input type="file" required={!pledgeFile} accept="image/*,.pdf" className="hidden" onChange={handleFileChange} />
              </label>
            </div>
            {pledgeFile && (
              <div className="mt-4 flex justify-between items-center bg-white p-3 rounded-lg border border-teal-200 text-sm shadow-sm bg-teal-50">
                <div className="flex items-center gap-2 text-teal-800 font-bold">
                  <CheckCircle2 size={18} />
                  <span className="truncate max-w-[200px]">{pledgeFile.name}</span>
                </div>
                <button type="button" onClick={removeFile} className="text-red-500 hover:bg-red-50 p-1 rounded"><X size={18}/></button>
              </div>
            )}
          </Card>

          {/* Employee Signatures & Pledge */}
          <Card title="توثيق الموظف والاعتماد" icon={ShieldAlert}>
            <div className="grid grid-cols-1 md:grid-cols-2 gap-5 mb-6 border-b border-gray-100 pb-6">
              <Input label="اسم الباحث/ة (أو المهندس المشرف)" required value={data.researcherName} onChange={e => setData({...data, researcherName: e.target.value})} placeholder="الاسم الرباعي للموظف المسؤول..." />
              <Input label="اسم ممثل الأسرة المُوقِّع" required value={data.familyRepName} onChange={e => setData({...data, familyRepName: e.target.value})} />
            </div>

            <div className={`p-5 rounded-xl border-2 transition-all ${data.employeePledge ? 'bg-teal-50 border-teal-500' : 'bg-gray-50 border-gray-200'}`}>
              <div className="flex items-start gap-4">
                <input 
                  type="checkbox" 
                  id="employeePledge"
                  required
                  checked={data.employeePledge}
                  onChange={(e) => setData({...data, employeePledge: e.target.checked})}
                  className="w-6 h-6 mt-1 rounded text-teal-600 focus:ring-teal-500 cursor-pointer"
                />
                <label htmlFor="employeePledge" className="cursor-pointer select-none">
                  <h4 className="text-base font-bold text-gray-900 mb-2 flex items-center gap-2">
                    <FileSignature className="w-5 h-5 text-teal-600" />
                    إقرار الموظف المسؤول
                  </h4>
                  <p className="text-sm text-gray-700 leading-loose text-justify font-medium">
                    أقر أنا الموظف المسؤول الموضح اسمي أعلاه، بأن ممثل الأسرة قد استوعب ووقّع على التعهد الورقي المرفق، 
                    والذي يفيد بالتزامهم بإخلاء المنزل في التاريخ المحدد ( {data.evacuationDate || '---'} ) وعدم التدخل في عمل المقاول. 
                    وعلى ذلك أقوم برقمنة التعهد واعتماده في النظام.
                  </p>
                </label>
              </div>
            </div>
          </Card>

          {/* Submit */}
          <button 
            type="submit"
            disabled={!data.employeePledge || !pledgeFile || isSubmitting}
            className={`w-full py-4 rounded-xl text-base font-bold flex items-center justify-center gap-2 transition-all shadow-md
              ${(data.employeePledge && pledgeFile)
                ? 'bg-gradient-to-r from-teal-600 to-teal-500 hover:from-teal-700 text-white cursor-pointer' 
                : 'bg-gray-200 text-gray-400 cursor-not-allowed'}`}
          >
            <UserCheck className="w-5 h-5" /> 
            {isSubmitting ? 'جاري توثيق التعهد...' : 'رفع المستند واعتماد التعهد رقمياً'}
          </button>
        </form>
      </main>
    </div>
  );
}


```
