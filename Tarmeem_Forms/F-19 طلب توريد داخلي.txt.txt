```react
import React, { useState } from 'react';
import { 
  ShoppingCart, Building, CheckCircle2, UserCheck, ShieldAlert, 
  FileText, Plus, Trash2, ArrowLeftRight, Clock, XCircle
} from 'lucide-react';

const Card = ({ title, icon: Icon, children, className = '' }) => (
  <div className={`bg-white rounded-xl shadow-sm border border-gray-200 overflow-hidden mb-4 transition-all duration-300 ${className}`}>
    <div className="bg-gradient-to-r from-purple-800 to-purple-600 px-4 py-2.5 flex items-center gap-2 text-white">
      {Icon && <Icon className="w-5 h-5 text-teal-200" />}
      <h3 className="text-base font-bold">{title}</h3>
    </div>
    <div className="p-4 md:p-5">{children}</div>
  </div>
);

const ReadOnlyField = ({ label, value, highlight = false }) => (
  <div className="flex flex-col gap-1">
    <label className="text-xs font-semibold text-gray-500">{label}</label>
    <div className={`px-3 py-2 text-sm rounded-lg border font-medium
      ${highlight ? 'bg-teal-50 border-teal-200 text-teal-800' : 'bg-gray-100 border-gray-200 text-gray-700'}`}>
      {value || '-'}
    </div>
  </div>
);

const Input = ({ label, type = 'text', value, onChange, placeholder, className = '', required = false }) => (
  <div className={`flex flex-col gap-1 ${className}`}>
    <label className="text-xs font-semibold text-gray-700">{label} {required && <span className="text-red-500">*</span>}</label>
    <input type={type} value={value} onChange={onChange} placeholder={placeholder} required={required} className="px-3 py-1.5 text-sm rounded-lg border border-gray-200 focus:border-purple-500 outline-none text-gray-800 bg-gray-50" />
  </div>
);

export default function F19_InternalPurchaseRequest({ 
  onSubmit, 
  projectData = {}, 
  // PROCUREMENT_ENTRY, SS_MANAGER_APPROVAL, FINANCE_APPROVAL
  workflowStep = 'PROCUREMENT_ENTRY', 
  existingData = {} 
}) {
  const [isSubmitting, setIsSubmitting] = useState(false);
  
  // Beneficiary & Project Data
  const pData = Object.keys(projectData).length > 0 ? projectData : {
    caseRef: 'CS-9082', familyName: 'عبدالله محمد السالم', projectNumber: 'PRJ-2024-884', contractor: 'مؤسسة البناء الحديث'
  };

  // State for Materials Table (Merged from the two paper forms)
  const [materials, setMaterials] = useState(existingData.materials || [
    { id: 1, category: 'الكهرباء', name: '', unit: 'حبة', quantity: '', estimatedPrice: '' }
  ]);
  
  // Procurement Strategy & Timeline
  const [procurementMethod, setProcurementMethod] = useState(existingData.procurementMethod || 'شراء مباشر');
  const [targetDate, setTargetDate] = useState(existingData.targetDate || '');
  
  // Notes State
  const [procurementNotes, setProcurementNotes] = useState(existingData.procurementNotes || '');
  const [ssManagerNotes, setSsManagerNotes] = useState(existingData.ssManagerNotes || '');
  const [financeNotes, setFinanceNotes] = useState(existingData.financeNotes || '');
  
  const [pledge, setPledge] = useState(false);

  const isReadOnly = workflowStep !== 'PROCUREMENT_ENTRY';

  const addMaterial = () => {
    setMaterials([...materials, { id: Date.now(), category: 'أخرى', name: '', unit: 'حبة', quantity: '', estimatedPrice: '' }]);
  };

  const removeMaterial = (id) => {
    setMaterials(materials.filter(m => m.id !== id));
  };

  const updateMaterial = (id, field, value) => {
    setMaterials(materials.map(m => m.id === id ? { ...m, [field]: value } : m));
  };

  const getTotalEstimatedPrice = () => {
    return materials.reduce((sum, item) => sum + ((Number(item.quantity) || 0) * (Number(item.estimatedPrice) || 0)), 0);
  };

  const handleAction = async (actionType) => {
    if (!pledge || isSubmitting) return;
    setIsSubmitting(true);
    
    let payload = { ...existingData };

    // State Machine Routing
    if (workflowStep === 'PROCUREMENT_ENTRY') {
      payload = { 
        ...payload, 
        projectInfo: pData,
        materials, 
        procurementMethod,
        targetDate,
        procurementNotes, 
        totalEstimatedPrice: getTotalEstimatedPrice(),
        status: 'PENDING_SS_MANAGER' 
      };
    } else if (workflowStep === 'SS_MANAGER_APPROVAL') {
      payload = { 
        ...payload, 
        // Manager can override the procurement method
        procurementMethod,
        ssManagerNotes, 
        status: actionType === 'APPROVE' ? 'PENDING_FINANCE' : 'REJECTED' 
      };
    } else if (workflowStep === 'FINANCE_APPROVAL') {
      payload = { 
        ...payload, 
        financeNotes, 
        status: actionType === 'APPROVE' ? 'APPROVED' : 'REJECTED' 
      };
    }

    if (onSubmit) {
      await onSubmit({ formId: 'F-19', action: actionType, payload });
    }
  };

  if (isSubmitting && !onSubmit) {
    return (
       <div className="min-h-screen bg-gray-50 flex items-center justify-center p-4" dir="rtl">
         <div className="bg-teal-500 text-white rounded-xl p-8 shadow-lg text-center max-w-md w-full">
            <CheckCircle2 className="w-16 h-16 mx-auto mb-4 text-teal-100" />
            <h2 className="text-2xl font-bold mb-2">تم رفع الطلب بنجاح!</h2>
            <p className="text-teal-100 text-sm leading-relaxed">تم توجيه طلب المشتريات للخطوة التالية في النظام للاعتماد.</p>
         </div>
       </div>
    );
  }

  return (
    <div className="min-h-screen bg-gray-50 font-sans text-gray-800 pb-16" dir="rtl">
      <header className="bg-gradient-to-r from-purple-900 to-purple-700 text-white shadow-md pt-6 pb-12 px-4 text-center rounded-b-[2rem] relative z-10">
        <h1 className="text-2xl md:text-3xl font-extrabold mb-1">طلب توريد / مشتريات مواد (F-19)</h1>
        <h2 className="text-sm md:text-base text-teal-300 font-medium">
          {workflowStep === 'PROCUREMENT_ENTRY' && 'المرحلة 1: المشتريات والخدمات المساندة (تحديد الكميات وطريقة التوريد)'}
          {workflowStep === 'SS_MANAGER_APPROVAL' && 'المرحلة 2: مدير الخدمات المساندة (مراجعة واعتماد طريقة التوريد)'}
          {workflowStep === 'FINANCE_APPROVAL' && 'المرحلة 3: الإدارة المالية (الاعتماد المالي وتأكيد الميزانية)'}
        </h2>
      </header>

      <main className="container mx-auto px-4 -mt-8 relative z-20 max-w-5xl">
        
        {/* Project Info */}
        <Card title="بيانات المشروع" icon={Building}>
          <div className="grid grid-cols-1 md:grid-cols-4 gap-4">
            <ReadOnlyField label="رقم المشروع" value={pData.projectNumber} highlight />
            <ReadOnlyField label="اسم المستفيد" value={pData.familyName} />
            <ReadOnlyField label="رقم المنزل" value={pData.caseRef} />
            <ReadOnlyField label="المقاول الأساسي للمشروع" value={pData.contractor} />
          </div>
        </Card>

        {/* Dynamic Materials List (Combines logic from both paper forms) */}
        <Card title="حصر أنواع وكميات المواد المطلوبة" icon={ShoppingCart}>
          {isReadOnly && workflowStep !== 'SS_MANAGER_APPROVAL' ? (
            <div className="overflow-x-auto">
              <table className="w-full text-right border-collapse mb-4">
                <thead>
                  <tr className="bg-teal-600 text-white text-sm">
                    <th className="p-3 border border-teal-700 font-bold rounded-tr-lg">الاحتياج (القسم)</th>
                    <th className="p-3 border border-teal-700 font-bold">النوع (وصف المادة)</th>
                    <th className="p-3 border border-teal-700 font-bold text-center">الوحدة</th>
                    <th className="p-3 border border-teal-700 font-bold text-center">الكمية</th>
                    <th className="p-3 border border-teal-700 font-bold rounded-tl-lg">التكلفة التقديرية</th>
                  </tr>
                </thead>
                <tbody>
                  {materials.map((m, index) => (
                    <tr key={m.id} className="bg-gray-50 border-b border-gray-200 text-sm">
                      <td className="p-3 border-x border-gray-200 font-bold text-teal-800">{m.category}</td>
                      <td className="p-3 border-x border-gray-200 font-medium">{m.name}</td>
                      <td className="p-3 border-x border-gray-200 text-center">{m.unit}</td>
                      <td className="p-3 border-x border-gray-200 text-center font-bold">{m.quantity}</td>
                      <td className="p-3 border-x border-gray-200 font-bold text-teal-700">{(m.quantity * m.estimatedPrice).toLocaleString()} ر.س</td>
                    </tr>
                  ))}
                </tbody>
                <tfoot>
                  <tr className="bg-teal-50">
                    <td colSpan="4" className="p-3 border border-gray-200 font-bold text-left">التكلفة التقديرية الإجمالية:</td>
                    <td className="p-3 border border-gray-200 font-black text-lg text-teal-800">{existingData.totalEstimatedPrice?.toLocaleString()} ر.س</td>
                  </tr>
                </tfoot>
              </table>
            </div>
          ) : (
            <>
              <div className="overflow-x-auto mb-4">
                <table className="w-full text-right border-collapse min-w-[700px]">
                  <thead>
                    <tr className="bg-purple-100 text-purple-900 text-sm">
                      <th className="p-2 border border-purple-200 font-bold w-1/5 rounded-tr-lg">الاحتياج (القسم)</th>
                      <th className="p-2 border border-purple-200 font-bold w-1/3">النوع (وصف المادة المطلوبة)</th>
                      <th className="p-2 border border-purple-200 font-bold w-24">الوحدة</th>
                      <th className="p-2 border border-purple-200 font-bold w-24">الكمية</th>
                      <th className="p-2 border border-purple-200 font-bold w-32">السعر الإفرادي</th>
                      <th className="p-2 border border-purple-200 font-bold rounded-tl-lg w-10"></th>
                    </tr>
                  </thead>
                  <tbody>
                    {materials.map((m) => (
                      <tr key={m.id} className="bg-white border-b border-gray-200">
                        <td className="p-1 border-x border-gray-200">
                          <select disabled={isReadOnly} value={m.category} onChange={e => updateMaterial(m.id, 'category', e.target.value)} className="w-full px-2 py-1.5 text-sm rounded border-gray-300 outline-none bg-gray-50 focus:border-purple-500">
                            <option value="الأدوات الصحية">الأدوات الصحية</option>
                            <option value="الكهرباء">الكهرباء</option>
                            <option value="مواد البناء">مواد البناء</option>
                            <option value="الدهانات">الدهانات</option>
                            <option value="أخرى">أخرى</option>
                          </select>
                        </td>
                        <td className="p-1 border-x border-gray-200"><input disabled={isReadOnly} type="text" required value={m.name} onChange={e => updateMaterial(m.id, 'name', e.target.value)} className="w-full px-2 py-1.5 text-sm rounded border-gray-300 outline-none focus:border-purple-500 bg-white" placeholder="مثال: أسلاك، أسمنت..." /></td>
                        <td className="p-1 border-x border-gray-200">
                          <select disabled={isReadOnly} value={m.unit} onChange={e => updateMaterial(m.id, 'unit', e.target.value)} className="w-full px-2 py-1.5 text-sm rounded border-gray-300 outline-none bg-gray-50 focus:border-purple-500">
                            <option value="حبة">حبة</option><option value="متر">متر</option><option value="كيس">كيس</option><option value="لتر">لتر</option><option value="مقطوعية">مقطوعية</option>
                          </select>
                        </td>
                        <td className="p-1 border-x border-gray-200"><input disabled={isReadOnly} type="number" min="1" required value={m.quantity} onChange={e => updateMaterial(m.id, 'quantity', e.target.value)} className="w-full px-2 py-1.5 text-sm rounded border-gray-300 outline-none text-center bg-white" /></td>
                        <td className="p-1 border-x border-gray-200"><input disabled={isReadOnly} type="number" min="0" required value={m.estimatedPrice} onChange={e => updateMaterial(m.id, 'estimatedPrice', e.target.value)} className="w-full px-2 py-1.5 text-sm rounded border-gray-300 outline-none text-center bg-white" placeholder="تقديري" /></td>
                        <td className="p-1 text-center border-x border-gray-200">
                          {materials.length > 1 && !isReadOnly && (
                            <button type="button" onClick={() => removeMaterial(m.id)} className="text-red-500 hover:bg-red-100 p-1 rounded"><Trash2 size={18}/></button>
                          )}
                        </td>
                      </tr>
                    ))}
                  </tbody>
                </table>
              </div>
              
              {!isReadOnly && (
                <div className="flex flex-col md:flex-row items-center justify-between gap-4 border-t border-gray-100 pt-4">
                  <button type="button" onClick={addMaterial} className="bg-purple-50 text-purple-700 border border-dashed border-purple-300 px-4 py-2.5 rounded-lg text-sm font-bold flex items-center gap-2 hover:bg-purple-100 transition-colors w-full md:w-auto">
                    <Plus size={18} /> إضافة صنف جديد للطلب
                  </button>
                  <div className="text-lg font-bold text-gray-800 bg-gray-100 px-4 py-2 rounded-lg border border-gray-200 shadow-inner">
                    الميزانية التقديرية للطلب: <span className="text-teal-700 font-black">{getTotalEstimatedPrice().toLocaleString()} ر.س</span>
                  </div>
                </div>
              )}
            </>
          )}
        </Card>

        {/* Procurement Strategy & Timeline */}
        <Card title="آلية التنفيذ والتوريد المقترحة" icon={Clock}>
          <div className="grid grid-cols-1 md:grid-cols-2 gap-5 mb-4">
            
            <div className="bg-blue-50 border border-blue-200 p-4 rounded-xl">
              <label className="block text-sm font-bold text-blue-900 mb-3">طريقة التوريد الموصى بها:</label>
              <select 
                disabled={workflowStep === 'FINANCE_APPROVAL'}
                value={procurementMethod}
                onChange={e => setProcurementMethod(e.target.value)}
                className={`w-full p-2.5 rounded-lg border focus:border-blue-600 outline-none font-bold shadow-sm
                  ${procurementMethod === 'تعميد المقاول بالتوريد' ? 'bg-red-50 border-red-300 text-red-800' : 'bg-white border-blue-300 text-blue-900'}`}
              >
                <option value="شراء مباشر">شراء مباشر (عبر الخدمات المساندة)</option>
                <option value="تعميد المقاول بالتوريد">تعميد المقاول بالتوريد (يُخصم/يُضاف لمستخلصه)</option>
                <option value="دعم عيني / شراكات">توفير عبر داعم عيني / شراكات</option>
              </select>
              {procurementMethod === 'تعميد المقاول بالتوريد' && (
                <p className="text-xs text-red-600 mt-2 font-bold animate-fade-in">
                  * تحذير: اختيار هذا البند سيُصدر أمر تفويض للمقاول بشراء هذه المواد واعتماد المبالغ في مستخلصه.
                </p>
              )}
            </div>

            <div className="bg-gray-50 border border-gray-200 p-4 rounded-xl">
              <Input 
                label="تاريخ التوريد المستهدف للبدء بالعمل *" 
                type="date" 
                required 
                disabled={isReadOnly}
                value={targetDate} 
                onChange={e => setTargetDate(e.target.value)} 
              />
              <p className="text-xs text-gray-500 mt-2 leading-relaxed">يُحدد هذا التاريخ لتجنب تأخير جدول المقاول الزمني.</p>
            </div>
          </div>
        </Card>

        {/* Historical Opinions (Read Only for downstream steps) */}
        {workflowStep !== 'PROCUREMENT_ENTRY' && (
          <Card title="الآراء والتوصيات السابقة" icon={FileText}>
            <div className="space-y-3">
              <ReadOnlyField label="مبررات الطلب (موظف المشتريات)" value={existingData.procurementNotes} />
              {workflowStep === 'FINANCE_APPROVAL' && <ReadOnlyField label="رأي واعتماد مدير الخدمات المساندة" value={existingData.ssManagerNotes} />}
            </div>
          </Card>
        )}

        {/* Current Actor Action Box */}
        <Card title="الإجراء والاعتماد الحالي" icon={ShieldAlert}>
          <div className="bg-gray-50 border border-gray-200 p-4 rounded-xl mb-5">
            <div className="flex flex-col gap-1 mb-4">
              <label className="text-xs font-bold text-gray-700">
                {workflowStep === 'PROCUREMENT_ENTRY' && 'المبررات والتوجيه (موظف المشتريات) *'}
                {workflowStep === 'SS_MANAGER_APPROVAL' && 'ملاحظات وتوجيه مدير الخدمات المساندة *'}
                {workflowStep === 'FINANCE_APPROVAL' && 'ملاحظات المحاسب (التدقيق المالي والميزانية) *'}
              </label>
              <textarea 
                required rows={3} 
                value={
                  workflowStep === 'PROCUREMENT_ENTRY' ? procurementNotes :
                  workflowStep === 'SS_MANAGER_APPROVAL' ? ssManagerNotes : financeNotes
                }
                onChange={e => {
                  const val = e.target.value;
                  if (workflowStep === 'PROCUREMENT_ENTRY') setProcurementNotes(val);
                  else if (workflowStep === 'SS_MANAGER_APPROVAL') setSsManagerNotes(val);
                  else setFinanceNotes(val);
                }}
                placeholder="اكتب التوصية هنا قبل الاعتماد..." 
                className="px-3 py-2 text-sm rounded-lg border border-gray-300 focus:border-purple-500 outline-none resize-none bg-white shadow-sm" 
              />
            </div>

            <div className="flex items-start gap-4 p-3 bg-white rounded-lg border border-gray-200">
              <input type="checkbox" id="pledge" checked={pledge} onChange={(e) => setPledge(e.target.checked)} className="w-5 h-5 mt-0.5 rounded text-purple-600 focus:ring-purple-500 cursor-pointer" />
              <label htmlFor="pledge" className="cursor-pointer select-none">
                <h4 className="text-sm font-bold text-gray-900 mb-1">إقرار وتوقيع رقمي</h4>
                <p className="text-xs text-gray-600 font-medium">أقر بصفتي المعتمد الحالي بصحة الإجراءات، وأوافق على توجيه الطلب للمرحلة التالية في سلسلة الصلاحيات.</p>
              </label>
            </div>
          </div>

          {workflowStep === 'PROCUREMENT_ENTRY' ? (
            <button 
              type="button" onClick={() => handleAction('APPROVE')} disabled={!pledge || isSubmitting || !procurementNotes}
              className={`w-full py-4 rounded-xl text-base font-bold flex items-center justify-center gap-2 transition-all shadow-md
                ${(pledge && procurementNotes) ? 'bg-gradient-to-r from-teal-600 to-teal-500 text-white hover:from-teal-700' : 'bg-gray-200 text-gray-400 cursor-not-allowed'}`}
            >
              <UserCheck className="w-5 h-5" /> {isSubmitting ? 'جاري رفع الطلب...' : 'إرسال الطلب لمدير الخدمات المساندة للاعتماد'}
            </button>
          ) : (
            <div className="flex gap-3">
              <button onClick={() => handleAction('APPROVE')} disabled={!pledge || isSubmitting} className="flex-1 bg-green-600 hover:bg-green-700 text-white py-3 rounded-xl font-bold text-sm flex justify-center items-center gap-2 shadow-md">
                <CheckCircle2 size={18} /> موافقة وتحويل للتالي
              </button>
              <button onClick={() => handleAction('REJECT')} disabled={!pledge || isSubmitting} className="flex-1 bg-red-600 hover:bg-red-700 text-white py-3 rounded-xl font-bold text-sm flex justify-center items-center gap-2 shadow-md">
                <XCircle size={18} /> رفض الطلب وإرجاعه
              </button>
            </div>
          )}
        </Card>
      </main>
      <style dangerouslySetInnerHTML={{__html: `.animate-fade-in{animation:fadeIn .4s ease-in-out}@keyframes fadeIn{from{opacity:0;transform:translateY(10px)}to{opacity:1;transform:translateY(0)}}`}} />
    </div>
  );
}
```
