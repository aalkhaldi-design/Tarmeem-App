```react
import React, { useState } from 'react';
import { 
  Building, CheckCircle2, UserCheck, ShieldAlert, 
  FileText, Plus, Trash2, ArrowLeftRight, Truck, ClipboardList, CalendarClock
} from 'lucide-react';

const Card = ({ title, icon: Icon, children, className = '' }) => (
  <div className={`bg-white rounded-xl shadow-sm border border-gray-200 overflow-hidden mb-4 transition-all duration-300 ${className}`}>
    <div className="bg-gradient-to-r from-purple-800 to-purple-600 px-4 py-2.5 flex items-center gap-2 text-white">
      {Icon && <Icon className="w-5 h-5 text-teal-200" />}
      <h3 className="text-base font-bold">{title}</h3>
    </div>
    <div className="p-4 md:p-5">{children}</div>
  </div>
);

const ReadOnlyField = ({ label, value, highlight = false }) => (
  <div className="flex flex-col gap-1">
    <label className="text-xs font-semibold text-gray-500">{label}</label>
    <div className={`px-3 py-2 text-sm rounded-lg border font-medium
      ${highlight ? 'bg-teal-50 border-teal-200 text-teal-800' : 'bg-gray-100 border-gray-200 text-gray-700'}`}>
      {value || '-'}
    </div>
  </div>
);

const Input = ({ label, type = 'text', value, onChange, placeholder, className = '', required = false, disabled = false }) => (
  <div className={`flex flex-col gap-1 ${className}`}>
    <label className="text-xs font-semibold text-gray-700">{label} {required && <span className="text-red-500">*</span>}</label>
    <input 
      type={type} value={value} onChange={onChange} placeholder={placeholder} required={required} disabled={disabled}
      className={`px-3 py-1.5 text-sm rounded-lg border focus:outline-none transition-colors
        ${disabled ? 'bg-gray-100 border-gray-200 text-gray-500 cursor-not-allowed' : 'bg-gray-50 border-gray-200 focus:border-purple-500 text-gray-800'}`} 
    />
  </div>
);

export default function F20_MaterialSupplyPlan({ 
  onSubmit, 
  projectData = {}, 
  // ENGINEER_ENTRY, SUPPORT_SERVICES_PLANNING
  workflowStep = 'ENGINEER_ENTRY', 
  existingData = {} 
}) {
  const [isSubmitting, setIsSubmitting] = useState(false);
  
  // Beneficiary & Project Data
  const pData = Object.keys(projectData).length > 0 ? projectData : {
    caseRef: 'CS-9082', familyName: 'عبدالله محمد السالم', projectNumber: 'PRJ-2024-884', 
    cityNeighborhood: 'الدمام - حي النور', contractor: 'مؤسسة البناء الحديث',
    engineer: 'م. أحمد خالد'
  };

  const categories = ['مواد الكهرباء', 'الأدوات الصحية', 'العزل', 'السيراميك', 'الدهان', 'النجارة', 'الألمنيوم', 'الحدادة', 'أخرى'];
  const supplyTypes = ['شراء مباشر', 'دعم عيني (مجاني)', 'مخفض (شراكات)', 'من مستودع الجمعية'];

  // The Master Materials Array
  const [materials, setMaterials] = useState(existingData.materials || [
    { 
      id: 1, category: 'مواد الكهرباء', name: '', unit: 'حبة', quantity: '', 
      // Logistics (filled in Step 2)
      supplier: '', supplyType: '', pricingDate: '', paymentDate: '', supplyDate: '' 
    }
  ]);
  
  // Notes & Signatures
  const [engineerNotes, setEngineerNotes] = useState(existingData.engineerNotes || '');
  const [ssNotes, setSsNotes] = useState(existingData.ssNotes || '');
  const [pledge, setPledge] = useState(false);

  const isEngineerStep = workflowStep === 'ENGINEER_ENTRY';
  const isSsStep = workflowStep === 'SUPPORT_SERVICES_PLANNING';

  const addMaterial = () => {
    setMaterials([...materials, { 
      id: Date.now(), category: 'مواد الكهرباء', name: '', unit: 'حبة', quantity: '', 
      supplier: '', supplyType: '', pricingDate: '', paymentDate: '', supplyDate: '' 
    }]);
  };

  const removeMaterial = (id) => {
    setMaterials(materials.filter(m => m.id !== id));
  };

  const updateMaterial = (id, field, value) => {
    setMaterials(materials.map(m => m.id === id ? { ...m, [field]: value } : m));
  };

  const handleAction = async (actionType) => {
    if (!pledge || isSubmitting) return;
    setIsSubmitting(true);
    
    let payload = { ...existingData };

    if (isEngineerStep) {
      payload = { 
        ...payload, 
        projectInfo: pData,
        materials, 
        engineerNotes,
        status: 'PENDING_SUPPORT_SERVICES' 
      };
    } else if (isSsStep) {
      payload = { 
        ...payload, 
        materials, 
        ssNotes, 
        status: 'APPROVED' // Finalized Supply Plan
      };
    }

    if (onSubmit) {
      await onSubmit({ formId: 'F-20', action: actionType, payload });
    }
  };

  if (isSubmitting && !onSubmit) {
    return (
       <div className="min-h-screen bg-gray-50 flex items-center justify-center p-4" dir="rtl">
         <div className="bg-teal-500 text-white rounded-xl p-8 shadow-lg text-center max-w-md w-full">
            <CheckCircle2 className="w-16 h-16 mx-auto mb-4 text-teal-100" />
            <h2 className="text-2xl font-bold mb-2">تم تحديث خطة التوريد!</h2>
            <p className="text-teal-100 text-sm leading-relaxed">
              {isEngineerStep ? 'تم توجيه حصر المواد إلى إدارة الخدمات المساندة لاستكمال خطة التوريد.' : 'تم اعتماد خطة التوريد وإشعار المقاول والإدارة المالية بالجدول الزمني.'}
            </p>
         </div>
       </div>
    );
  }

  return (
    <div className="min-h-screen bg-gray-50 font-sans text-gray-800 pb-16" dir="rtl">
      <header className="bg-gradient-to-r from-purple-900 to-purple-700 text-white shadow-md pt-6 pb-12 px-4 text-center rounded-b-[2rem] relative z-10">
        <h1 className="text-2xl md:text-3xl font-extrabold mb-1">خطة توريد المواد (F-20)</h1>
        <h2 className="text-sm md:text-base text-teal-300 font-medium">
          {isEngineerStep && 'المرحلة 1: المهندس المشرف (حصر أنواع وكميات المواد المطلوبة)'}
          {isSsStep && 'المرحلة 2: الخدمات المساندة (جدولة الموردين وتواريخ السداد والتوريد)'}
        </h2>
      </header>

      <main className="container mx-auto px-4 -mt-8 relative z-20 max-w-6xl">
        
        {/* Project Info */}
        <Card title="بيانات المشروع" icon={Building}>
          <div className="grid grid-cols-2 md:grid-cols-5 gap-4">
            <ReadOnlyField label="رقم المشروع" value={pData.projectNumber} highlight />
            <ReadOnlyField label="رقم المنزل" value={pData.caseRef} />
            <ReadOnlyField label="المدينة - الحي" value={pData.cityNeighborhood} />
            <ReadOnlyField label="المهندس المشرف" value={pData.engineer} />
            <ReadOnlyField label="المقاول المنفذ" value={pData.contractor} />
          </div>
        </Card>

        {/* Master Materials Table */}
        <Card title="حصر أنواع وكميات المواد المطلوبة + الخطة اللوجستية للتوريد" icon={ClipboardList}>
          <div className="overflow-x-auto mb-4 pb-2">
            <table className="w-full text-right border-collapse min-w-[1000px]">
              <thead>
                {/* Two-tier header to separate Engineering vs Support Services responsibilities */}
                <tr>
                  <th colSpan={isEngineerStep ? 5 : 4} className="bg-purple-800 text-white p-2 border border-purple-900 text-center font-bold rounded-tr-lg">
                    القسم الهندسي (حصر الكميات)
                  </th>
                  {!isEngineerStep && (
                    <th colSpan="5" className="bg-teal-700 text-white p-2 border border-teal-800 text-center font-bold rounded-tl-lg">
                      قسم الخدمات المساندة (خطة التوريد)
                    </th>
                  )}
                </tr>
                <tr className="bg-gray-100 text-gray-700 text-xs">
                  <th className="p-2 border border-gray-300 font-bold w-32">القسم (النوع)</th>
                  <th className="p-2 border border-gray-300 font-bold w-48">وصف المادة المطلوبة</th>
                  <th className="p-2 border border-gray-300 font-bold w-20">الوحدة</th>
                  <th className="p-2 border border-gray-300 font-bold w-20 text-center">الكمية</th>
                  {isEngineerStep && <th className="p-2 border border-gray-300 font-bold w-12 text-center">حذف</th>}
                  
                  {!isEngineerStep && (
                    <>
                      <th className="p-2 border border-gray-300 font-bold bg-teal-50 w-32">جهة التوريد</th>
                      <th className="p-2 border border-gray-300 font-bold bg-teal-50 w-32">نوع التوريد</th>
                      <th className="p-2 border border-gray-300 font-bold bg-teal-50 w-28 text-center">تاريخ التسعير</th>
                      <th className="p-2 border border-gray-300 font-bold bg-teal-50 w-28 text-center">تاريخ السداد</th>
                      <th className="p-2 border border-gray-300 font-bold bg-teal-50 w-28 text-center">تاريخ التوريد</th>
                    </>
                  )}
                </tr>
              </thead>
              <tbody>
                {materials.map((m, index) => (
                  <tr key={m.id} className="bg-white border-b border-gray-200 text-sm hover:bg-gray-50 transition-colors">
                    {/* Engineering Section */}
                    <td className="p-1 border-x border-gray-200">
                      {isEngineerStep ? (
                        <select value={m.category} onChange={e => updateMaterial(m.id, 'category', e.target.value)} className="w-full p-1.5 rounded border border-gray-300 outline-none focus:border-purple-500">
                          {categories.map(c => <option key={c} value={c}>{c}</option>)}
                        </select>
                      ) : <span className="px-2 font-bold text-purple-800">{m.category}</span>}
                    </td>
                    <td className="p-1 border-x border-gray-200">
                      {isEngineerStep ? (
                        <input type="text" required value={m.name} onChange={e => updateMaterial(m.id, 'name', e.target.value)} placeholder="مثال: أسلاك كهرباء 4 ملم" className="w-full p-1.5 rounded border border-gray-300 outline-none focus:border-purple-500" />
                      ) : <span className="px-2">{m.name}</span>}
                    </td>
                    <td className="p-1 border-x border-gray-200">
                      {isEngineerStep ? (
                        <input type="text" required value={m.unit} onChange={e => updateMaterial(m.id, 'unit', e.target.value)} className="w-full p-1.5 rounded border border-gray-300 outline-none focus:border-purple-500 text-center" />
                      ) : <span className="px-2 text-gray-600 block text-center">{m.unit}</span>}
                    </td>
                    <td className="p-1 border-x border-gray-200 text-center">
                      {isEngineerStep ? (
                        <input type="number" min="1" required value={m.quantity} onChange={e => updateMaterial(m.id, 'quantity', e.target.value)} className="w-full p-1.5 rounded border border-gray-300 outline-none focus:border-purple-500 text-center font-bold" />
                      ) : <span className="font-bold text-lg">{m.quantity}</span>}
                    </td>
                    
                    {isEngineerStep && (
                      <td className="p-1 border-x border-gray-200 text-center">
                        {materials.length > 1 && (
                          <button type="button" onClick={() => removeMaterial(m.id)} className="text-red-500 hover:bg-red-100 p-1 rounded"><Trash2 size={16}/></button>
                        )}
                      </td>
                    )}

                    {/* Support Services Section */}
                    {!isEngineerStep && (
                      <>
                        <td className="p-1 border-x border-gray-200 bg-teal-50/30">
                          <input type="text" required value={m.supplier} onChange={e => updateMaterial(m.id, 'supplier', e.target.value)} placeholder="اسم المورد..." className="w-full p-1.5 rounded border border-teal-200 outline-none focus:border-teal-500 text-xs" />
                        </td>
                        <td className="p-1 border-x border-gray-200 bg-teal-50/30">
                          <select required value={m.supplyType} onChange={e => updateMaterial(m.id, 'supplyType', e.target.value)} className="w-full p-1.5 rounded border border-teal-200 outline-none focus:border-teal-500 text-xs">
                            <option value="">-- اختر --</option>
                            {supplyTypes.map(t => <option key={t} value={t}>{t}</option>)}
                          </select>
                        </td>
                        <td className="p-1 border-x border-gray-200 bg-teal-50/30">
                          <input type="date" required value={m.pricingDate} onChange={e => updateMaterial(m.id, 'pricingDate', e.target.value)} className="w-full p-1 rounded border border-teal-200 outline-none focus:border-teal-500 text-xs text-center" />
                        </td>
                        <td className="p-1 border-x border-gray-200 bg-teal-50/30">
                          <input type="date" required value={m.paymentDate} onChange={e => updateMaterial(m.id, 'paymentDate', e.target.value)} className="w-full p-1 rounded border border-teal-200 outline-none focus:border-teal-500 text-xs text-center" />
                        </td>
                        <td className="p-1 border-x border-gray-200 bg-teal-50/30">
                          <input type="date" required value={m.supplyDate} onChange={e => updateMaterial(m.id, 'supplyDate', e.target.value)} className="w-full p-1 rounded border border-teal-200 outline-none focus:border-teal-500 text-xs text-center" />
                        </td>
                      </>
                    )}
                  </tr>
                ))}
              </tbody>
            </table>
          </div>

          {isEngineerStep && (
            <button type="button" onClick={addMaterial} className="bg-purple-50 text-purple-700 border border-dashed border-purple-300 px-4 py-2.5 rounded-lg text-sm font-bold flex items-center gap-2 hover:bg-purple-100 transition-colors w-full md:w-auto justify-center">
              <Plus size={18} /> إضافة مادة جديدة للحصر
            </button>
          )}

          {!isEngineerStep && (
            <div className="bg-blue-50 border border-blue-200 p-3 rounded-lg text-xs text-blue-800 font-medium mt-2 flex items-start gap-2">
              <ShieldAlert size={16} className="shrink-0 mt-0.5" />
              <p>
                <strong>تنبيهات للموظف:</strong> يُقصد بـ (نوع التوريد) هل هو شراء مباشر بمقابل، أم دعم عيني. هناك آلية معتمدة لتوريد المواد المجانية أو المخفضة مع الجهات الداعمة ويجب تحديد التواريخ بدقة لكي لا يتوقف المقاول عن العمل.
              </p>
            </div>
          )}
        </Card>

        {/* Action Box for Current User */}
        <Card title="الاعتماد الميداني واللوجستي" icon={UserCheck}>
          <div className="bg-gray-50 border border-gray-200 p-4 rounded-xl mb-5">
            
            {!isEngineerStep && existingData.engineerNotes && (
              <div className="mb-4 bg-white p-3 border border-gray-200 rounded-lg">
                <span className="text-xs font-bold text-gray-500 block mb-1">ملاحظات المهندس المشرف (صاحب الطلب):</span>
                <p className="text-sm font-medium text-gray-800">{existingData.engineerNotes}</p>
              </div>
            )}

            <div className="flex flex-col gap-1 mb-4">
              <label className="text-xs font-bold text-gray-700">
                {isEngineerStep ? 'ملاحظات وتوجيهات المهندس المشرف للخدمات المساندة *' : 'ملاحظات الخدمات المساندة بشأن خطة التوريد *'}
              </label>
              <textarea 
                required rows={3} 
                value={isEngineerStep ? engineerNotes : ssNotes}
                onChange={e => isEngineerStep ? setEngineerNotes(e.target.value) : setSsNotes(e.target.value)}
                placeholder="اكتب التوصية أو الملاحظات الهامة هنا..." 
                className="px-3 py-2 text-sm rounded-lg border border-gray-300 focus:border-purple-500 outline-none resize-none bg-white shadow-sm" 
              />
            </div>

            <div className="flex items-start gap-4 p-3 bg-white rounded-lg border border-gray-200">
              <input type="checkbox" id="pledge" checked={pledge} onChange={(e) => setPledge(e.target.checked)} className="w-5 h-5 mt-0.5 rounded text-purple-600 focus:ring-purple-500 cursor-pointer" />
              <label htmlFor="pledge" className="cursor-pointer select-none">
                <h4 className="text-sm font-bold text-gray-900 mb-1">إقرار بصحة البيانات واعتماد المرحلة</h4>
                <p className="text-xs text-gray-600 font-medium">
                  {isEngineerStep 
                    ? 'أقر بصفتي المهندس المشرف بأن الكميات أعلاه دقيقة ومطابقة للتشخيص، وأحيلها للخدمات المساندة للبدء في توفيرها.' 
                    : 'أقر بصفتي موظف الخدمات المساندة بأنه تم جدولة الموردين والتواريخ بناءً على الآلية المعتمدة للجمعية.'}
                </p>
              </label>
            </div>
          </div>

          <button 
            type="button" onClick={() => handleAction('APPROVE')} 
            disabled={!pledge || isSubmitting || (isEngineerStep ? !engineerNotes : !ssNotes)}
            className={`w-full py-4 rounded-xl text-base font-bold flex items-center justify-center gap-2 transition-all shadow-md
              ${pledge ? 'bg-gradient-to-r from-teal-600 to-teal-500 text-white hover:from-teal-700' : 'bg-gray-200 text-gray-400 cursor-not-allowed'}`}
          >
            {isEngineerStep ? <ArrowLeftRight className="w-5 h-5" /> : <CalendarClock className="w-5 h-5" />}
            {isSubmitting ? 'جاري الاعتماد...' : isEngineerStep ? 'اعتماد حصر المواد وإرساله للخدمات المساندة' : 'اعتماد خطة التوريد النهائية للمشروع'}
          </button>
        </Card>
      </main>
    </div>
  );
}
```
