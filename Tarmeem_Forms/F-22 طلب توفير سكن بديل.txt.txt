```react
import React, { useState } from 'react';
import { 
  Building, Calendar, FileText, CheckCircle2, UserCheck, ShieldAlert, UploadCloud, X, Mail
} from 'lucide-react';

const Card = ({ title, icon: Icon, children, className = '' }) => (
  <div className={`bg-white rounded-xl shadow-sm border border-gray-200 overflow-hidden mb-4 transition-all duration-300 ${className}`}>
    <div className="bg-gradient-to-r from-purple-800 to-purple-600 px-4 py-2.5 flex items-center gap-2 text-white">
      {Icon && <Icon className="w-5 h-5 text-teal-200" />}
      <h3 className="text-base font-bold">{title}</h3>
    </div>
    <div className="p-4 md:p-5">{children}</div>
  </div>
);

const ReadOnlyField = ({ label, value }) => (
  <div className="flex flex-col gap-1">
    <label className="text-xs font-semibold text-gray-500">{label}</label>
    <div className="px-3 py-2 text-sm rounded-lg bg-gray-100 border border-gray-200 text-gray-700 font-medium">
      {value || '-'}
    </div>
  </div>
);

const Input = ({ label, type = 'text', value, onChange, placeholder, className = '', required = false }) => (
  <div className={`flex flex-col gap-1 ${className}`}>
    <label className="text-xs font-semibold text-gray-700">{label} {required && <span className="text-red-500">*</span>}</label>
    <input type={type} value={value} onChange={onChange} placeholder={placeholder} required={required} className="px-3 py-1.5 text-sm rounded-lg border border-gray-200 focus:border-teal-500 focus:ring-1 focus:ring-teal-200 outline-none text-gray-800 bg-gray-50" />
  </div>
);

export default function F22_AlternativeHousing({ onSubmit, f02Data = {}, f18Data = {} }) {
  const [isSubmitting, setIsSubmitting] = useState(false);
  const [attachedFile, setAttachedFile] = useState(null);
  
  // Safe extraction from previous forms (F-02 for Family/Partner, F-18 for Dates)
  const caseData = Object.keys(f02Data).length > 0 ? f02Data : {
    caseRef: 'CS-9082',
    personal: { fullName: 'عبدالله محمد السالم', city: 'الدمام', neighborhood: 'حي النور', mobile1: '0500000000' },
    family: { total: '6' },
    partnershipInfo: { partnerEntity: 'جمعية البر بالمنطقة الشرقية' }
  };

  const datesData = Object.keys(f18Data).length > 0 ? f18Data : {
    renovationStartDate: '2024-11-01',
    evacuationDate: '2024-10-28',
    renovationEndDate: '2024-12-01'
  };

  // Helper to get day name from date string
  const getDayName = (dateString) => {
    if (!dateString) return '---';
    const date = new Date(dateString);
    return date.toLocaleDateString('ar-SA', { weekday: 'long' });
  };

  const [data, setData] = useState({
    partnerEntity: caseData.partnershipInfo?.partnerEntity || '',
    employeeName: '',
    employeeTitle: '',
    periodFrom: datesData.evacuationDate || '',
    periodTo: datesData.renovationEndDate || '',
    employeePledge: false
  });

  const handleFileChange = (e) => {
    if (e.target.files && e.target.files[0]) {
      setAttachedFile({
        file: e.target.files[0],
        name: e.target.files[0].name,
        previewUrl: URL.createObjectURL(e.target.files[0])
      });
    }
  };

  const removeFile = () => {
    if (attachedFile) URL.revokeObjectURL(attachedFile.previewUrl);
    setAttachedFile(null);
  };

  const handleSubmit = async (e) => {
    e.preventDefault();
    if (!data.employeePledge) return;
    setIsSubmitting(true);
    
    if (onSubmit) {
      await onSubmit({ 
        formId: 'F-22', 
        payload: { ...data, statementDocument: attachedFile } 
      });
    }
  };

  if (isSubmitting && !onSubmit) {
    return (
       <div className="min-h-screen bg-gray-50 flex items-center justify-center p-4" dir="rtl">
         <div className="bg-teal-500 text-white rounded-xl p-8 shadow-lg text-center max-w-md w-full">
            <CheckCircle2 className="w-16 h-16 mx-auto mb-4 text-teal-100" />
            <h2 className="text-2xl font-bold mb-2">تم إصدار الخطاب بنجاح!</h2>
            <p className="text-teal-100 text-sm">تم أتمتة طلب السكن البديل (F-22) وإرساله في النظام للتنسيق مع الجهة الشريكة.</p>
         </div>
       </div>
    );
  }

  return (
    <div className="min-h-screen bg-gray-50 font-sans text-gray-800 pb-16" dir="rtl">
      <header className="bg-gradient-to-r from-purple-900 to-purple-700 text-white shadow-md pt-6 pb-12 px-4 text-center rounded-b-[2rem] relative z-10">
        <h1 className="text-2xl md:text-3xl font-extrabold mb-1">طلب توفير سكن بديل وأثاث (F-22)</h1>
        <h2 className="text-sm md:text-base text-teal-300 font-medium">خطاب موجه للجهة الشريكة (مؤتمت)</h2>
      </header>

      <main className="container mx-auto px-4 -mt-8 relative z-20 max-w-4xl">
        <form onSubmit={handleSubmit}>
          
          {/* Header of the Official Letter */}
          <Card title="بيانات التوجيه" icon={Mail}>
            <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
              <Input label="إلى (الجهة الشريكة الموجه لها الخطاب)" required value={data.partnerEntity} onChange={e => setData({...data, partnerEntity: e.target.value})} />
              <ReadOnlyField label="من" value="جمعية ترميم" />
            </div>
          </Card>

          {/* Auto-Generated Letter Body (Matches the paper form) */}
          <Card title="نص الخطاب الآلي (يُولد من بيانات F-18 و F-02)" icon={FileText}>
            <div className="bg-white border-2 border-gray-200 p-6 rounded-xl text-gray-800 leading-loose text-justify font-medium text-lg shadow-sm">
              <p>
                إشارةً إلى الاتفاقية المبرمة بين جمعية ترميم وجمعية <span className="text-purple-700 font-bold border-b border-purple-300 pb-0.5 px-1">{data.partnerEntity || '................'}</span>
              </p>
              <p className="mt-4">
                نفيدكم بأن المقاول سوف يبدأ أعمال الترميم في منزل <span className="text-teal-700 font-bold bg-teal-50 px-2 py-0.5 rounded">{caseData.personal?.fullName}</span>
              </p>
              <p className="mt-4">
                يوم <span className="font-bold">{getDayName(datesData.renovationStartDate)}</span> 
                {' '}تاريخ <span className="font-bold">{datesData.renovationStartDate || '................'}</span>، 
                ولذلك يلزم إخلاء المنزل 
                يوم <span className="font-bold text-red-600">{getDayName(datesData.evacuationDate)}</span> 
                {' '}تاريخ <span className="font-bold text-red-600">{datesData.evacuationDate || '................'}</span>.
              </p>
              <p className="mt-4">
                آملين منكم توفير سكن بديل للأسرة وتوفير الأثاث الموضح في البيان المرفق خلال الفترة 
                من <input type="date" required value={data.periodFrom} onChange={e => setData({...data, periodFrom: e.target.value})} className="inline-block px-2 py-1 mx-1 text-sm border border-gray-300 rounded outline-none focus:border-purple-500" /> 
                إلى <input type="date" required value={data.periodTo} onChange={e => setData({...data, periodTo: e.target.value})} className="inline-block px-2 py-1 mx-1 text-sm border border-gray-300 rounded outline-none focus:border-purple-500" />.
              </p>
            </div>
          </Card>

          {/* Family Table from the Form */}
          <Card title="بيانات الأسرة المستفيدة" icon={Building}>
            <div className="overflow-x-auto">
              <table className="w-full text-right border-collapse">
                <thead>
                  <tr className="bg-teal-600 text-white text-sm">
                    <th className="p-3 border border-teal-700 font-bold rounded-tr-lg">اسم الأسرة</th>
                    <th className="p-3 border border-teal-700 font-bold">عدد الأفراد</th>
                    <th className="p-3 border border-teal-700 font-bold">الموقع (المدينة - الحي)</th>
                    <th className="p-3 border border-teal-700 font-bold rounded-tl-lg">رقم التواصل</th>
                  </tr>
                </thead>
                <tbody>
                  <tr className="bg-white text-sm">
                    <td className="p-3 border border-gray-200 font-medium">{caseData.personal?.fullName}</td>
                    <td className="p-3 border border-gray-200">{caseData.family?.total}</td>
                    <td className="p-3 border border-gray-200">{caseData.personal?.city} - {caseData.personal?.neighborhood}</td>
                    <td className="p-3 border border-gray-200" dir="ltr">{caseData.personal?.mobile1}</td>
                  </tr>
                </tbody>
              </table>
            </div>
          </Card>

          {/* Optional Attachment (البيان المرفق) */}
          <Card title="مرفقات الخطاب (البيان المرفق للأثاث)" icon={UploadCloud}>
            <div className="bg-purple-50 p-6 rounded-xl border-2 border-dashed border-purple-200 text-center hover:bg-purple-100 transition-colors">
              <UploadCloud className="w-10 h-10 text-purple-500 mx-auto mb-2" />
              <p className="text-sm font-bold text-purple-900 mb-1">إرفاق بيان الأثاث (اختياري)</p>
              <p className="text-xs text-purple-600 mb-4">في حال وجود بيان ورقي أو ملف للتسعيرات والأثاث، يمكنك إرفاقه هنا ليرسل مع الخطاب.</p>
              <label className="cursor-pointer bg-purple-600 text-white px-6 py-2.5 rounded-lg text-sm font-bold hover:bg-purple-700 shadow-sm">
                تصفح الملفات
                <input type="file" accept="image/*,.pdf,.xlsx,.xls" className="hidden" onChange={handleFileChange} />
              </label>
            </div>
            {attachedFile && (
              <div className="mt-4 flex justify-between items-center bg-white p-3 rounded-lg border border-teal-200 text-sm shadow-sm bg-teal-50">
                <div className="flex items-center gap-2 text-teal-800 font-bold">
                  <CheckCircle2 size={18} />
                  <span className="truncate max-w-[200px]">{attachedFile.name}</span>
                </div>
                <button type="button" onClick={removeFile} className="text-red-500 hover:bg-red-50 p-1 rounded"><X size={18}/></button>
              </div>
            )}
          </Card>

          {/* Employee Signatures & Digital Pledge */}
          <Card title="التوثيق والاعتماد لجمعية ترميم" icon={ShieldAlert}>
            <div className="grid grid-cols-1 md:grid-cols-2 gap-5 mb-6 border-b border-gray-100 pb-6">
              <Input label="الاسم (الموظف المسؤول عن الطلب)" required value={data.employeeName} onChange={e => setData({...data, employeeName: e.target.value})} placeholder="الاسم الرباعي..." />
              <Input label="المسمى الوظيفي" required value={data.employeeTitle} onChange={e => setData({...data, employeeTitle: e.target.value})} placeholder="مثال: الباحث الاجتماعي / المهندس المشرف" />
            </div>

            <div className={`p-5 rounded-xl border-2 transition-all ${data.employeePledge ? 'bg-teal-50 border-teal-500' : 'bg-gray-50 border-gray-200'}`}>
              <div className="flex items-start gap-4">
                <input 
                  type="checkbox" 
                  id="employeePledge"
                  required
                  checked={data.employeePledge}
                  onChange={(e) => setData({...data, employeePledge: e.target.checked})}
                  className="w-6 h-6 mt-1 rounded text-teal-600 focus:ring-teal-500 cursor-pointer"
                />
                <label htmlFor="employeePledge" className="cursor-pointer select-none">
                  <h4 className="text-base font-bold text-gray-900 mb-2">اعتماد إصدار الخطاب وتوجيهه</h4>
                  <p className="text-sm text-gray-700 leading-loose font-medium">
                    أقر أنا الموظف المسؤول بصحة التواريخ والبيانات المسجلة أعلاه، وأعتمد إصدار هذا الخطاب (طلب توفير سكن بديل وأثاث) 
                    ليُرسل أو يُشارك مع الجهة الشريكة، ويُعد حسابي في النظام بمثابة التوقيع الرسمي للجمعية.
                  </p>
                </label>
              </div>
            </div>
          </Card>

          {/* Submit */}
          <button 
            type="submit"
            disabled={!data.employeePledge || isSubmitting}
            className={`w-full py-4 rounded-xl text-base font-bold flex items-center justify-center gap-2 transition-all shadow-md
              ${data.employeePledge 
                ? 'bg-gradient-to-r from-teal-600 to-teal-500 hover:from-teal-700 text-white cursor-pointer' 
                : 'bg-gray-200 text-gray-400 cursor-not-allowed'}`}
          >
            <UserCheck className="w-5 h-5" /> 
            {isSubmitting ? 'جاري إصدار الخطاب...' : 'اعتماد وإصدار الخطاب الرسمي'}
          </button>
        </form>
      </main>
    </div>
  );
}

```
