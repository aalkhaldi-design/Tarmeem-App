```react
import React, { useState } from 'react';
import { 
  CheckCircle2, XCircle, FileText, UserCheck, ShieldAlert, 
  Plus, Trash2, HardHat, HeartHandshake, Calculator, Building
} from 'lucide-react';

const Card = ({ title, icon: Icon, children, className = '' }) => (
  <div className={`bg-white rounded-xl shadow-sm border border-gray-200 overflow-hidden mb-4 transition-all duration-300 ${className}`}>
    <div className="bg-gradient-to-r from-purple-800 to-purple-600 px-4 py-2.5 flex items-center gap-2 text-white">
      {Icon && <Icon className="w-5 h-5 text-teal-200" />}
      <h3 className="text-base font-bold">{title}</h3>
    </div>
    <div className="p-4 md:p-5">{children}</div>
  </div>
);

const ReadOnlyField = ({ label, value, highlight = false }) => (
  <div className="flex flex-col gap-1">
    <label className="text-xs font-semibold text-gray-500">{label}</label>
    <div className={`px-3 py-2 text-sm rounded-lg border font-medium
      ${highlight ? 'bg-teal-50 border-teal-200 text-teal-800' : 'bg-gray-100 border-gray-200 text-gray-700'}`}>
      {value || '-'}
    </div>
  </div>
);

const Input = ({ label, type = 'text', value, onChange, placeholder, className = '', required = false }) => (
  <div className={`flex flex-col gap-1 ${className}`}>
    <label className="text-xs font-semibold text-gray-700">{label} {required && <span className="text-red-500">*</span>}</label>
    <input type={type} value={value} onChange={onChange} placeholder={placeholder} required={required} className="px-3 py-1.5 text-sm rounded-lg border border-gray-200 focus:border-purple-500 outline-none text-gray-800 bg-gray-50" />
  </div>
);

export default function F23_ScopeChangeWorkflow({ 
  onSubmit, 
  projectData = {}, 
  // ENGINEER_ENTRY, PM_APPROVAL, SAMAYA_APPROVAL, FINANCE_APPROVAL, EXECUTIVE_APPROVAL
  workflowStep = 'ENGINEER_ENTRY', 
  existingData = {} 
}) {
  const [isSubmitting, setIsSubmitting] = useState(false);
  
  // Beneficiary & Project Data
  const pData = Object.keys(projectData).length > 0 ? projectData : {
    caseRef: 'CS-9082', familyName: 'عبدالله محمد السالم', projectNumber: 'PRJ-2024-884', contractor: 'مؤسسة البناء الحديث'
  };

  // State for additional works (Dynamic Array)
  const [additionalWorks, setAdditionalWorks] = useState(existingData.additionalWorks || [
    { id: 1, description: '', dimensions: '', contractorName: pData.contractor, price: '' }
  ]);
  
  // Opinions & Notes State
  const [engineerNotes, setEngineerNotes] = useState(existingData.engineerNotes || '');
  const [pmNotes, setPmNotes] = useState(existingData.pmNotes || '');
  const [samayaNotes, setSamayaNotes] = useState(existingData.samayaNotes || '');
  const [financeNotes, setFinanceNotes] = useState(existingData.financeNotes || '');
  const [executiveNotes, setExecutiveNotes] = useState(existingData.executiveNotes || '');
  
  const [pledge, setPledge] = useState(false);

  const isReadOnly = workflowStep !== 'ENGINEER_ENTRY';

  const addWorkItem = () => {
    setAdditionalWorks([...additionalWorks, { id: Date.now(), description: '', dimensions: '', contractorName: pData.contractor, price: '' }]);
  };

  const removeWorkItem = (id) => {
    setAdditionalWorks(additionalWorks.filter(w => w.id !== id));
  };

  const updateWorkItem = (id, field, value) => {
    setAdditionalWorks(additionalWorks.map(w => w.id === id ? { ...w, [field]: value } : w));
  };

  const getTotalPrice = () => {
    return additionalWorks.reduce((sum, item) => sum + (Number(item.price) || 0), 0);
  };

  const handleAction = async (actionType) => {
    if (!pledge || isSubmitting) return;
    setIsSubmitting(true);
    
    let payload = { ...existingData };

    // State Machine Routing
    if (workflowStep === 'ENGINEER_ENTRY') {
      payload = { ...payload, additionalWorks, engineerNotes, status: 'PENDING_PM' };
    } else if (workflowStep === 'PM_APPROVAL') {
      payload = { ...payload, pmNotes, status: actionType === 'APPROVE' ? 'PENDING_SAMAYA' : 'REJECTED' };
    } else if (workflowStep === 'SAMAYA_APPROVAL') {
      payload = { ...payload, samayaNotes, status: actionType === 'APPROVE' ? 'PENDING_FINANCE' : 'REJECTED' };
    } else if (workflowStep === 'FINANCE_APPROVAL') {
      payload = { ...payload, financeNotes, status: actionType === 'APPROVE' ? 'PENDING_EXECUTIVE' : 'REJECTED' };
    } else if (workflowStep === 'EXECUTIVE_APPROVAL') {
      payload = { ...payload, executiveNotes, status: actionType === 'APPROVE' ? 'APPROVED' : 'REJECTED' };
    }

    if (onSubmit) {
      await onSubmit({ formId: 'F-23', action: actionType, payload });
    }
  };

  if (isSubmitting && !onSubmit) {
    return (
       <div className="min-h-screen bg-gray-50 flex items-center justify-center p-4" dir="rtl">
         <div className="bg-teal-500 text-white rounded-xl p-8 shadow-lg text-center max-w-md w-full">
            <CheckCircle2 className="w-16 h-16 mx-auto mb-4 text-teal-100" />
            <h2 className="text-2xl font-bold mb-2">تم اعتماد الإجراء بنجاح!</h2>
            <p className="text-teal-100 text-sm leading-relaxed">تم توجيه طلب الأعمال الإضافية للخطوة التالية في النظام.</p>
         </div>
       </div>
    );
  }

  return (
    <div className="min-h-screen bg-gray-50 font-sans text-gray-800 pb-16" dir="rtl">
      <header className="bg-gradient-to-r from-purple-900 to-purple-700 text-white shadow-md pt-6 pb-12 px-4 text-center rounded-b-[2rem] relative z-10">
        <h1 className="text-2xl md:text-3xl font-extrabold mb-1">اعتماد بنود الأعمال الإضافية (F-23)</h1>
        <h2 className="text-sm md:text-base text-teal-300 font-medium">
          {workflowStep === 'ENGINEER_ENTRY' && 'المرحلة 1: المهندس المشرف (طلب الأعمال وتسعيرها)'}
          {workflowStep === 'PM_APPROVAL' && 'المرحلة 2: مدير المشاريع (الاعتماد الفني)'}
          {workflowStep === 'SAMAYA_APPROVAL' && 'المرحلة 3: مركز سمايا (رأي التطوع للعمل)'}
          {workflowStep === 'FINANCE_APPROVAL' && 'المرحلة 4: الإدارة المالية (الاعتماد المالي)'}
          {workflowStep === 'EXECUTIVE_APPROVAL' && 'المرحلة 5: المدير التنفيذي (الاعتماد النهائي)'}
        </h2>
      </header>

      <main className="container mx-auto px-4 -mt-8 relative z-20 max-w-4xl">
        
        {/* Project Info */}
        <Card title="بيانات المشروع" icon={Building}>
          <div className="grid grid-cols-1 md:grid-cols-3 gap-4">
            <ReadOnlyField label="رقم المشروع" value={pData.projectNumber} highlight />
            <ReadOnlyField label="اسم المستفيد" value={pData.familyName} />
            <ReadOnlyField label="المقاول الأساسي للمشروع" value={pData.contractor} />
          </div>
        </Card>

        {/* Dynamic Additional Works */}
        <Card title="بيان الأعمال والمواد الإضافية المطلوبة" icon={HardHat}>
          {isReadOnly ? (
            <div className="overflow-x-auto">
              <table className="w-full text-right border-collapse mb-4">
                <thead>
                  <tr className="bg-teal-600 text-white text-sm">
                    <th className="p-3 border border-teal-700 font-bold rounded-tr-lg">وصف العمل / المادة</th>
                    <th className="p-3 border border-teal-700 font-bold">المساحة / الأبعاد</th>
                    <th className="p-3 border border-teal-700 font-bold">المُنفذ المقترح</th>
                    <th className="p-3 border border-teal-700 font-bold rounded-tl-lg">السعر (ر.س)</th>
                  </tr>
                </thead>
                <tbody>
                  {additionalWorks.map((work, index) => (
                    <tr key={work.id} className="bg-gray-50 border-b border-gray-200">
                      <td className="p-3 border-x border-gray-200 text-sm font-bold">{work.description}</td>
                      <td className="p-3 border-x border-gray-200 text-sm">{work.dimensions}</td>
                      <td className="p-3 border-x border-gray-200 text-sm">{work.contractorName}</td>
                      <td className="p-3 border-x border-gray-200 text-sm font-bold text-teal-700">{Number(work.price).toLocaleString()}</td>
                    </tr>
                  ))}
                </tbody>
                <tfoot>
                  <tr className="bg-teal-50">
                    <td colSpan="3" className="p-3 border border-gray-200 font-bold text-left">التكلفة الإضافية الإجمالية:</td>
                    <td className="p-3 border border-gray-200 font-black text-lg text-teal-800">{getTotalPrice().toLocaleString()} ر.س</td>
                  </tr>
                </tfoot>
              </table>
            </div>
          ) : (
            <>
              <div className="space-y-4 mb-4">
                {additionalWorks.map((work, index) => (
                  <div key={work.id} className="p-4 bg-gray-50 border border-gray-200 rounded-xl relative">
                    {additionalWorks.length > 1 && (
                      <button type="button" onClick={() => removeWorkItem(work.id)} className="absolute top-3 left-3 text-red-500 hover:bg-red-100 p-1.5 rounded-lg"><Trash2 size={18}/></button>
                    )}
                    <h4 className="text-xs font-bold text-teal-700 mb-3">البند الإضافي رقم ({index + 1})</h4>
                    <div className="grid grid-cols-1 md:grid-cols-2 gap-3 mb-3">
                      <Input label="وصف العمل (مثال: معالجة سقف بمساحة غرفة)" required value={work.description} onChange={e => updateWorkItem(work.id, 'description', e.target.value)} className="md:col-span-2" />
                      <Input label="المساحة / الأبعاد (مثال: 5.30*3.99)" required value={work.dimensions} onChange={e => updateWorkItem(work.id, 'dimensions', e.target.value)} />
                      <Input label="اسم المقاول / المورد" required value={work.contractorName} onChange={e => updateWorkItem(work.id, 'contractorName', e.target.value)} />
                    </div>
                    <div className="w-full md:w-1/2">
                      <Input label="السعر التقديري (ر.س)" type="number" required value={work.price} onChange={e => updateWorkItem(work.id, 'price', e.target.value)} />
                    </div>
                  </div>
                ))}
              </div>
              
              <div className="flex flex-col md:flex-row items-center justify-between gap-4 border-t border-gray-100 pt-4">
                <button type="button" onClick={addWorkItem} className="bg-teal-50 text-teal-700 border border-dashed border-teal-300 px-4 py-2.5 rounded-lg text-sm font-bold flex items-center gap-2 hover:bg-teal-100 transition-colors w-full md:w-auto">
                  <Plus size={18} /> إضافة بند عمل / مادة أخرى
                </button>
                <div className="text-lg font-bold text-gray-800 bg-gray-100 px-4 py-2 rounded-lg border border-gray-200">
                  الإجمالي: <span className="text-teal-700 font-black">{getTotalPrice().toLocaleString()} ر.س</span>
                </div>
              </div>
            </>
          )}
        </Card>

        {/* Historical Opinions (Read Only for downstream steps) */}
        {workflowStep !== 'ENGINEER_ENTRY' && (
          <Card title="الآراء والتوصيات السابقة" icon={FileText}>
            <div className="space-y-3">
              <ReadOnlyField label="الرأي الهندسي ومبررات الزيادة (المهندس المشرف)" value={existingData.engineerNotes} />
              {workflowStep !== 'PM_APPROVAL' && <ReadOnlyField label="رأي مدير المشاريع" value={existingData.pmNotes} />}
              {['FINANCE_APPROVAL', 'EXECUTIVE_APPROVAL'].includes(workflowStep) && <ReadOnlyField label="رأي التطوع - رئيس مركز سمايا (هل يمكن توفير متطوعين؟)" value={existingData.samayaNotes} />}
              {workflowStep === 'EXECUTIVE_APPROVAL' && <ReadOnlyField label="الاعتماد المالي (المحاسب)" value={existingData.financeNotes} />}
            </div>
          </Card>
        )}

        {/* Current Actor Action Box */}
        <Card title="الإجراء والاعتماد الحالي" icon={ShieldAlert}>
          <div className="bg-gray-50 border border-gray-200 p-4 rounded-xl mb-5">
            <div className="flex flex-col gap-1 mb-4">
              <label className="text-xs font-bold text-gray-700">
                {workflowStep === 'ENGINEER_ENTRY' && 'المبررات الفنية لطلب هذه الأعمال الإضافية *'}
                {workflowStep === 'PM_APPROVAL' && 'رأي واعتماد مدير المشاريع *'}
                {workflowStep === 'SAMAYA_APPROVAL' && 'رأي التطوع (هل توجد فرصة لتنفيذ هذا البند عبر متطوعين لتخفيض التكلفة؟) *'}
                {workflowStep === 'FINANCE_APPROVAL' && 'ملاحظات واعتماد المحاسب (التأكد من توفر ميزانية) *'}
                {workflowStep === 'EXECUTIVE_APPROVAL' && 'الاعتماد النهائي والتوجيه (المدير التنفيذي) *'}
              </label>
              <textarea 
                required
                rows={3} 
                value={
                  workflowStep === 'ENGINEER_ENTRY' ? engineerNotes :
                  workflowStep === 'PM_APPROVAL' ? pmNotes :
                  workflowStep === 'SAMAYA_APPROVAL' ? samayaNotes :
                  workflowStep === 'FINANCE_APPROVAL' ? financeNotes : executiveNotes
                }
                onChange={e => {
                  const val = e.target.value;
                  if (workflowStep === 'ENGINEER_ENTRY') setEngineerNotes(val);
                  else if (workflowStep === 'PM_APPROVAL') setPmNotes(val);
                  else if (workflowStep === 'SAMAYA_APPROVAL') setSamayaNotes(val);
                  else if (workflowStep === 'FINANCE_APPROVAL') setFinanceNotes(val);
                  else setExecutiveNotes(val);
                }}
                placeholder="اكتب التوصية هنا قبل الاعتماد..." 
                className="px-3 py-2 text-sm rounded-lg border border-gray-300 focus:border-purple-500 outline-none resize-none bg-white shadow-sm" 
              />
            </div>

            <div className="flex items-start gap-4 p-3 bg-white rounded-lg border border-gray-200">
              <input 
                type="checkbox" 
                id="pledge"
                checked={pledge}
                onChange={(e) => setPledge(e.target.checked)}
                className="w-5 h-5 mt-0.5 rounded text-purple-600 focus:ring-purple-500 cursor-pointer"
              />
              <label htmlFor="pledge" className="cursor-pointer select-none">
                <h4 className="text-sm font-bold text-gray-900 mb-1">إقرار بصحة البيانات واعتماد المرحلة</h4>
                <p className="text-xs text-gray-600 font-medium">أقر بصفتي المسؤول في هذه المرحلة، بالموافقة على ما ورد أعلاه وسيتم توثيق الاعتماد باسمي رقمياً.</p>
              </label>
            </div>
          </div>

          {workflowStep === 'ENGINEER_ENTRY' ? (
            <button 
              type="button" onClick={() => handleAction('APPROVE')} disabled={!pledge || isSubmitting || !engineerNotes}
              className={`w-full py-4 rounded-xl text-base font-bold flex items-center justify-center gap-2 transition-all shadow-md
                ${(pledge && engineerNotes) ? 'bg-gradient-to-r from-teal-600 to-teal-500 text-white hover:from-teal-700' : 'bg-gray-200 text-gray-400 cursor-not-allowed'}`}
            >
              <UserCheck className="w-5 h-5" /> {isSubmitting ? 'جاري رفع الطلب...' : 'إرسال طلب الأعمال الإضافية لمدير المشاريع'}
            </button>
          ) : (
            <div className="flex gap-3">
              <button onClick={() => handleAction('APPROVE')} disabled={!pledge || isSubmitting} className="flex-1 bg-green-600 hover:bg-green-700 text-white py-3 rounded-xl font-bold text-sm flex justify-center items-center gap-2 shadow-md">
                <CheckCircle2 size={18} /> موافقة وتحويل للتالي
              </button>
              <button onClick={() => handleAction('REJECT')} disabled={!pledge || isSubmitting} className="flex-1 bg-red-600 hover:bg-red-700 text-white py-3 rounded-xl font-bold text-sm flex justify-center items-center gap-2 shadow-md">
                <XCircle size={18} /> رفض الطلب وإرجاعه
              </button>
            </div>
          )}
        </Card>
      </main>
    </div>
  );
}

```
