```react
import React, { useState } from 'react';
import { 
  Camera, CheckCircle2, UserCheck, ShieldAlert, FileText, 
  MapPin, CalendarClock, ArrowLeftRight, UploadCloud, X, Link as LinkIcon
} from 'lucide-react';

const Card = ({ title, icon: Icon, children, className = '' }) => (
  <div className={`bg-white rounded-xl shadow-sm border border-gray-200 overflow-hidden mb-4 transition-all duration-300 ${className}`}>
    <div className="bg-gradient-to-r from-purple-800 to-purple-600 px-4 py-2.5 flex items-center gap-2 text-white">
      {Icon && <Icon className="w-5 h-5 text-teal-200" />}
      <h3 className="text-base font-bold">{title}</h3>
    </div>
    <div className="p-4 md:p-5">{children}</div>
  </div>
);

const ReadOnlyField = ({ label, value, highlight = false }) => (
  <div className="flex flex-col gap-1">
    <label className="text-xs font-semibold text-gray-500">{label}</label>
    <div className={`px-3 py-2 text-sm rounded-lg border font-bold
      ${highlight ? 'bg-purple-50 border-purple-200 text-purple-800' : 'bg-gray-100 border-gray-200 text-gray-700'}`}>
      {value || '-'}
    </div>
  </div>
);

const Input = ({ label, type = 'text', value, onChange, placeholder, className = '', required = false, disabled = false }) => (
  <div className={`flex flex-col gap-1 ${className}`}>
    <label className="text-xs font-semibold text-gray-700">{label} {required && <span className="text-red-500">*</span>}</label>
    <input 
      type={type} value={value} onChange={onChange} placeholder={placeholder} required={required} disabled={disabled}
      className={`px-3 py-1.5 text-sm rounded-lg border focus:outline-none transition-colors
        ${disabled ? 'bg-gray-100 border-gray-200 text-gray-500 cursor-not-allowed' : 'bg-gray-50 border-gray-200 focus:border-purple-500 text-gray-800'}`} 
    />
  </div>
);

export default function F52_MediaRequestWorkflow({ 
  onSubmit, 
  projectData = {}, 
  // REQUEST_ENTRY (إدارة المشاريع/النظام), MEDIA_EXECUTION (إدارة الاتصال المؤسسي)
  workflowStep = 'REQUEST_ENTRY', 
  existingData = {} 
}) {
  const [isSubmitting, setIsSubmitting] = useState(false);
  
  // Beneficiary & Project Data
  const pData = Object.keys(projectData).length > 0 ? projectData : {
    caseRef: 'CS-9082', projectNumber: 'PRJ-2024-884', familyName: 'عبدالله محمد السالم', 
    city: 'الدمام', neighborhood: 'حي النور',
    triggerSource: 'إدارة المشاريع (طلب آلي من F-07)' 
  };

  // Step 1: Request Details
  const [fromDept, setFromDept] = useState(existingData.fromDept || pData.triggerSource || 'إدارة المشاريع');
  const [toDept] = useState('إدارة الاتصال المؤسسي (قسم الإعلام)');
  const [docPhase, setDocPhase] = useState(existingData.docPhase || 'يوم التسليم');
  const [targetDate, setTargetDate] = useState(existingData.targetDate || '');
  
  // Step 2: Media Execution Details
  const [executionNotes, setExecutionNotes] = useState(existingData.executionNotes || '');
  const [mediaDriveLink, setMediaDriveLink] = useState(existingData.mediaDriveLink || '');
  const [mediaFiles, setMediaFiles] = useState([]);
  
  const [pledge, setPledge] = useState(false);

  const isReadOnly = workflowStep !== 'REQUEST_ENTRY';
  const isMediaStep = workflowStep === 'MEDIA_EXECUTION';

  const phases = [
    'قبل (تشخيص)', 'قبل (بنود جديدة)', 'أثناء', 'بعد', 
    'زيارة التقييم والجودة', 'يوم التسليم', 'مبادرة تطوعية'
  ];

  const handleFileChange = (e) => {
    if (e.target.files) {
      const files = Array.from(e.target.files).map(file => ({
        file, name: file.name, previewUrl: URL.createObjectURL(file)
      }));
      setMediaFiles(prev => [...prev, ...files]);
    }
  };

  const removeFile = (index) => {
    const newFiles = [...mediaFiles];
    URL.revokeObjectURL(newFiles[index].previewUrl);
    newFiles.splice(index, 1);
    setMediaFiles(newFiles);
  };

  const handleAction = async (actionType) => {
    if (!pledge || isSubmitting) return;
    setIsSubmitting(true);
    
    let payload = { ...existingData };

    if (workflowStep === 'REQUEST_ENTRY') {
      payload = { 
        ...payload, 
        projectInfo: pData,
        fromDept, toDept, docPhase, targetDate,
        status: 'PENDING_MEDIA_EXECUTION' 
      };
    } else if (workflowStep === 'MEDIA_EXECUTION') {
      payload = { 
        ...payload, 
        executionNotes, mediaDriveLink, uploadedMedia: mediaFiles,
        status: 'COMPLETED' 
      };
    }

    if (onSubmit) {
      await onSubmit({ formId: 'F-52', action: actionType, payload });
    }
  };

  if (isSubmitting && !onSubmit) {
    return (
       <div className="min-h-screen bg-gray-50 flex items-center justify-center p-4" dir="rtl">
         <div className="bg-teal-500 text-white rounded-xl p-8 shadow-lg text-center max-w-md w-full">
            <CheckCircle2 className="w-16 h-16 mx-auto mb-4 text-teal-100" />
            <h2 className="text-2xl font-bold mb-2">تم اعتماد التوثيق!</h2>
            <p className="text-teal-100 text-sm leading-relaxed">
              {workflowStep === 'REQUEST_ENTRY' 
                ? 'تم توجيه طلب التغطية إلى إدارة الاتصال المؤسسي.' 
                : 'تم إغلاق الطلب وحفظ المواد الإعلامية في ملف المشروع بنجاح.'}
            </p>
         </div>
       </div>
    );
  }

  return (
    <div className="min-h-screen bg-gray-50 font-sans text-gray-800 pb-16" dir="rtl">
      <header className="bg-gradient-to-r from-purple-900 to-purple-700 text-white shadow-md pt-6 pb-12 px-4 text-center rounded-b-[2rem] relative z-10">
        <h1 className="text-2xl md:text-3xl font-extrabold mb-1">طلب تصوير وتوثيق (F-52)</h1>
        <h2 className="text-sm md:text-base text-teal-300 font-medium">
          {workflowStep === 'REQUEST_ENTRY' && 'المرحلة 1: إصدار وتوجيه الطلب'}
          {workflowStep === 'MEDIA_EXECUTION' && 'المرحلة 2: الاتصال المؤسسي (إنجاز المهمة ورفع المواد)'}
        </h2>
      </header>

      <main className="container mx-auto px-4 -mt-8 relative z-20 max-w-4xl">
        
        {/* Routing Info */}
        <Card title="مسار التوجيه" icon={ArrowLeftRight}>
          <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
            <Input label="مُقدم الطلب (من)" value={fromDept} onChange={e => setFromDept(e.target.value)} disabled={isReadOnly} required />
            <ReadOnlyField label="مُوجه إلى (إلى)" value={toDept} highlight />
          </div>
        </Card>

        {/* Request Details (Mirrors the exact physical form layout) */}
        <Card title="تفاصيل المهمة والمشروع" icon={Camera}>
          <div className="bg-gray-50 border border-gray-200 p-4 rounded-xl mb-5">
            <label className="block text-sm font-bold text-gray-800 mb-3">مرحلة التوثيق المطلوبة:</label>
            <div className="flex flex-wrap gap-3">
              {phases.map(phase => (
                <label key={phase} className={`flex items-center gap-2 px-3 py-2 border rounded-lg cursor-pointer transition-all ${docPhase === phase ? 'bg-purple-100 border-purple-500 text-purple-900 shadow-sm' : 'bg-white border-gray-300 text-gray-600'}`}>
                  <input type="radio" name="docPhase" value={phase} checked={docPhase === phase} onChange={e => !isReadOnly && setDocPhase(e.target.value)} disabled={isReadOnly} className="hidden" />
                  <span className="font-bold text-xs">{phase}</span>
                </label>
              ))}
            </div>
          </div>

          <div className="bg-white border-2 border-purple-100 p-6 rounded-xl text-gray-800 leading-loose text-justify font-medium shadow-sm mb-4">
            بناءً على موافقة الأسرة على توثيق منزلهم رقم <span className="font-bold text-purple-700 bg-purple-50 px-2 py-0.5 rounded">{pData.caseRef}</span> {' '}
            الواقع في مدينة <span className="font-bold text-purple-700 bg-purple-50 px-2 py-0.5 rounded">{pData.city}</span> {' '}
            حي <span className="font-bold text-purple-700 bg-purple-50 px-2 py-0.5 rounded">{pData.neighborhood}</span>.
            <br className="mb-2"/>
            نحتاج منكم التوثيق بتاريخ:{' '}
            {isReadOnly ? (
              <span className="font-bold text-teal-700 bg-teal-50 px-2 py-0.5 rounded">{existingData.targetDate}</span>
            ) : (
              <input type="date" required value={targetDate} onChange={e => setTargetDate(e.target.value)} className="inline-block px-2 py-1 mx-1 text-sm border border-purple-300 rounded outline-none focus:border-purple-600 bg-purple-50 text-purple-900 font-bold" />
            )}
          </div>
        </Card>

        {/* Stage 2: Execution (Media Dept) */}
        {isMediaStep && (
          <Card title="إبلاغ بإنجاز المهمة (يُعبأ بواسطة الاتصال المؤسسي)" icon={FileText}>
            <div className="flex flex-col gap-1 mb-5">
              <label className="text-xs font-bold text-gray-700">التقرير الإعلامي للإنجاز *</label>
              <textarea 
                required rows={4} 
                value={executionNotes}
                onChange={e => setExecutionNotes(e.target.value)}
                placeholder="نفيدكم بأنه تم إنجاز التغطية الإعلامية بنجاح، حيث تم تصوير..." 
                className="px-3 py-2 text-sm rounded-lg border border-teal-300 focus:border-teal-500 outline-none resize-none bg-teal-50/30 shadow-sm leading-relaxed" 
              />
            </div>

            <h4 className="text-sm font-bold text-gray-800 mb-3 flex items-center gap-2"><LinkIcon size={18}/> المخرجات الإعلامية</h4>
            
            <div className="grid grid-cols-1 md:grid-cols-2 gap-5 mb-4">
              <div className="bg-gray-50 p-4 rounded-xl border border-gray-200">
                <Input label="رابط مجلد التغطية (Google Drive / OneDrive)" type="url" value={mediaDriveLink} onChange={e => setMediaDriveLink(e.target.value)} placeholder="https://drive.google.com/..." />
                <p className="text-xs text-gray-500 mt-2">يفضل وضع رابط المجلد لاحتوائه على الفيديوهات والصور عالية الدقة.</p>
              </div>

              <div className="bg-gray-50 p-4 rounded-xl border border-gray-200 border-dashed">
                <label className="text-xs font-semibold text-gray-700 mb-2 block">الرفع المباشر للصور المصغرة / التقارير</label>
                <label className="cursor-pointer bg-white border border-gray-300 text-gray-700 px-4 py-2 rounded-lg text-sm font-bold flex items-center justify-center gap-2 hover:bg-gray-100">
                  <UploadCloud size={18} /> رفع الملفات
                  <input type="file" multiple accept="image/*,.pdf" className="hidden" onChange={handleFileChange} />
                </label>
              </div>
            </div>

            {mediaFiles.length > 0 && (
              <div className="grid grid-cols-1 sm:grid-cols-2 gap-2 mt-2">
                {mediaFiles.map((f, i) => (
                  <div key={i} className="flex justify-between items-center bg-white p-2 rounded-lg border border-teal-200 text-xs shadow-sm">
                    <span className="truncate font-bold text-teal-800">{f.name}</span>
                    <button type="button" onClick={() => removeFile(i)} className="text-red-500"><X size={16}/></button>
                  </div>
                ))}
              </div>
            )}
          </Card>
        )}

        {/* Read-Only History for Viewers */}
        {workflowStep === 'VIEW_ONLY' && existingData.executionNotes && (
          <Card title="المخرجات الإعلامية (مكتمل)" icon={CheckCircle2}>
            <ReadOnlyField label="التقرير الإعلامي" value={existingData.executionNotes} highlight />
            {existingData.mediaDriveLink && (
              <div className="mt-4 p-3 bg-blue-50 rounded-lg border border-blue-200 flex items-center justify-between">
                <span className="text-sm font-bold text-blue-900">مجلد التغطية:</span>
                <a href={existingData.mediaDriveLink} target="_blank" rel="noreferrer" className="text-sm font-bold text-blue-600 underline">فتح الرابط</a>
              </div>
            )}
          </Card>
        )}

        {/* Action Box */}
        {workflowStep !== 'VIEW_ONLY' && (
          <Card title="الاعتماد الحالي" icon={ShieldAlert}>
            <div className="bg-gray-50 border border-gray-200 p-4 rounded-xl mb-4">
              <div className="flex items-start gap-4 p-3 bg-white rounded-lg border border-gray-200">
                <input type="checkbox" id="pledge" checked={pledge} onChange={(e) => setPledge(e.target.checked)} className="w-5 h-5 mt-0.5 rounded text-purple-600 focus:ring-purple-500 cursor-pointer" />
                <label htmlFor="pledge" className="cursor-pointer select-none">
                  <h4 className="text-sm font-bold text-gray-900 mb-1">إقرار وتوقيع رقمي</h4>
                  <p className="text-xs text-gray-600 font-medium">
                    {workflowStep === 'REQUEST_ENTRY' 
                      ? 'أقر بصفتي مُصدر الطلب بصحة التوجيه وأطلب من قسم الإعلام تنفيذ التوثيق بالموعد المحدد.' 
                      : 'أقر بصفتي ممثل الاتصال المؤسسي بأنه تم إنجاز المهمة وتسليم المخرجات وفق المطلوب.'}
                  </p>
                </label>
              </div>
            </div>

            <button 
              type="button" onClick={() => handleAction('SUBMIT')} 
              disabled={!pledge || isSubmitting || (workflowStep === 'REQUEST_ENTRY' && !targetDate) || (workflowStep === 'MEDIA_EXECUTION' && !executionNotes)}
              className={`w-full py-4 rounded-xl text-base font-bold flex items-center justify-center gap-2 transition-all shadow-md
                ${pledge ? 'bg-gradient-to-r from-teal-600 to-teal-500 text-white hover:from-teal-700' : 'bg-gray-200 text-gray-400 cursor-not-allowed'}`}
            >
              <UserCheck className="w-5 h-5" /> 
              {isSubmitting ? 'جاري التنفيذ...' : (workflowStep === 'REQUEST_ENTRY' ? 'توجيه الطلب لقسم الإعلام' : 'اعتماد الإنجاز وإغلاق الطلب')}
            </button>
          </Card>
        )}
      </main>
    </div>
  );
}
```
