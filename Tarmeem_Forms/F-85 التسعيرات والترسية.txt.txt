```react
import React, { useState } from 'react';
import { 
  Building, Calculator, CheckCircle2, UserCheck, ShieldAlert, 
  Plus, Trash2, Trophy, ArrowLeftRight, CheckCircle, XCircle
} from 'lucide-react';

const Card = ({ title, icon: Icon, children, className = '' }) => (
  <div className={`bg-white rounded-xl shadow-sm border border-gray-200 overflow-hidden mb-4 transition-all duration-300 ${className}`}>
    <div className="bg-gradient-to-r from-purple-800 to-purple-600 px-4 py-2.5 flex items-center gap-2 text-white">
      {Icon && <Icon className="w-5 h-5 text-teal-200" />}
      <h3 className="text-base font-bold">{title}</h3>
    </div>
    <div className="p-4 md:p-5">{children}</div>
  </div>
);

const ReadOnlyField = ({ label, value, highlight = false }) => (
  <div className="flex flex-col gap-1">
    <label className="text-xs font-semibold text-gray-500">{label}</label>
    <div className={`px-3 py-2 text-sm rounded-lg border font-medium
      ${highlight ? 'bg-teal-50 border-teal-200 text-teal-800' : 'bg-gray-100 border-gray-200 text-gray-700'}`}>
      {value || '-'}
    </div>
  </div>
);

const Input = ({ label, type = 'text', value, onChange, placeholder, className = '', required = false }) => (
  <div className={`flex flex-col gap-1 ${className}`}>
    <label className="text-xs font-semibold text-gray-700">{label} {required && <span className="text-red-500">*</span>}</label>
    <input type={type} value={value} onChange={onChange} placeholder={placeholder} required={required} className="px-3 py-1.5 text-sm rounded-lg border border-gray-200 focus:border-purple-500 focus:ring-1 focus:ring-purple-200 outline-none text-gray-800 bg-gray-50" />
  </div>
);

export default function F85_PricingAwardWorkflow({ 
  onSubmit, 
  f08Data = {}, 
  // PM_ENTRY (مدير المشاريع), FINANCE_APPROVAL (المالية), EXECUTIVE_APPROVAL (المدير التنفيذي)
  workflowStep = 'PM_ENTRY', 
  existingData = {} 
}) {
  const [isSubmitting, setIsSubmitting] = useState(false);
  
  // Beneficiary Data (From F-08 or F-02)
  const caseData = Object.keys(f08Data).length > 0 ? f08Data : {
    general: { caseRef: 'CS-9082', familyName: 'عبدالله محمد السالم', cityNeighborhood: 'الدمام - حي النور' },
    projectNumber: 'PRJ-2024-884'
  };

  // State
  const [contractors, setContractors] = useState(existingData.contractors || [
    { id: 1, name: '', amount: '' },
    { id: 2, name: '', amount: '' },
    { id: 3, name: '', amount: '' }
  ]);
  
  const [selectedContractorId, setSelectedContractorId] = useState(existingData.selectedContractorId || '');
  const [notes, setNotes] = useState('');
  const [employeePledge, setEmployeePledge] = useState(false);

  const winningContractor = contractors.find(c => c.id === Number(selectedContractorId));
  const isReadOnly = workflowStep !== 'PM_ENTRY';

  const addContractor = () => {
    setContractors([...contractors, { id: Date.now(), name: '', amount: '' }]);
  };

  const removeContractor = (id) => {
    setContractors(contractors.filter(c => c.id !== id));
    if (selectedContractorId === id) setSelectedContractorId('');
  };

  const updateContractor = (id, field, value) => {
    setContractors(contractors.map(c => c.id === id ? { ...c, [field]: value } : c));
  };

  const handleAction = async (actionType) => {
    if (!employeePledge || isSubmitting) return;
    setIsSubmitting(true);
    
    let payload = { ...existingData };

    // Workflow Routing Logic
    if (workflowStep === 'PM_ENTRY') {
      payload = { 
        ...payload, 
        caseData: caseData.general,
        contractors, 
        selectedContractorId,
        winningContractor,
        status: 'PENDING_FINANCE',
        pmNotes: notes
      };
    } else if (workflowStep === 'FINANCE_APPROVAL') {
      payload = { ...payload, status: actionType === 'APPROVE' ? 'PENDING_EXECUTIVE' : 'RETURNED_TO_PM', financeNotes: notes };
    } else if (workflowStep === 'EXECUTIVE_APPROVAL') {
      payload = { ...payload, status: actionType === 'APPROVE' ? 'AWARDED' : 'RETURNED_TO_PM', executiveNotes: notes };
    }

    if (onSubmit) {
      await onSubmit({ formId: 'F-85', action: actionType, payload });
    }
  };

  if (isSubmitting && !onSubmit) {
    return (
       <div className="min-h-screen bg-gray-50 flex items-center justify-center p-4" dir="rtl">
         <div className="bg-teal-500 text-white rounded-xl p-8 shadow-lg text-center max-w-md w-full">
            <CheckCircle2 className="w-16 h-16 mx-auto mb-4 text-teal-100" />
            <h2 className="text-2xl font-bold mb-2">تم اعتماد الإجراء بنجاح!</h2>
            <p className="text-teal-100 text-sm leading-relaxed">
              تم تحديث حالة محضر الترسية وتوجيهه للمرحلة القادمة في النظام.
            </p>
         </div>
       </div>
    );
  }

  return (
    <div className="min-h-screen bg-gray-50 font-sans text-gray-800 pb-16" dir="rtl">
      <header className="bg-gradient-to-r from-purple-900 to-purple-700 text-white shadow-md pt-6 pb-12 px-4 text-center rounded-b-[2rem] relative z-10">
        <h1 className="text-2xl md:text-3xl font-extrabold mb-1">اعتماد التسعيرات والترسية (F-85)</h1>
        <h2 className="text-sm md:text-base text-teal-300 font-medium">
          {workflowStep === 'PM_ENTRY' && 'المرحلة 1: مدير المشاريع (إدخال عروض الأسعار والترسية)'}
          {workflowStep === 'FINANCE_APPROVAL' && 'المرحلة 2: الإدارة المالية (الاعتماد المحاسبي)'}
          {workflowStep === 'EXECUTIVE_APPROVAL' && 'المرحلة 3: الاعتماد النهائي للمدير التنفيذي'}
        </h2>
      </header>

      <main className="container mx-auto px-4 -mt-8 relative z-20 max-w-4xl">
        
        {/* Beneficiary Info */}
        <Card title="بيانات المشروع" icon={Building}>
          <div className="grid grid-cols-1 md:grid-cols-4 gap-4">
            <ReadOnlyField label="اسم المستفيد" value={caseData.general?.familyName} />
            <ReadOnlyField label="رقم المستفيد (الحالة)" value={caseData.general?.caseRef} />
            <ReadOnlyField label="رقم المشروع" value={caseData.projectNumber || existingData.projectNumber} highlight />
            <ReadOnlyField label="الموقع (المدينة/الحي)" value={caseData.general?.cityNeighborhood} />
          </div>
        </Card>

        {/* Dynamic Contractors List */}
        <Card title="عروض الأسعار للمقاولين المتقدمين" icon={Calculator}>
          {isReadOnly ? (
            <div className="overflow-x-auto">
              <table className="w-full text-right border-collapse mb-2">
                <thead>
                  <tr className="bg-purple-100 text-purple-900 text-sm">
                    <th className="p-3 border border-purple-200 font-bold rounded-tr-lg w-16 text-center">م</th>
                    <th className="p-3 border border-purple-200 font-bold">اسم المقاول / المؤسسة</th>
                    <th className="p-3 border border-purple-200 font-bold rounded-tl-lg">قيمة العرض (ر.س)</th>
                  </tr>
                </thead>
                <tbody>
                  {contractors.filter(c => c.name && c.amount).map((contractor, index) => (
                    <tr key={contractor.id} className="bg-white text-sm">
                      <td className="p-3 border border-gray-200 text-center font-bold">{index + 1}</td>
                      <td className="p-3 border border-gray-200 font-medium">{contractor.name}</td>
                      <td className="p-3 border border-gray-200 font-bold text-teal-700">{Number(contractor.amount).toLocaleString()} ريال</td>
                    </tr>
                  ))}
                </tbody>
              </table>
            </div>
          ) : (
            <>
              <p className="text-sm text-gray-500 mb-4">يُعبأ بواسطة مدير المشاريع: أدخل أسماء المقاولين وإجمالي عروض الأسعار المقدمة (يمكنك إضافة أي عدد).</p>
              <div className="space-y-3 mb-4">
                {contractors.map((contractor, index) => (
                  <div key={contractor.id} className={`flex flex-col md:flex-row gap-3 p-3 rounded-xl border items-center transition-all ${selectedContractorId === contractor.id ? 'bg-purple-50 border-purple-400 shadow-sm' : 'bg-white border-gray-200'}`}>
                    <div className="flex items-center gap-3 w-full md:w-auto flex-1">
                      <span className="font-bold text-gray-400 text-sm bg-gray-100 w-8 h-8 flex items-center justify-center rounded-full shrink-0">{index + 1}</span>
                      <Input label="اسم المقاول / المؤسسة" required className="w-full" value={contractor.name} onChange={e => updateContractor(contractor.id, 'name', e.target.value)} />
                    </div>
                    <div className="flex items-center gap-3 w-full md:w-auto flex-1">
                      <Input label="قيمة العرض (ر.س)" type="number" required className="w-full" value={contractor.amount} onChange={e => updateContractor(contractor.id, 'amount', e.target.value)} />
                      {contractors.length > 1 && (
                        <button type="button" onClick={() => removeContractor(contractor.id)} className="mt-5 text-red-500 hover:bg-red-50 p-2 rounded-lg transition-colors"><Trash2 size={18}/></button>
                      )}
                    </div>
                  </div>
                ))}
              </div>
              <button type="button" onClick={addContractor} className="bg-purple-50 text-purple-700 border border-dashed border-purple-300 px-4 py-2.5 rounded-lg text-sm font-bold flex items-center gap-2 hover:bg-purple-100 transition-colors w-full justify-center">
                <Plus size={18} /> إضافة مقاول آخر للمقارنة
              </button>
            </>
          )}
        </Card>

        {/* Award Decision */}
        <Card title="قرار الترسية واختيار العرض الأنسب" icon={Trophy}>
          {isReadOnly ? (
             <div className="bg-teal-600 text-white p-6 rounded-xl shadow-md flex flex-col md:flex-row items-center justify-between gap-4">
               <div>
                 <h4 className="text-sm text-teal-100 mb-1">المقاول المُعتمد للترسية من قبل مدير المشاريع:</h4>
                 <p className="text-2xl font-extrabold">{existingData.winningContractor?.name}</p>
               </div>
               <div className="bg-white text-teal-800 px-6 py-3 rounded-lg border-2 border-teal-300 shadow-inner text-center">
                 <p className="text-xs font-bold text-teal-600 mb-1">بمبلغ إجمالي</p>
                 <p className="text-xl font-black">{Number(existingData.winningContractor?.amount).toLocaleString()} <span className="text-sm">ريال</span></p>
               </div>
             </div>
          ) : (
            <div className="bg-blue-50 border border-blue-200 p-5 rounded-xl">
              <label className="block text-sm font-bold text-blue-900 mb-3">بصفتك مدير المشاريع، اختر المقاول الأنسب للترسية:</label>
              <select 
                required
                value={selectedContractorId}
                onChange={e => setSelectedContractorId(Number(e.target.value))}
                className="w-full p-3 rounded-lg border border-blue-300 focus:border-blue-600 outline-none text-blue-900 font-bold bg-white shadow-sm"
              >
                <option value="">-- اختر المقاول الفائز --</option>
                {contractors.filter(c => c.name && c.amount).map(c => (
                  <option key={c.id} value={c.id}>{c.name}</option>
                ))}
              </select>
            </div>
          )}
        </Card>

        {/* Action Box & Signatures Map */}
        <Card title="الاعتمادات والموافقات" icon={ShieldAlert}>
          
          <div className="overflow-x-auto mb-6">
            <table className="w-full text-right border-collapse">
              <thead>
                <tr className="bg-gray-100 text-gray-700 text-sm">
                  <th className="p-2 border border-gray-300 font-bold">المسمى</th>
                  <th className="p-2 border border-gray-300 font-bold">الجهة</th>
                  <th className="p-2 border border-gray-300 font-bold">الحالة</th>
                </tr>
              </thead>
              <tbody className="text-sm font-medium">
                <tr>
                  <td className="p-2 border border-gray-200">مدير المشاريع</td>
                  <td className="p-2 border border-gray-200">إدارة المشاريع</td>
                  <td className="p-2 border border-gray-200 text-teal-600">{workflowStep !== 'PM_ENTRY' ? <span className="flex items-center gap-1"><CheckCircle size={14}/> معتمد</span> : 'جاري العمل...'}</td>
                </tr>
                <tr>
                  <td className="p-2 border border-gray-200">مدير الإدارة المالية</td>
                  <td className="p-2 border border-gray-200">الإدارة المالية</td>
                  <td className="p-2 border border-gray-200 text-teal-600">{workflowStep === 'EXECUTIVE_APPROVAL' ? <span className="flex items-center gap-1"><CheckCircle size={14}/> معتمد</span> : 'قيد الانتظار'}</td>
                </tr>
                <tr>
                  <td className="p-2 border border-gray-200">المدير التنفيذي</td>
                  <td className="p-2 border border-gray-200">الإدارة التنفيذية</td>
                  <td className="p-2 border border-gray-200 text-gray-400">قيد الانتظار</td>
                </tr>
              </tbody>
            </table>
          </div>

          <div className="bg-gray-50 border border-gray-200 p-4 rounded-xl mb-4">
            <div className="flex flex-col gap-1 mb-4">
              <label className="text-xs font-semibold text-gray-700">ملاحظات / توصية إضافية ({workflowStep === 'PM_ENTRY' ? 'مدير المشاريع' : workflowStep === 'FINANCE_APPROVAL' ? 'الإدارة المالية' : 'المدير التنفيذي'})</label>
              <textarea rows={2} value={notes} onChange={e => setNotes(e.target.value)} placeholder="اكتب ملاحظاتك للاعتماد هنا..." className="px-3 py-2 text-sm rounded-lg border border-gray-300 focus:border-purple-500 outline-none resize-none" />
            </div>

            <div className="flex items-start gap-4">
              <input 
                type="checkbox" 
                id="employeePledge"
                checked={employeePledge}
                onChange={(e) => setEmployeePledge(e.target.checked)}
                className="w-5 h-5 mt-1 rounded text-purple-600 focus:ring-purple-500 cursor-pointer"
              />
              <label htmlFor="employeePledge" className="cursor-pointer select-none">
                <h4 className="text-sm font-bold text-gray-900 mb-1">إقرار وتوقيع رقمي</h4>
                <p className="text-xs text-gray-600 leading-loose">
                  أقر بصفتي المعتمد الحالي بصحة الإجراءات والموافقة على الترسية للمقاول المرشح، ليتم تحويل المحضر في سلسلة الاعتمادات.
                </p>
              </label>
            </div>
          </div>

          {workflowStep === 'PM_ENTRY' ? (
            <button 
              type="button"
              onClick={() => handleAction('APPROVE')}
              disabled={!employeePledge || !selectedContractorId || isSubmitting}
              className={`w-full py-4 rounded-xl text-base font-bold flex items-center justify-center gap-2 transition-all shadow-md
                ${(employeePledge && selectedContractorId) ? 'bg-gradient-to-r from-teal-600 to-teal-500 hover:from-teal-700 text-white' : 'bg-gray-200 text-gray-400 cursor-not-allowed'}`}
            >
              <UserCheck className="w-5 h-5" /> {isSubmitting ? 'جاري توجيه الاعتماد...' : 'اعتماد مدير المشاريع وتوجيهه للإدارة المالية'}
            </button>
          ) : (
            <div className="flex gap-3">
              <button onClick={() => handleAction('APPROVE')} disabled={!employeePledge || isSubmitting} className="flex-1 bg-green-600 hover:bg-green-700 text-white py-3 rounded-lg font-bold text-sm flex justify-center items-center gap-2">
                <CheckCircle2 size={18} /> موافقة (Approve)
              </button>
              <button onClick={() => handleAction('REJECT')} disabled={!employeePledge || isSubmitting} className="flex-1 bg-red-600 hover:bg-red-700 text-white py-3 rounded-lg font-bold text-sm flex justify-center items-center gap-2">
                <XCircle size={18} /> إعادة لمدير المشاريع (Reject)
              </button>
            </div>
          )}
        </Card>

      </main>
      <style dangerouslySetInnerHTML={{__html: `.animate-fade-in{animation:fadeIn .4s ease-in-out}@keyframes fadeIn{from{opacity:0;transform:translateY(10px)}to{opacity:1;transform:translateY(0)}}`}} />
    </div>
  );
}


```
