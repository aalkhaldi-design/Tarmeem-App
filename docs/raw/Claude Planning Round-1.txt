5. The RBAC Rule (formAwaitsUser, final version)
A user can approve/edit a form iff all of the following hold:
	1.	form.status === 'pending' (not draft, approved, rejected, deferred, completed)
	2.	AND one of:
	•	The user is Super Admin (user.isAdmin === true), OR
	•	The user's role strictly equals form.approvalChain[form.approvalIndex], AND
	•	If form.assigneeId is set, then form.assigneeId === user.id
	3.	Manager override: if the user's role is isManager:true AND the user's department equals the next-role's department, they can act on behalf of a subordinate (used for sick days, recovery).
	4.	Helper override: if user.id ∈ form.data.helpers[], they can edit collaboratively but not finally approve.
	5.	Cross-dept collab: F-20 (Material Supply) explicitly grants edit access to SUPPORT department members while it's pending.
Any user (regardless of role) can view any form on a project that's visible to them — the portal is the gate, the action button is the lock.

// Business-day SLA calculator. Skips Friday & Saturday (Saudi weekend).
import { Timestamp } from 'firebase/firestore';

export type SlaTone = 'ok' | 'warn' | 'late' | 'neutral';
export interface SlaStatus { text: string; tone: SlaTone; remaining: number; daysTaken: number; }

const ONE_DAY_MS = 86_400_000;

/** Count business days between two dates (Sun–Thu working week). */
export function businessDaysBetween(start: Date, end: Date): number {
  if (end <= start) return 0;
  let cursor = new Date(start);
  cursor.setHours(0, 0, 0, 0);
  const endMidnight = new Date(end);
  endMidnight.setHours(0, 0, 0, 0);
  let count = 0;
  while (cursor < endMidnight) {
    const d = cursor.getDay(); // 0=Sun … 6=Sat
    if (d !== 5 && d !== 6) count++; // skip Fri (5) & Sat (6)
    cursor = new Date(cursor.getTime() + ONE_DAY_MS);
  }
  return count;
}

export function slaStatus(stepStartedAtIso: string | null | undefined, slaDays: number | undefined): SlaStatus {
  if (!stepStartedAtIso || !slaDays) return { text: 'بدون SLA', tone: 'neutral', remaining: 0, daysTaken: 0 };
  const start = new Date(stepStartedAtIso);
  const now = new Date();
  const taken = businessDaysBetween(start, now);
  const remaining = slaDays - taken;
  if (remaining < 0)        return { text: `متأخر ${Math.abs(remaining)} يوم عمل`, tone: 'late',    remaining, daysTaken: taken };
  if (remaining === 0)      return { text: 'ينتهي اليوم',                          tone: 'warn',    remaining, daysTaken: taken };
  if (remaining <= 1)       return { text: `متبقّي ${remaining} يوم عمل`,           tone: 'warn',    remaining, daysTaken: taken };
  return                            { text: `متبقّي ${remaining} يوم عمل`,           tone: 'ok',      remaining, daysTaken: taken };
}

/** Convert Firestore Timestamp or ISO string to Date (defensive). */
export const toDate = (v: Timestamp | string | Date | null | undefined): Date | null => {
  if (!v) return null;
  if (v instanceof Date) return v;
  if (typeof v === 'string') return new Date(v);
  if (typeof (v as Timestamp).toDate === 'function') return (v as Timestamp).toDate();
  return null;
};

export const formatRelativeTime = (input: string | Date | null | undefined): string => {
  const d = toDate(input);
  if (!d) return '';
  const diff = Math.floor((Date.now() - d.getTime()) / 1000);
  if (diff < 60) return 'الآن';
  if (diff < 3600) return `قبل ${Math.floor(diff / 60)} دقيقة`;
  if (diff < 86400) return `قبل ${Math.floor(diff / 3600)} ساعة`;
  return `قبل ${Math.floor(diff / 86400)} يوم`;
};

import { FORM_BY_CODE, ROLE_BY_KEY, DEPARTMENTS } from './data';
import type { FormCode, RoleKey, DepartmentKey, FormDef } from './data';
import type { UserProfile } from '../components/Auth';
import type { FormRecord } from '../components/Forms';

/** Can a user originate (create) a form of this code? */
export function formCanBeOriginatedBy(form: FormDef, user: UserProfile): boolean {
  if (user.isAdmin) return true;
  return form.originRoles.includes(user.role as RoleKey);
}

/**
 * Is this pending form awaiting THIS user's action?
 * Strict, layered rule:
 *   1. status must be 'pending'
 *   2. Super-admin override → true
 *   3. Strict role match against approvalChain[approvalIndex]
 *   4. If assigneeId is set, it must equal user.id
 *   5. Special case F-20 (cross-dept collab with SUPPORT)
 *   6. Manager-of-department override (manager can act for absent subordinate)
 */
export function formAwaitsUser(record: FormRecord, user: UserProfile): boolean {
  if (record.status !== 'pending') return false;
  if (user.isAdmin) return true;

  const userRoleDef = ROLE_BY_KEY[user.role as RoleKey];
  if (!userRoleDef) return false;
  const userDept: DepartmentKey | undefined = (user.department as DepartmentKey) || userRoleDef.department;
  const nextRole = record.approvalChain[record.approvalIndex];
  if (!nextRole) return false;
  const nextRoleDef = ROLE_BY_KEY[nextRole];

  // Cross-dept collab: F-20 explicitly allows SUPPORT employees to act
  if (record.code === 'F-20' && userDept === 'SUPPORT') return true;

  // Manager override for own department
  if ((userRoleDef.isManager || user.isManager) && nextRoleDef?.department === userDept) return true;

  // Strict role match
  if (nextRole !== user.role) return false;

  // Assignee gate
  if (record.assigneeId && record.assigneeId !== user.id) return false;

  return true;
}

/** Can the user see a department portal? */
export function portalAccessForRole(user: UserProfile): DepartmentKey[] {
  if (user.isAdmin) return DEPARTMENTS.map(d => d.key);
  const def = ROLE_BY_KEY[user.role as RoleKey];
  if (!def) return [];
  if (def.isExecutive) return DEPARTMENTS.map(d => d.key); // exec sees everything
  return [def.department];
}

/** Can the user view (read-only) a specific form on a project? */
export function canViewForm(rec: FormRecord, user: UserProfile): boolean {
  if (user.isAdmin) return true;
  const allowedDepts = portalAccessForRole(user);
  if (allowedDepts.includes(rec.ownerDept)) return true;
  if ((rec.bridgesTo || []).some(d => allowedDepts.includes(d))) return true;
  return false;
}

/** Helper: is this user a helper assigned to this form? */
export function isFormHelper(rec: FormRecord, user: UserProfile): boolean {
  const helpers: string[] = (rec.data?.helpers as string[]) || [];
  return helpers.includes(user.id);
}

/** Execute a cascade result atomically. */
export async function applyCascade(result: CascadeResult, ctx: CascadeContext): Promise<void> {
  const batch = writeBatch(db);
  const now = serverTimestamp();

  // 1. Project patch
  if (result.projectPatch && ctx.approvedRecord.projectRefId) {
    batch.update(doc(db, 'projects', ctx.approvedRecord.projectRefId), {
      ...result.projectPatch,
      updatedAt: now,
    });
  }

  // 2. Activate forms (draft → pending)
  for (const a of result.activate || []) {
    batch.update(doc(db, 'forms', a.formId), {
      status: 'pending',
      stepStartedAt: now,
      assigneeId: a.assigneeId ?? null,
      updatedAt: now,
    });
  }

  // 3. Auto-complete forms
  for (const a of result.autoComplete || []) {
    batch.update(doc(db, 'forms', a.formId), {
      status: 'completed',
      approvals: [{ role: 'SYSTEM' as RoleKey, actorId: 'system', actorName: 'النظام', at: new Date().toISOString(), decision: 'approved', note: a.note }],
      updatedAt: now,
    });
  }

  // 4. Create new forms
  const newFormRefs: DocumentReference[] = [];
  for (const cf of result.createForms || []) {
    const def = FORM_BY_CODE[cf.code];
    const ref = doc(collection(db, 'forms'));
    batch.set(ref, {
      code: cf.code,
      title: cf.title || def.title,
      projectId: ctx.approvedRecord.projectId || null,
      projectRefId: ctx.approvedRecord.projectRefId || null,
      beneficiaryName: ctx.approvedRecord.beneficiaryName || '',
      status: 'pending',
      approvalIndex: 0,
      approvalChain: def.approvalChain,
      approvals: [],
      createdBy: 'system',
      createdByName: 'النظام',
      createdByRole: 'SYSTEM' as RoleKey,
      createdAt: now,
      updatedAt: now,
      ownerDept: def.ownerDept,
      bridgesTo: def.bridgesTo || [],
      notes: '',
      data: cf.data,
      assigneeId: cf.assigneeId ?? null,
      files: [],
      stepStartedAt: now,
    });
    newFormRefs.push(ref);
  }

  await batch.commit();

  // 5. Notifications (after batch — they don't need atomicity with the workflow)
  for (const n of result.notify || []) {
    await addDoc(collection(db, 'notifications'), {
      ...n,
      readBy: [],
      createdAt: new Date().toISOString(),
    });
  }
}

## `src/hooks/useAuth.ts`

```ts
import { useState, useEffect } from 'react';
import { onAuthStateChanged, User } from 'firebase/auth';
import { doc, onSnapshot, setDoc, serverTimestamp } from 'firebase/firestore';
import { auth, db } from '../lib/firebase';
import { isAdminEmail } from '../lib/data';
import type { UserProfile } from '../components/Auth';

export function useAuth() {
  const [firebaseUser, setFirebaseUser] = useState<User | null>(null);
  const [profile, setProfile] = useState<UserProfile | null>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    const unsub = onAuthStateChanged(auth, (u) => {
      setFirebaseUser(u);
      if (!u) { setProfile(null); setLoading(false); }
    });
    return unsub;
  }, []);

  // Bind only to uid (primitive) to avoid effect thrashing.
  const uid = firebaseUser?.uid;
  const email = firebaseUser?.email;

  useEffect(() => {
    if (!uid) return;
    const userRef = doc(db, 'users', uid);
    const unsub = onSnapshot(
      userRef,
      async (snap) => {
        if (!snap.exists()) {
          const admin = isAdminEmail(email);
          await setDoc(userRef, {
            email: (email || '').toLowerCase().trim(),
            fullName: email || 'مستخدم جديد',
            role: admin ? 'EXEC_DIRECTOR' : 'PENDING',
            department: admin ? 'EXEC' : null,
            isAdmin: admin,
            isManager: false,
            status: admin ? 'active' : 'pending',
            createdAt: serverTimestamp(),
          });
          return; // snapshot will re-fire with the new doc
        }
        const data = snap.data();
        // Legacy auto-promote: known admin email but flag missing
        if (isAdminEmail(email) && !data.isAdmin) {
          await setDoc(userRef, { isAdmin: true, status: 'active' }, { merge: true });
          return;
        }
        setProfile({ id: uid, ...(data as Omit<UserProfile, 'id'>) });
        setLoading(false);
      },
      () => setLoading(false),
    );
    return unsub;
  }, [uid, email]);

  return { firebaseUser, profile, loading };
}
```

## `src/hooks/useFirestoreCollection.ts`

```ts
import { useState, useEffect } from 'react';
import { collection, onSnapshot, query, orderBy, limit as fbLimit, QueryConstraint } from 'firebase/firestore';
import { db } from '../lib/firebase';

interface Options {
  defaultLimit?: number;
  orderByField?: string;
  enabled?: boolean;
  /** Stable cache key — change this to re-subscribe with new constraints. */
  key?: string;
}

export function useFirestoreCollection<T>(
  collectionName: string,
  constraints: QueryConstraint[] = [],
  options: Options = {},
) {
  const { defaultLimit = 500, orderByField = 'updatedAt', enabled = true, key = '' } = options;
  const [data, setData] = useState<T[]>([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    if (!enabled) { setLoading(false); return; }
    const q = query(
      collection(db, collectionName),
      ...constraints,
      orderBy(orderByField, 'desc'),
      fbLimit(defaultLimit),
    );
    const unsub = onSnapshot(
      q,
      (snap) => {
        setData(snap.docs.map((d) => ({ id: d.id, ...d.data() } as unknown as T)));
        setLoading(false);
      },
      (err) => {
        console.error(`[${collectionName}] snapshot error`, err);
        setLoading(false);
      },
    );
    return unsub;
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [collectionName, key, defaultLimit, orderByField, enabled]);

  return { data, loading };
}
```

```tsx
import React, { createContext, useCallback, useContext, useEffect, useState } from 'react';

type Theme = 'light' | 'dark';
interface Ctx { theme: Theme; toggleTheme: () => void; setTheme: (t: Theme) => void; }

const ThemeContext = createContext<Ctx>({ theme: 'dark', toggleTheme: () => {}, setTheme: () => {} });
const STORAGE_KEY = 'tarmeem.theme';

export const ThemeProvider: React.FC<{ children: React.ReactNode }> = ({ children }) => {
  const [theme, setThemeState] = useState<Theme>(() => {
    if (typeof window === 'undefined') return 'dark';
    return (localStorage.getItem(STORAGE_KEY) as Theme) || 'dark';
  });

  useEffect(() => {
    const root = document.documentElement;
    root.classList.toggle('dark', theme === 'dark');
    root.style.colorScheme = theme;
    try { localStorage.setItem(STORAGE_KEY, theme); } catch { /* ignore */ }
  }, [theme]);

  const setTheme = useCallback((t: Theme) => setThemeState(t), []);
  const toggleTheme = useCallback(() => setThemeState((t) => (t === 'dark' ? 'light' : 'dark')), []);

  return <ThemeContext.Provider value={{ theme, toggleTheme, setTheme }}>{children}</ThemeContext.Provider>;
};

export const useTheme = () => useContext(ThemeContext);
```

```tsx
        status: newStatus,
        approvals: arrayUnion(approval),
        stepStartedAt: newStatus === 'pending' ? serverTimestamp() : rec.stepStartedAt ?? serverTimestamp(),

        updatedAt: serverTimestamp(),
      });
      await batch.commit();

      if (newStatus === 'approved') {
        const trigger = TRIGGER_MAP[rec.code];
        if (trigger) {

            ...rec,
            data: { ...(rec.data || {}), ...(dataPatch || {}) },
            approvalIndex: nextIndex,
            status: 'approved',
          };
          const ctx = { forms, projects, users, actor: user, approvedRecord: updatedRec, dataPatch };
          const result = trigger(ctx);
          await applyCascade(result, ctx);
        }
      } else {
        await notifyNextStep({ ...rec, approvalIndex: nextIndex }, users);
      }
    },

    rejectForm: async (formId, user, note) => {
      const rec = stateRef.current.forms.find((f) => f.id === formId);
      if (!rec) throw new Error('Form not found');
      if (!formAwaitsUser(rec, user)) throw new Error('Not authorized to reject');
      await updateDoc(doc(db, 'forms', formId), {
        status: 'rejected',
        approvals: arrayUnion({
          role: rec.approvalChain[rec.approvalIndex],
          actorId: user.id, actorName: user.fullName,
          at: new Date().toISOString(), decision: 'rejected', note: note || '',
        }),
        updatedAt: serverTimestamp(),
      });
    },

    deferForm: async (formId, user, note) => {
      const rec = stateRef.current.forms.find((f) => f.id === formId);
      if (!rec) throw new Error('Form not found');
      if (!formAwaitsUser(rec, user)) throw new Error('Not authorized to defer');
      await updateDoc(doc(db, 'forms', formId), {
        status: 'deferred',
        approvals: arrayUnion({
          role: rec.approvalChain[rec.approvalIndex],

      });
    },

    updateFormData: async (formId, dataPatch) => {
      const rec = stateRef.current.forms.find((f) => f.id === formId);

      await updateDoc(doc(db, 'forms', formId), {
        data: { ...(rec.data || {}), ...dataPatch },
```
