Note on the approveForm cascade timing

Note on useFirestoreCollection's key contract
The key parameter is the only thing that re-subscribes. If a consumer passes new constraints without bumping key, the listener stays on the old query and quietly returns stale data. This is a footgun. Either:
	•	Add a console.warn in dev mode when the JSON of constraints changes but key doesn't, or
	•	Make key non-optional in the type (force callers to think about it).
I'd take option 2.

```ts
// ============================================================================
export async function applyCascade(result: CascadeResult, ctx: CascadeContext): Promise<void> {
  const batch = writeBatch(db);
  const now = serverTimestamp();

  if (result.projectPatch && ctx.approvedRecord.projectRefId) {
    batch.update(doc(db, 'projects', ctx.approvedRecord.projectRefId), {
      ...result.projectPatch,
      updatedAt: now,
    });
  }

  for (const a of result.activate || []) {
    batch.update(doc(db, 'forms', a.formId), {
      status: 'pending',
      stepStartedAt: now,
      assigneeId: a.assigneeId ?? null,
      updatedAt: now,
    });
  }

  for (const a of result.autoComplete || []) {
    batch.update(doc(db, 'forms', a.formId), {
      status: 'completed',
      approvals: [{
        role: 'SYSTEM' as RoleKey,
        actorId: 'system',
        actorName: 'النظام',
        at: new Date().toISOString(),
        decision: 'approved',
        note: a.note,
      }],
      updatedAt: now,
    });
  }

  const newFormRefs: DocumentReference[] = [];
  for (const cf of result.createForms || []) {
    const def = FORM_BY_CODE[cf.code];
    const ref = doc(collection(db, 'forms'));
    batch.set(ref, {
      code: cf.code,
      title: cf.title || def.title,
      projectId: ctx.approvedRecord.projectId || null,
      projectRefId: ctx.approvedRecord.projectRefId || null,
      beneficiaryName: ctx.approvedRecord.beneficiaryName || '',
      status: 'pending',
      approvalIndex: 0,
      approvalChain: def.approvalChain,
      approvals: [],
      createdBy: 'system',
      createdByName: 'النظام',
      createdByRole: 'SYSTEM' as RoleKey,
      createdAt: now,
      updatedAt: now,
      ownerDept: def.ownerDept,
      bridgesTo: def.bridgesTo || [],
      notes: '',
      data: cf.data,
      assigneeId: cf.assigneeId ?? null,
      files: [],
      stepStartedAt: now,
    });
    newFormRefs.push(ref);
  }

  await batch.commit();

  for (const n of result.notify || []) {
    if (!n.recipients.length) continue;
    await addDoc(collection(db, 'notifications'), {
      ...n,
      readBy: [],
      createdAt: new Date().toISOString(),
    });
  }
}
```

## `src/lib/sla.ts` — full re-ship (adds hold-elapsed-time subtraction)

```ts
import { Timestamp } from 'firebase/firestore';
import type { ProjectStatus } from './data';

export type SlaTone = 'ok' | 'warn' | 'late' | 'neutral' | 'frozen';
export interface SlaStatus { text: string; tone: SlaTone; remaining: number; daysTaken: number; }

const ONE_DAY_MS = 86_400_000;

/** Count business days between two dates (Sun–Thu working week). */
export function businessDaysBetween(start: Date, end: Date): number {
  if (end <= start) return 0;
  let cursor = new Date(start); cursor.setHours(0, 0, 0, 0);
  const endMidnight = new Date(end); endMidnight.setHours(0, 0, 0, 0);
  let count = 0;
  while (cursor < endMidnight) {
    const d = cursor.getDay();
    if (d !== 5 && d !== 6) count++; // skip Fri/Sat
    cursor = new Date(cursor.getTime() + ONE_DAY_MS);
  }
  return count;
}

/** Optional project-level hold state to freeze/subtract from SLA. */
export interface ProjectHoldState {
  status?: ProjectStatus;
  holdedAt?: string | Timestamp | Date | null;
  /** Business days already accumulated across past hold periods (recorded on resume). */
  pastHoldDays?: number;
}

export function slaStatus(
  stepStartedAtIso: string | null | undefined,
  slaDays: number | undefined,
  hold?: ProjectHoldState,
): SlaStatus {
  if (!stepStartedAtIso || !slaDays) {
    return { text: 'بدون SLA', tone: 'neutral', remaining: 0, daysTaken: 0 };
  }
  const start = new Date(stepStartedAtIso);

  // Currently on hold → SLA is frozen at the moment of hold start.
  if (hold?.status === 'on_hold') {
    const holdStart = toDate(hold.holdedAt) || new Date();
    const taken = Math.max(0, businessDaysBetween(start, holdStart) - (hold.pastHoldDays || 0));
    const remaining = slaDays - taken;
    return { text: 'متوقف — في وضع التعليق', tone: 'frozen', remaining, daysTaken: taken };
  }

  const taken = Math.max(0, businessDaysBetween(start, new Date()) - (hold?.pastHoldDays || 0));
  const remaining = slaDays - taken;
  if (remaining < 0)        return { text: `متأخر ${Math.abs(remaining)} يوم عمل`, tone: 'late', remaining, daysTaken: taken };
  if (remaining === 0)      return { text: 'ينتهي اليوم',                          tone: 'warn', remaining, daysTaken: taken };
  if (remaining <= 1)       return { text: `متبقّي ${remaining} يوم عمل`,           tone: 'warn', remaining, daysTaken: taken };
  return                            { text: `متبقّي ${remaining} يوم عمل`,           tone: 'ok',   remaining, daysTaken: taken };
}

export const toDate = (v: Timestamp | string | Date | null | undefined): Date | null => {
  if (!v) return null;
  if (v instanceof Date) return v;
  if (typeof v === 'string') return new Date(v);
  if (typeof (v as Timestamp).toDate === 'function') return (v as Timestamp).toDate();
  return null;
};

export const formatRelativeTime = (input: string | Date | null | undefined): string => {
  const d = toDate(input); if (!d) return '';
  const diff = Math.floor((Date.now() - d.getTime()) / 1000);
  if (diff < 60) return 'الآن';
  if (diff < 3600) return `قبل ${Math.floor(diff / 60)} دقيقة`;
  if (diff < 86400) return `قبل ${Math.floor(diff / 3600)} ساعة`;
  return `قبل ${Math.floor(diff / 86400)} يوم`;
};

/** Business days elapsed during a hold period — call on Resume and add to project.pastHoldDays. */
export function holdElapsedBusinessDays(holdedAt: string | Date, resumedAt: string | Date = new Date()): number {
  const s = typeof holdedAt === 'string' ? new Date(holdedAt) : holdedAt;
  const e = typeof resumedAt === 'string' ? new Date(resumedAt) : resumedAt;
  return businessDaysBetween(s, e);
}
```

## `src/components/ThemeProvider.tsx`

```tsx
import React, { createContext, useCallback, useContext, useEffect, useState } from 'react';

type Theme = 'light' | 'dark';
interface Ctx { theme: Theme; toggleTheme: () => void; setTheme: (t: Theme) => void; }

const ThemeContext = createContext<Ctx>({ theme: 'dark', toggleTheme: () => {}, setTheme: () => {} });
const STORAGE_KEY = 'tarmeem-theme';

export const ThemeProvider: React.FC<{ children: React.ReactNode }> = ({ children }) => {
  const [theme, setThemeState] = useState<Theme>(() => {
    if (typeof window === 'undefined') return 'dark';
    return (localStorage.getItem(STORAGE_KEY) as Theme) || 'dark';
  });

  useEffect(() => {
    document.documentElement.classList.toggle('dark', theme === 'dark');
    try { localStorage.setItem(STORAGE_KEY, theme); } catch { /* ignore */ }
  }, [theme]);

  const setTheme = useCallback((t: Theme) => setThemeState(t), []);
  const toggleTheme = useCallback(() => setThemeState((t) => (t === 'dark' ? 'light' : 'dark')), []);

  return <ThemeContext.Provider value={{ theme, toggleTheme, setTheme }}>{children}</ThemeContext.Provider>;
};

export const useTheme = () => useContext(ThemeContext);
```

## `src/hooks/useFirestoreCollection.ts` — tightened key contract

```ts
import { useEffect, useState } from 'react';
import { collection, onSnapshot, query, orderBy, limit as fbLimit, QueryConstraint } from 'firebase/firestore';
import { db } from '../lib/firebase';

interface Options {
  defaultLimit?: number;
  orderByField?: string;
  enabled?: boolean;
  /** REQUIRED: stable cache key. Bump it to re-subscribe with new constraints.
   *  Footgun avoidance — if you pass new constraints without bumping the key,
   *  the listener stays on the old query and silently returns stale data. */
  key: string;
}

export function useFirestoreCollection<T>(
  collectionName: string,
  constraints: QueryConstraint[] = [],
  options: Options,
) {
  const { defaultLimit = 500, orderByField = 'updatedAt', enabled = true, key } = options;
  const [data, setData] = useState<T[]>([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    if (!enabled) { setLoading(false); return; }
    const q = query(
      collection(db, collectionName),
      ...constraints,
      orderBy(orderByField, 'desc'),
      fbLimit(defaultLimit),
    );
    const unsub = onSnapshot(
      q,
      (snap) => {
        setData(snap.docs.map((d) => ({ id: d.id, ...d.data() } as unknown as T)));
        setLoading(false);
      },
      (err) => {
        console.error(`[${collectionName}] snapshot error`, err);
        setLoading(false);
      },
    );
    return unsub;
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [collectionName, key, defaultLimit, orderByField, enabled]);

  return { data, loading };
}
```

```tsx
      } else {

          recipients: [rec.createdBy], readBy: [],
          text: `نموذج ${rec.code} مرفوض للتعديل — السبب: ${note || 'بدون سبب محدد'}`,

        status: 'declined',
        approvals: arrayUnion({
          role: rec.approvalChain[rec.approvalIndex],

          at: new Date().toISOString(), decision: 'declined', note: note || '',
        }),
        updatedAt: serverTimestamp(),
      });

      const cascade = DECLINE_MAP[rec.code];
      if (cascade) {
        const updatedRec: FormRecord = { ...rec, status: 'declined' };

/** Notify users who can act on the form's next step.
 *  Admin spam fix: admins only get pinged if they share the next-role's department
 *  (i.e., acting in-line). Fallback to all admins only when no one holds the role. */
async function notifyNextStep(rec: FormRecord, users: UserProfile[]): Promise<void> {
  const nextRole = rec.approvalChain[rec.approvalIndex];
  if (!nextRole) return;
  const def = ROLE_BY_KEY[nextRole as RoleKey];

  let recipients = users
    .filter((u) => u.status === 'active')
    .filter((u) => {
      if (u.role === nextRole) return true;
      if (u.isAdmin && def?.department && u.department === def.department) return true;
      return false;
    })
    .map((u) => u.id);

  // Fallback: if nobody holds the role, ping all active admins so work doesn't stall.
  if (recipients.length === 0) {
    recipients = users.filter((u) => u.isAdmin && u.status === 'active').map((u) => u.id);
  }
  if (!recipients.length) return;

  await addDoc(collection(db, 'notifications'), {
    recipients,
    readBy: [],
    text: `بانتظار اعتمادك: ${rec.title} (${rec.code})`,
    subject: 'مهمة جديدة',
    type: 'form_pending',
    link: rec.id,
    createdAt: new Date().toISOString(),
  });
}
```

### `src/lib/sla.ts` — full re-ship (frozen-while-late text added)

```ts
import { Timestamp } from 'firebase/firestore';
import type { ProjectStatus } from './data';

export type SlaTone = 'ok' | 'warn' | 'late' | 'neutral' | 'frozen';
export interface SlaStatus { text: string; tone: SlaTone; remaining: number; daysTaken: number; }

const ONE_DAY_MS = 86_400_000;

export function businessDaysBetween(start: Date, end: Date): number {
  if (end <= start) return 0;
  let cursor = new Date(start); cursor.setHours(0, 0, 0, 0);
  const endMidnight = new Date(end); endMidnight.setHours(0, 0, 0, 0);
  let count = 0;
  while (cursor < endMidnight) {
    const d = cursor.getDay();
    if (d !== 5 && d !== 6) count++;
    cursor = new Date(cursor.getTime() + ONE_DAY_MS);
  }
  return count;
}

export interface ProjectHoldState {
  status?: ProjectStatus;
  holdedAt?: string | Timestamp | Date | null;
  pastHoldDays?: number;
}

export function slaStatus(
  stepStartedAtIso: string | null | undefined,
  slaDays: number | undefined,
  hold?: ProjectHoldState,
): SlaStatus {
  if (!stepStartedAtIso || !slaDays) {
    return { text: 'بدون SLA', tone: 'neutral', remaining: 0, daysTaken: 0 };
  }
  const start = new Date(stepStartedAtIso);

  if (hold?.status === 'on_hold') {
    const holdStart = toDate(hold.holdedAt) || new Date();
    const taken = Math.max(0, businessDaysBetween(start, holdStart) - (hold.pastHoldDays || 0));
    const remaining = slaDays - taken;
    if (remaining < 0) {
      return { text: `متوقف — كان متأخراً ${Math.abs(remaining)} يوم`, tone: 'frozen', remaining, daysTaken: taken };
    }
    return { text: 'متوقف — في وضع التعليق', tone: 'frozen', remaining, daysTaken: taken };
  }

  const taken = Math.max(0, businessDaysBetween(start, new Date()) - (hold?.pastHoldDays || 0));
  const remaining = slaDays - taken;
  if (remaining < 0)  return { text: `متأخر ${Math.abs(remaining)} يوم عمل`, tone: 'late', remaining, daysTaken: taken };
  if (remaining === 0)return { text: 'ينتهي اليوم',                          tone: 'warn', remaining, daysTaken: taken };
  if (remaining <= 1) return { text: `متبقّي ${remaining} يوم عمل`,           tone: 'warn', remaining, daysTaken: taken };
  return                     { text: `متبقّي ${remaining} يوم عمل`,           tone: 'ok',   remaining, daysTaken: taken };
}

export const toDate = (v: Timestamp | string | Date | null | undefined): Date | null => {
  if (!v) return null;
  if (v instanceof Date) return v;
  if (typeof v === 'string') return new Date(v);
  if (typeof (v as Timestamp).toDate === 'function') return (v as Timestamp).toDate();
  return null;
};

export const formatRelativeTime = (input: string | Date | null | undefined): string => {
  const d = toDate(input); if (!d) return '';
  const diff = Math.floor((Date.now() - d.getTime()) / 1000);
  if (diff < 60) return 'الآن';
  if (diff < 3600) return `قبل ${Math.floor(diff / 60)} دقيقة`;
  if (diff < 86400) return `قبل ${Math.floor(diff / 3600)} ساعة`;
  return `قبل ${Math.floor(diff / 86400)} يوم`;
};

export function holdElapsedBusinessDays(holdedAt: string | Date, resumedAt: string | Date = new Date()): number {
  const s = typeof holdedAt === 'string' ? new Date(holdedAt) : holdedAt;
  const e = typeof resumedAt === 'string' ? new Date(resumedAt) : resumedAt;
  return businessDaysBetween(s, e);
}
```

```ts
// ============================================================================
// applyCascade — commits all cascade effects in one writeBatch.
// ============================================================================
export async function applyCascade(result: CascadeResult, ctx: CascadeContext): Promise<void> {
  const batch = writeBatch(db);
  const now = serverTimestamp();

  if (result.projectPatch && ctx.approvedRecord.projectRefId) {
    batch.update(doc(db, 'projects', ctx.approvedRecord.projectRefId), {
      ...result.projectPatch,
      updatedAt: now,
    });
  }

  for (const a of result.activate || []) {
    batch.update(doc(db, 'forms', a.formId), {
      status: 'pending',
      stepStartedAt: now,
      assigneeId: a.assigneeId ?? null,
      updatedAt: now,
    });
  }

  // Defensive: use arrayUnion (forms may have an empty approvals[] from creation)
  // and stamp completedAt so the timeline can show when the skip happened.
  for (const a of result.autoComplete || []) {
    batch.update(doc(db, 'forms', a.formId), {
      status: 'completed',
      approvals: arrayUnion({
        role: 'SYSTEM' as RoleKey,
        actorId: 'system',
        actorName: 'النظام',
        at: new Date().toISOString(),
        decision: 'approved',
        note: a.note,
      }),
      completedAt: now,
      updatedAt: now,
    });
  }

  for (const cf of result.createForms || []) {
    const def = FORM_BY_CODE[cf.code];
    const ref = doc(collection(db, 'forms'));
    batch.set(ref, {
      code: cf.code,
      title: cf.title || def.title,
      projectId: ctx.approvedRecord.projectId || null,
      projectRefId: ctx.approvedRecord.projectRefId || null,
      beneficiaryName: ctx.approvedRecord.beneficiaryName || '',
      status: 'pending',
      approvalIndex: 0,
      approvalChain: def.approvalChain,
      approvals: [],
      createdBy: 'system',
      createdByName: 'النظام',
      createdByRole: 'SYSTEM' as RoleKey,
      createdAt: now,
      updatedAt: now,
      ownerDept: def.ownerDept,
      bridgesTo: def.bridgesTo || [],
      notes: '',
      data: cf.data,
      assigneeId: cf.assigneeId ?? null,
      files: [],
      stepStartedAt: now,
    });
  }

  await batch.commit();

  for (const n of result.notify || []) {
    if (!n.recipients.length) continue;
    await addDoc(collection(db, 'notifications'), {
      ...n,
      readBy: [],
      createdAt: new Date().toISOString(),
    });
  }
}
```

```tsx
// FormActionBar — Approve / Reject / Decline / Defer

type ActionMode = null | 'approve' | 'reject' | 'decline' | 'defer';

const FormActionBar: React.FC<{
  rec: FormRecord;
  user: UserProfile;
  api: FormsApiLike;
  awaitsMe: boolean;
  canDecline: boolean;
  approvalPayload?: Record<string, unknown>;
  canApprove?: () => string | null;
}> = ({ rec, user, api, awaitsMe, canDecline, approvalPayload, canApprove }) => {
  const [mode, setMode] = useState<ActionMode>(null);
  const [note, setNote] = useState('');
  const [busy, setBusy] = useState(false);
  const [error, setError] = useState<string | null>(null);

  if (!awaitsMe) {
    return (
      <div className="mt-6 pt-4 border-t border-border-subtle text-xs text-fg-faint flex items-center gap-2">
        <AlertTriangle className="w-3.5 h-3.5" /> هذا النموذج بانتظار شخص آخر — يمكنك القراءة فقط.
      </div>
    );
  }

  const submit = async () => {
    setError(null);
    if (mode === 'approve' && canApprove) {
      const err = canApprove();
      if (err) { setError(err); return; }
    }
    if ((mode === 'decline' || mode === 'reject') && !note.trim()) {
      setError(mode === 'decline' ? 'السبب مطلوب للرفض النهائي' : 'السبب مطلوب لإعادة النموذج للتعديل');
      return;
    }
    setBusy(true);
    try {
      if      (mode === 'approve') await api.approveForm(rec.id, user, note.trim(), approvalPayload);
      else if (mode === 'reject')  await api.rejectForm (rec.id, user, note.trim());
      else if (mode === 'decline') await api.declineForm(rec.id, user, note.trim());
      else if (mode === 'defer')   await api.deferForm  (rec.id, user, note.trim());
      setMode(null);
      setNote('');
    } catch (err) {
      setError((err as Error).message || 'تعذّر تنفيذ العملية');
    } finally {
      setBusy(false);
    }
  };

  return (
    <div className="mt-6 pt-4 border-t border-border-subtle">
      {!mode ? (
        <div className="flex flex-wrap gap-2">
          <button onClick={() => setMode('approve')} className="px-4 py-2 rounded-lg bg-emerald-600 text-white text-sm font-medium hover:bg-emerald-700 flex items-center gap-1.5">

            </button>
          )}
          <button onClick={() => setMode('defer')} className="px-4 py-2 rounded-lg bg-surface-down text-fg-muted text-sm font-medium hover:bg-border-subtle flex items-center gap-1.5">
            <Clock className="w-4 h-4" /> تأجيل
          </button>
        </div>
      ) : (
        <div className="space-y-3">
          {mode === 'decline' && (
            <div className="p-3 rounded-lg bg-red-50 dark:bg-red-900/20 border border-red-200 dark:border-red-800 text-sm text-red-700 dark:text-red-300 flex items-start gap-2">

            </div>
          )}
          <TextArea
            label={
              mode === 'approve' ? 'ملاحظة الاعتماد (اختياري)' :
              mode === 'reject'  ? 'سبب الإرجاع للتعديل (مطلوب)' :
              mode === 'decline' ? 'سبب الرفض النهائي (مطلوب)' :
                                   'سبب التأجيل (اختياري)'
            }
            value={note}
            onChange={(e) => setNote(e.target.value)}
            rows={3}
            disabled={busy}

            <button

              className={`px-4 py-2 rounded-lg text-white text-sm font-medium disabled:opacity-50 ${
                mode === 'approve'  ? 'bg-emerald-600 hover:bg-emerald-700' :
```

```tsx
import React, { useState } from 'react';
import { ref, uploadBytesResumable, getDownloadURL } from 'firebase/storage';
import { arrayUnion, doc, serverTimestamp, updateDoc } from 'firebase/firestore';
import { Upload, FileText, Loader2, Image as ImageIcon } from 'lucide-react';
import { db, storage } from '../../lib/firebase';
import type { UserProfile } from '../Auth';

export interface UploadedFile {
  name: string;
  url: string;
  size: number;
  mimeType: string;
  uploadedBy: string;
  uploadedAt: string;
  storagePath: string;
}

interface Props {
  formId: string;
  projectRefId: string;
  user: UserProfile;
  files: UploadedFile[];
  accept?: string;
  maxSizeMb?: number;
  disabled?: boolean;
}

export const FileUploader: React.FC<Props> = ({
  formId, projectRefId, user, files, accept, maxSizeMb = 20, disabled,
}) => {
  const [progress, setProgress] = useState<Record<string, number>>({});
  const [error, setError] = useState<string | null>(null);

  const onPick = (e: React.ChangeEvent<HTMLInputElement>) => {
    setError(null);
    const list = e.target.files;
    if (!list) return;
    Array.from(list).forEach(uploadOne);
    e.target.value = '';
  };

  const uploadOne = (file: File) => {
    if (file.size > maxSizeMb * 1024 * 1024) {
      setError(`${file.name}: حجم الملف يتجاوز ${maxSizeMb} م.ب`);
      return;
    }
    const uuid = (typeof crypto !== 'undefined' && 'randomUUID' in crypto)
      ? crypto.randomUUID()
      : `${Date.now()}-${Math.random().toString(36).slice(2)}`;
    const path = `projects/${projectRefId}/forms/${formId}/${uuid}-${file.name}`;
    const storageRef = ref(storage, path);
    const task = uploadBytesResumable(storageRef, file);

    task.on(
      'state_changed',
      (snap) => {
        const pct = Math.round((snap.bytesTransferred / snap.totalBytes) * 100);
        setProgress(p => ({ ...p, [file.name]: pct }));
      },
      (err) => {
        setError(`${file.name}: ${err.message}`);
        setProgress(p => { const { [file.name]: _, ...rest } = p; return rest; });
      },
      async () => {
        const url = await getDownloadURL(task.snapshot.ref);
        const meta: UploadedFile = {
          name: file.name, url,
          size: file.size,
          mimeType: file.type || 'application/octet-stream',
          uploadedBy: user.id,
          uploadedAt: new Date().toISOString(),
          storagePath: path,
        };
        await updateDoc(doc(db, 'forms', formId), {
          files: arrayUnion(meta),
          updatedAt: serverTimestamp(),
        });
        setProgress(p => { const { [file.name]: _, ...rest } = p; return rest; });
      },
    );
  };

  return (
    <div className="space-y-3">
      <label className={`flex items-center gap-3 p-3 rounded-lg border-2 border-dashed border-border-default cursor-pointer hover:border-brand-teal transition-colors ${disabled ? 'opacity-50 cursor-not-allowed' : ''}`}>
        <input type="file" multiple accept={accept} onChange={onPick} disabled={disabled} className="hidden" />
        <Upload className="w-5 h-5 text-fg-muted" />
        <div className="flex-1">
          <div className="text-sm font-medium text-fg">رفع ملفات</div>
          <div className="text-xs text-fg-faint">السحب والإفلات أو الضغط للاختيار · حد أقصى {maxSizeMb} م.ب لكل ملف</div>
        </div>
      </label>

      {error && <p className="text-xs text-red-600 dark:text-red-400">{error}</p>}

      {Object.entries(progress).map(([name, pct]) => (
        <div key={name} className="flex items-center gap-2 text-xs text-fg-muted">
          <Loader2 className="w-3.5 h-3.5 animate-spin shrink-0" />
          <span className="truncate flex-1">{name}</span>
          <span className="shrink-0">{pct}%</span>
        </div>
      ))}

      {files.length > 0 && (
        <ul className="space-y-1.5">
          {files.map(f => (
            <li key={f.storagePath} className="flex items-center gap-2 text-xs">
              {f.mimeType.startsWith('image/')
                ? <ImageIcon className="w-3.5 h-3.5 text-brand-teal shrink-0" />
                : <FileText  className="w-3.5 h-3.5 text-fg-muted shrink-0" />}
              <a href={f.url} target="_blank" rel="noreferrer" className="flex-1 truncate text-brand-teal hover:underline">{f.name}</a>
              <span className="text-fg-faint shrink-0">{(f.size / 1024).toFixed(0)} ك.ب</span>
            </li>
          ))}
        </ul>
      )}
    </div>
  );
};
```

### `src/components/forms/F04_AssignDiagnosisEngineer.tsx`

```tsx
import React, { useEffect, useMemo, useState } from 'react';
import { X as XIcon } from 'lucide-react';
import { FormShell, type RendererProps } from './_shared';
import { FormComments } from '../ui/FormComments';
import { Select, TextArea } from '../ui';

interface F04Data {
  engineerId?: string;
  helpers?: string[];
  assignmentNotes?: string;
}

const MAX_HELPERS = 3;

export const F04Renderer: React.FC<RendererProps> = ({ rec, project, user, users, api }) => {
  const data = (rec.data || {}) as F04Data;
  const [engineerId, setEngineerId] = useState(data.engineerId || '');
  const [helpers, setHelpers] = useState<string[]>(data.helpers || []);
  const [notes, setNotes] = useState(data.assignmentNotes || '');

  const engineerOptions = useMemo(() => users
    .filter(u => u.status === 'active' && u.role === 'DIAGNOSIS_ENGINEER')
    .map(u => ({ value: u.id, label: u.fullName })),
  [users]);

  const helperOptions = useMemo(() => users
    .filter(u => u.status === 'active' && u.department === 'PROJECTS' && u.id !== engineerId && !helpers.includes(u.id))
    .map(u => ({ value: u.id, label: u.fullName })),
  [users, engineerId, helpers]);

  useEffect(() => {
    const initialEng    = data.engineerId || '';
    const initialHelp   = (data.helpers || []).join(',');
    const initialNotes  = data.assignmentNotes || '';
    if (engineerId === initialEng && helpers.join(',') === initialHelp && notes === initialNotes) return;
    const t = setTimeout(() => {
      api.updateFormData(rec.id, { engineerId, helpers, assignmentNotes: notes });
    }, 600);
    return () => clearTimeout(t);
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [engineerId, helpers, notes]);

  const addHelper    = (id: string) => { if (id && !helpers.includes(id) && helpers.length < MAX_HELPERS) setHelpers([...helpers, id]); };
  const removeHelper = (id: string) => setHelpers(helpers.filter(h => h !== id));

  return (
    <FormShell
      rec={rec} project={project} user={user} api={api}
      approvalPayload={{ engineerId, helpers, assignmentNotes: notes }}
      canApprove={() => engineerId ? null : 'يجب اختيار مهندس التشخيص قبل الاعتماد'}
    >
      <div className="space-y-4">
        <Select
          label="مهندس التشخيص (المُسنَد)"
          value={engineerId}
          onChange={(e) => setEngineerId(e.target.value)}
          options={engineerOptions}
          placeholder={engineerOptions.length ? 'اختر المهندس' : 'لا يوجد مهندسون نشطون'}
        />

        <div>
          <span className="text-sm font-medium text-fg-muted">فريق الفزعة (حتى {MAX_HELPERS} أعضاء)</span>
          {helpers.length > 0 && (
            <div className="mt-2 flex flex-wrap gap-2">
              {helpers.map(id => {
                const u = users.find(x => x.id === id);
                return (
                  <span key={id} className="inline-flex items-center gap-1.5 px-2.5 py-1 rounded-full bg-brand-teal/10 text-brand-teal text-xs">
                    {u?.fullName || id}
                    <button onClick={() => removeHelper(id)} aria-label="إزالة" className="hover:opacity-70">
                      <XIcon className="w-3 h-3" />
                    </button>
                  </span>
                );
              })}
            </div>
          )}
          {helpers.length < MAX_HELPERS && (
            <Select
              className="mt-2"
              value=""
              onChange={(e) => addHelper(e.target.value)}
              options={helperOptions}
              placeholder={helperOptions.length ? 'إضافة عضو...' : 'لا يوجد أعضاء متاحون'}
            />
          )}
        </div>

        <TextArea
          label="ملاحظات الإسناد"
          value={notes}
          onChange={(e) => setNotes(e.target.value)}
          rows={3}
          hint="نطاق الزيارة، المواعيد المقترحة، ملاحظات خاصة بالمستفيد."
        />
      </div>
      <FormComments formId={rec.id} user={user} />
    </FormShell>
  );
};
```
