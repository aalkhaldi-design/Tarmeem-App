### `src/components/ui/FileUploader.tsx` — full re-ship (UUID-keyed progress + drag-and-drop)

```tsx
import React, { useState } from 'react';
import { ref, uploadBytesResumable, getDownloadURL } from 'firebase/storage';
import { arrayUnion, doc, serverTimestamp, updateDoc } from 'firebase/firestore';
import { Upload, FileText, Loader2, Image as ImageIcon } from 'lucide-react';
import { db, storage } from '../../lib/firebase';
import type { UserProfile } from '../Auth';

export interface UploadedFile {
  name: string;
  url: string;
  size: number;
  mimeType: string;
  uploadedBy: string;
  uploadedAt: string;
  storagePath: string;
}

interface ProgressEntry { pct: number; name: string; }
interface Props {
  formId: string;
  projectRefId: string;
  user: UserProfile;
  files: UploadedFile[];
  accept?: string;
  maxSizeMb?: number;
  disabled?: boolean;
}

export const FileUploader: React.FC<Props> = ({
  formId, projectRefId, user, files, accept, maxSizeMb = 20, disabled,
}) => {
  const [progress, setProgress] = useState<Record<string, ProgressEntry>>({});
  const [error, setError] = useState<string | null>(null);
  const [dragHot, setDragHot] = useState(false);

  const newId = (): string =>
    (typeof crypto !== 'undefined' && 'randomUUID' in crypto)
      ? crypto.randomUUID()
      : `${Date.now()}-${Math.random().toString(36).slice(2)}`;

  const uploadOne = (file: File) => {
    if (file.size > maxSizeMb * 1024 * 1024) {
      setError(`${file.name}: حجم الملف يتجاوز ${maxSizeMb} م.ب`);
      return;
    }
    const uploadId = newId();
    const path = `projects/${projectRefId}/forms/${formId}/${uploadId}-${file.name}`;
    const storageRef = ref(storage, path);
    const task = uploadBytesResumable(storageRef, file);

    setProgress(p => ({ ...p, [uploadId]: { pct: 0, name: file.name } }));

    task.on(
      'state_changed',
      (snap) => {
        const pct = Math.round((snap.bytesTransferred / snap.totalBytes) * 100);
        setProgress(p => ({ ...p, [uploadId]: { pct, name: file.name } }));
      },
      (err) => {
        setError(`${file.name}: ${err.message}`);
        setProgress(p => { const { [uploadId]: _, ...rest } = p; return rest; });
      },
      async () => {
        const url = await getDownloadURL(task.snapshot.ref);
        const meta: UploadedFile = {
          name: file.name,
          url,
          size: file.size,
          mimeType: file.type || 'application/octet-stream',
          uploadedBy: user.id,
          uploadedAt: new Date().toISOString(),
          storagePath: path,
        };
        await updateDoc(doc(db, 'forms', formId), {
          files: arrayUnion(meta),
          updatedAt: serverTimestamp(),
        });
        setProgress(p => { const { [uploadId]: _, ...rest } = p; return rest; });
      },
    );
  };

  const onPick = (e: React.ChangeEvent<HTMLInputElement>) => {
    setError(null);
    Array.from(e.target.files || []).forEach(uploadOne);
    e.target.value = '';
  };

  const onDrop = (e: React.DragEvent<HTMLLabelElement>) => {
    e.preventDefault();
    e.stopPropagation();
    setDragHot(false);
    if (disabled) return;
    setError(null);
    Array.from(e.dataTransfer.files || []).forEach(uploadOne);
  };

  return (
    <div className="space-y-3">
      <label
        onDragOver={(e) => { e.preventDefault(); e.stopPropagation(); if (!disabled) setDragHot(true); }}
        onDragLeave={(e) => { e.preventDefault(); e.stopPropagation(); setDragHot(false); }}
        onDrop={onDrop}
        className={`flex items-center gap-3 p-3 rounded-lg border-2 border-dashed cursor-pointer transition-colors ${
          disabled ? 'opacity-50 cursor-not-allowed border-border-default' :
          dragHot ? 'border-brand-teal bg-brand-teal/5' :
                    'border-border-default hover:border-brand-teal'
        }`}
      >
        <input type="file" multiple accept={accept} onChange={onPick} disabled={disabled} className="hidden" />
        <Upload className="w-5 h-5 text-fg-muted shrink-0" />
        <div className="flex-1 min-w-0">
          <div className="text-sm font-medium text-fg">
            {dragHot ? 'أفلت الملفات هنا' : 'رفع ملفات'}
          </div>
          <div className="text-xs text-fg-faint">
            اسحب الملفات وأفلتها هنا أو اضغط للاختيار · حد أقصى {maxSizeMb} م.ب لكل ملف
          </div>
        </div>
      </label>

      {error && <p className="text-xs text-red-600 dark:text-red-400">{error}</p>}

      {Object.entries(progress).map(([id, { pct, name }]) => (
        <div key={id} className="flex items-center gap-2 text-xs text-fg-muted">
          <Loader2 className="w-3.5 h-3.5 animate-spin shrink-0" />
          <span className="truncate flex-1">{name}</span>
          <span className="shrink-0">{pct}%</span>
        </div>
      ))}

      {files.length > 0 && (
        <ul className="space-y-1.5">
          {files.map(f => (
            <li key={f.storagePath} className="flex items-center gap-2 text-xs">
              {f.mimeType.startsWith('image/')
                ? <ImageIcon className="w-3.5 h-3.5 text-brand-teal shrink-0" />
                : <FileText  className="w-3.5 h-3.5 text-fg-muted shrink-0" />}
              <a href={f.url} target="_blank" rel="noreferrer" className="flex-1 truncate text-brand-teal hover:underline">{f.name}</a>
              <span className="text-fg-faint shrink-0">{(f.size / 1024).toFixed(0)} ك.ب</span>
            </li>
          ))}
        </ul>
      )}
    </div>
  );
};
```

## New: `src/lib/migrations.ts`

```ts
import {
  collection, deleteField, doc, getDocs, query, serverTimestamp,
  Timestamp, updateDoc, where, writeBatch,
} from 'firebase/firestore';
import { addDoc } from 'firebase/firestore';
import { db } from './firebase';

interface LegacyMessage {
  authorId?: string;
  authorName?: string;
  body?: string;
  text?: string;
  message?: string;
  createdAt?: string | Timestamp | null;
}

/**
 * One-time migration: copy F-20 `data.messages[]` into the
 * `forms/{id}/comments` subcollection, then strip `data.messages`.
 *
 * Idempotent — forms without `data.messages` are skipped, so it can be
 * re-run safely (e.g. after partial network failure).
 *
 * Trigger this from the Admin page via a button, or run once via the
 * Firebase Console with the dev SDK. Do not leave both paths live.
 */
export async function migrateF20Messages(): Promise<{ migrated: number; skipped: number; commentsCopied: number }> {
  let migrated = 0;
  let skipped = 0;
  let commentsCopied = 0;

  const snap = await getDocs(query(collection(db, 'forms'), where('code', '==', 'F-20')));

  for (const formDoc of snap.docs) {
    const data = formDoc.data();
    const messages = (data?.data?.messages || []) as LegacyMessage[];
    if (!Array.isArray(messages) || messages.length === 0) { skipped++; continue; }

    for (const m of messages) {
      const body = (m.body || m.text || m.message || '').trim();
      if (!body) continue;

      const legacyIso = typeof m.createdAt === 'string'
        ? m.createdAt
        : ((m.createdAt as Timestamp)?.toDate?.() || new Date()).toISOString();

      await addDoc(collection(db, 'forms', formDoc.id, 'comments'), {
        authorId: m.authorId || 'unknown',
        authorName: m.authorName || 'مستخدم سابق',
        body,
        createdAt: serverTimestamp(),
        legacyCreatedAt: legacyIso,
        migratedAt: new Date().toISOString(),
      });
      commentsCopied++;
    }

    await updateDoc(doc(db, 'forms', formDoc.id), {
      'data.messages': deleteField(),
      updatedAt: serverTimestamp(),
    });
    migrated++;
  }

  return { migrated, skipped, commentsCopied };
}
```

## Caveats

1. **`F-21` is intentionally one-screen, no rooms split.** The original "furniture-counter rooms" phrasing in your batch note read as "the form *about* furniture counters" rather than "split counters per room" — confirm. If you do want per-room granularity, it's a structural change to `F21Data` (rooms[].furnitureFixed) and adds an OPEN/COLLAPSE per room. Easy enough but I held off per your "no tabs/wizards" guidance.

2. **`F-20` bucket UI is row-based, not Kanban.** You can drag the bucket dropdown to reassign per item. A real Kanban with drag-between-columns is ~80 more lines of code and `@dnd-kit`; row-based is enough for the count of items typical of a renovation project (10–30). If you want Kanban, say so for Batch 5+.

3. **`migrateF20Messages` runs from the client.** For thousands of forms this would be slow and would write notifications spammy if you'd added migration triggers. We don't have that volume yet — fine to call once from the Admin page. If you ever need to batch-migrate other things, we'd move it to a Cloud Function.

```tsx
import { useRef } from 'react';
import { Pencil, MapPin, Undo2, Trash2 } from 'lucide-react';

/** Editable croquis canvas. Two modes (pen, pin) + per-instance undo (cap 20).
 *  Pointer events with getCoalescedEvents() for smooth strokes on iOS Safari.
 *  Pin labels entered inline via <foreignObject> — no window.prompt focus theft. */
export interface CroquisEditorProps {
  croquis: CroquisRecord;
  onChange: (next: CroquisRecord) => void;
  disabled?: boolean;
}

type CroquisEditorMode = 'pen' | 'pin';
const PEN_COLOR = '#4A1F66';
const HISTORY_CAP = 20;
const VIEW_W = 400;
const VIEW_H = 300;

export const CroquisEditor: React.FC<CroquisEditorProps> = ({ croquis, onChange, disabled }) => {
  const svgRef = useRef<SVGSVGElement | null>(null);
  const [mode, setMode] = useState<CroquisEditorMode>('pen');
  const [drawingPath, setDrawingPath] = useState<CroquisPath | null>(null);
  const [pendingPin, setPendingPin] = useState<{ x: number; y: number; label: string } | null>(null);
  // Per-instance history. Switching to a different croquis remounts this component
  // (parent passes a new croquis with different id), so undo correctly scopes per-room.
  const [history, setHistory] = useState<CroquisRecord[]>([]);

  const toSvgPoint = (clientX: number, clientY: number): { x: number; y: number } => {
    const svg = svgRef.current;
    if (!svg) return { x: 0, y: 0 };
    const pt = svg.createSVGPoint();
    pt.x = clientX; pt.y = clientY;
    const ctm = svg.getScreenCTM();
    if (!ctm) return { x: 0, y: 0 };
    const t = pt.matrixTransform(ctm.inverse());
    return { x: Math.round(t.x), y: Math.round(t.y) };
  };

  const snapshot = () => setHistory(h => [...h.slice(-(HISTORY_CAP - 1)), croquis]);

  const onPointerDown = (e: React.PointerEvent) => {
    if (disabled || pendingPin) return;
    e.preventDefault();
    (e.target as Element).setPointerCapture(e.pointerId);
    const pt = toSvgPoint(e.clientX, e.clientY);
    if (mode === 'pen') {
      setDrawingPath({ id: `p${Date.now()}`, points: [pt], color: PEN_COLOR });
    } else {
      setPendingPin({ x: pt.x, y: pt.y, label: '' });
    }
  };

  const onPointerMove = (e: React.PointerEvent) => {
    if (disabled || mode !== 'pen' || !drawingPath) return;
    e.preventDefault();
    const native = e.nativeEvent as PointerEvent;
    const coalesced = typeof native.getCoalescedEvents === 'function'
      ? native.getCoalescedEvents()
      : [native];
    const newPoints = coalesced.map(ev => toSvgPoint(ev.clientX, ev.clientY));
    setDrawingPath(p => p ? { ...p, points: [...p.points, ...newPoints] } : null);
  };

  const onPointerUp = (e: React.PointerEvent) => {
    if (disabled) return;
    try { (e.target as Element).releasePointerCapture(e.pointerId); } catch { /* ignore */ }
    if (mode === 'pen' && drawingPath && drawingPath.points.length > 1) {
      snapshot();
      onChange({ ...croquis, paths: [...croquis.paths, drawingPath] });
    }
    setDrawingPath(null);
  };

  const commitPin = () => {
    if (!pendingPin) return;
    const label = pendingPin.label.trim();
    if (label) {
      snapshot();
      onChange({
        ...croquis,
        fixtures: [...croquis.fixtures, { id: `f${Date.now()}`, x: pendingPin.x, y: pendingPin.y, label }],
      });
    }
    setPendingPin(null);
  };

  const undo = () => {
    if (history.length === 0) return;
    const prev = history[history.length - 1];
    setHistory(h => h.slice(0, -1));
    onChange(prev);
  };

  const clearAll = () => {
    if (croquis.paths.length === 0 && croquis.fixtures.length === 0) return;
    if (!window.confirm('سيتم حذف كل المسارات والعلامات في هذا الكروكي. هل أنت متأكد؟')) return;
    snapshot();
    onChange({ ...croquis, paths: [], fixtures: [] });
  };

  const removeFixture = (id: string) => {
    if (disabled) return;
    snapshot();
    onChange({ ...croquis, fixtures: croquis.fixtures.filter(f => f.id !== id) });
  };

  return (
    <div className="space-y-2">
      <div className="flex flex-wrap items-center gap-2">
        <div className="inline-flex rounded-lg border border-border-default overflow-hidden">
          <button
            type="button"
            onClick={() => setMode('pen')}
            disabled={disabled}
            className={`px-3 py-1.5 text-xs flex items-center gap-1 ${
              mode === 'pen' ? 'bg-brand-teal text-white' : 'bg-input-bg text-fg-muted hover:bg-surface-down'
            }`}
          >
            <Pencil className="w-3 h-3" /> رسم
          </button>
          <button
            type="button"
            onClick={() => setMode('pin')}
            disabled={disabled}
            className={`px-3 py-1.5 text-xs flex items-center gap-1 border-r border-border-default ${
              mode === 'pin' ? 'bg-brand-teal text-white' : 'bg-input-bg text-fg-muted hover:bg-surface-down'
            }`}
          >
            <MapPin className="w-3 h-3" /> علامة
          </button>
        </div>
        <button
          type="button" onClick={undo} disabled={disabled || history.length === 0}
          className="px-3 py-1.5 text-xs rounded-lg border border-border-default bg-input-bg text-fg-muted hover:bg-surface-down disabled:opacity-40 flex items-center gap-1"
        >
          <Undo2 className="w-3 h-3" /> تراجع
        </button>
        <button
          type="button" onClick={clearAll} disabled={disabled}
          className="px-3 py-1.5 text-xs rounded-lg border border-red-300 dark:border-red-800 bg-input-bg text-red-600 dark:text-red-400 hover:bg-red-50 dark:hover:bg-red-900/20 disabled:opacity-40 flex items-center gap-1"
        >
          <Trash2 className="w-3 h-3" /> مسح الكل
        </button>
      </div>

      <svg
        ref={svgRef}
        viewBox={`0 0 ${VIEW_W} ${VIEW_H}`}
        onPointerDown={onPointerDown}
        onPointerMove={onPointerMove}
        onPointerUp={onPointerUp}
        onPointerCancel={onPointerUp}
        style={{ width: '100%', height: 'auto', touchAction: 'none' }}
        className={`bg-surface-up rounded-lg border ${disabled ? 'border-border-subtle opacity-60' : 'border-border-default'} ${
          mode === 'pen' && !disabled ? 'cursor-crosshair' : mode === 'pin' && !disabled ? 'cursor-copy' : 'cursor-default'
        }`}
      >
        {/* Light grid for orientation */}
        <defs>
          <pattern id="croquis-grid" width="20" height="20" patternUnits="userSpaceOnUse">
            <path d="M 20 0 L 0 0 0 20" fill="none" stroke="currentColor" strokeWidth="0.3" className="text-border-subtle" />
          </pattern>
        </defs>
        <rect width={VIEW_W} height={VIEW_H} fill="url(#croquis-grid)" />

        {croquis.paths.map(p => (
          <polyline
            key={p.id}
            fill="none"
            stroke={p.color || PEN_COLOR}
            strokeWidth="2"
            strokeLinecap="round"
            strokeLinejoin="round"
            points={p.points.map(pt => `${pt.x},${pt.y}`).join(' ')}
          />
        ))}

        {drawingPath && drawingPath.points.length > 1 && (
          <polyline
            fill="none"
            stroke={drawingPath.color}
            strokeWidth="2"
            strokeLinecap="round"
            strokeLinejoin="round"
            points={drawingPath.points.map(pt => `${pt.x},${pt.y}`).join(' ')}
          />
        )}

        {croquis.fixtures.map(f => (
          <g key={f.id} className="cursor-pointer" onClick={() => removeFixture(f.id)}>
            <circle cx={f.x} cy={f.y} r="5" fill="#43bba1" stroke="#fff" strokeWidth="1.5" />
            <text x={f.x + 8} y={f.y + 4} fontSize="11" fill="currentColor" className="text-fg select-none pointer-events-none">
              {f.label}
            </text>
          </g>
        ))}

        {pendingPin && (
          <foreignObject x={pendingPin.x + 8} y={pendingPin.y - 14} width="140" height="34">
            <div className="bg-surface border border-brand-teal rounded shadow-card-hover flex items-center" style={{ direction: 'rtl' }}>
              <input
                autoFocus
                value={pendingPin.label}
                onChange={(e) => setPendingPin(p => p ? { ...p, label: e.target.value } : null)}
                onKeyDown={(e) => {
                  if (e.key === 'Enter') { e.preventDefault(); commitPin(); }
                  else if (e.key === 'Escape') { setPendingPin(null); }
                }}
                onBlur={commitPin}
                placeholder="اسم العنصر"
                className="flex-1 min-w-0 px-2 py-1 text-xs bg-transparent text-fg focus:outline-none"
              />
            </div>
          </foreignObject>
        )}
      </svg>

      <p className="text-[11px] text-fg-faint">
        {mode === 'pen' ? 'وضع الرسم — اسحب لرسم خطوط' : 'وضع العلامات — انقر لإضافة علامة، انقر علامة موجودة لحذفها'}
        {' · '}مسارات: {croquis.paths.length} · علامات: {croquis.fixtures.length}
        {history.length > 0 && ` · يمكن التراجع ${history.length} خطوة`}
      </p>
    </div>
  );
};
```

```tsx
interface CivilFixed {
  ceiling?:    { needed: boolean; note?: string };
  concrete?:   { needed: boolean; note?: string };
  ceramic?:    { needed: boolean; note?: string };
  insulation?: { needed: boolean; note?: string };
  plaster?:    { needed: boolean; note?: string };
  paint?:      { needed: boolean; note?: string };
  aluminum?:   { needed: boolean; note?: string };
  wood?:       { needed: boolean; note?: string };
}
interface ElectricalFixed {
  ceilingLight?: number; breaker?: number; normalSocket?: number; spotlight?: number;
}
interface PlumbingFixed {
  westernToilet?: number; arabicToilet?: number; ceramicSink?: number; heater?: number;
}
interface NoteItem  { id: string; name: string; note?: string; }
interface CustomItem { id: string; name: string; qty: number; }

interface WorkItem {
  id: string;
  name: string;
  civilFixed?: CivilFixed;
  civilCustom?: NoteItem[];
  electricalFixed?: ElectricalFixed;
  electricalCustom?: NoteItem[];
  plumbingFixed?: PlumbingFixed;
  plumbingCustom?: NoteItem[];
}

interface FurnitureFixed {
  bedSingle?: number; bed15?: number; bedDouble?: number;
  mattressSingle?: number; mattress15?: number; mattressDouble?: number;
  sofaMeters?: number; floorSeating?: number; carpet?: number;
  wardrobe2?: number; wardrobe3?: number; wardrobe4?: number;
  nightstand?: number; vanity?: number;
}
interface AppliancesFixed {
  acSplit1?: number; acSplit15?: number; acWindow15?: number;
  washer?: number; stove?: number; fridge?: number; vacuum?: number; waterCooler?: number;
}

interface F08Data {
  // Top-level (per RENDERER_CONTRACT — cascades read data.safetyHazard, NOT nested)
  safetyHazard?: boolean;
  // Basic
  buildingType?: string;
  area?: number;
  age?: number;
  diagnosisResult?: string;
  diagnosisSummary?: string;
  // Works
  works?: WorkItem[];
  // Furniture
  furnitureFixed?: FurnitureFixed;
  appliancesFixed?: AppliancesFixed;
  furnitureCustom?: CustomItem[];
  // Croquis
  croquises?: CroquisRecord[];
  // Approval
  engineerPledge?: boolean;
  approvalNotes?: string;
}

// Main renderer
// ============================================================================
export const F08Renderer: React.FC<RendererProps> = ({ rec, project, user, api }) => {
  const data = (rec.data || {}) as F08Data;

  // Basic
  const [safetyHazard,     setSafetyHazard]     = useState(!!data.safetyHazard);
  const [buildingType,     setBuildingType]     = useState(data.buildingType || '');
  const [area,             setArea]             = useState(data.area?.toString() || '');
  const [age,              setAge]              = useState(data.age?.toString() || '');
  const [diagnosisResult,  setDiagnosisResult]  = useState(data.diagnosisResult || '');
  const [diagnosisSummary, setDiagnosisSummary] = useState(data.diagnosisSummary || '');
  // Works
  const [works,            setWorks]            = useState<WorkItem[]>(data.works || []);
  // Furniture
  const [furnitureFixed,   setFurnitureFixed]   = useState<FurnitureFixed>(data.furnitureFixed || {});
  const [appliancesFixed,  setAppliancesFixed]  = useState<AppliancesFixed>(data.appliancesFixed || {});
  const [furnitureCustom,  setFurnitureCustom]  = useState<CustomItem[]>(data.furnitureCustom || []);
  // Croquis
  const [croquises,        setCroquises]        = useState<CroquisRecord[]>(data.croquises || []);
  // Approval
  const [engineerPledge,   setEngineerPledge]   = useState(!!data.engineerPledge);
  const [approvalNotes,    setApprovalNotes]    = useState(data.approvalNotes || '');

  const [activeTab, setActiveTab] = useState<TabKey>('basic');

  // Autosave (debounced 800ms)
  useEffect(() => {
    const t = setTimeout(() => {
      api.updateFormData(rec.id, {
        safetyHazard,
        buildingType,
        area: Number(area) || 0,
        age: Number(age) || 0,
        diagnosisResult,
        diagnosisSummary,
        works,
        furnitureFixed,
        appliancesFixed,
        furnitureCustom,
        croquises,
        engineerPledge,
        approvalNotes,
      });
    }, 800);
    return () => clearTimeout(t);
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [
    safetyHazard, buildingType, area, age, diagnosisResult, diagnosisSummary,
    works, furnitureFixed, appliancesFixed, furnitureCustom,
    croquises, engineerPledge, approvalNotes,
  ]);

  const isEngineerStage = rec.approvalIndex === 0;

  const summary = useMemo(() => ({
    worksCount:      works.length,
    croquisCount:    croquises.length,
    furnitureCount:  Object.values(furnitureFixed).reduce((s, v) => s + (Number(v) || 0), 0)
                   + furnitureCustom.reduce((s, x) => s + (Number(x.qty) || 0), 0),
    appliancesCount: Object.values(appliancesFixed).reduce((s, v) => s + (Number(v) || 0), 0),
  }), [works, croquises, furnitureFixed, furnitureCustom, appliancesFixed]);

  const approvalPayload = {
    safetyHazard, buildingType,
    area: Number(area) || 0, age: Number(age) || 0,
    diagnosisResult, diagnosisSummary,
    works, furnitureFixed, appliancesFixed, furnitureCustom,
    croquises, engineerPledge, approvalNotes,
  };

  const canApprove = (): string | null => {
    if (!buildingType)    return 'يجب تحديد نوع المبنى في تبويب «الأساسية»';
    if (!diagnosisResult) return 'يجب تحديد نتيجة التشخيص في تبويب «الأساسية»';
    if (isEngineerStage && !engineerPledge) return 'يجب التأكيد على التشخيص في تبويب «الاعتماد»';
    return null;
  };

  return (
    <FormShell
      rec={rec} project={project} user={user} api={api}
      approvalPayload={approvalPayload}
      canApprove={canApprove}
    >
      <div className="space-y-4">
        <TabBar activeTab={activeTab} onChange={setActiveTab} safetyHazard={safetyHazard} />

        {activeTab === 'basic' && (
          <BasicTab
            safetyHazard={safetyHazard}     setSafetyHazard={setSafetyHazard}
            buildingType={buildingType}     setBuildingType={setBuildingType}
            area={area}                     setArea={setArea}
            age={age}                       setAge={setAge}
            diagnosisResult={diagnosisResult} setDiagnosisResult={setDiagnosisResult}
            diagnosisSummary={diagnosisSummary} setDiagnosisSummary={setDiagnosisSummary}
          />
        )}
        {activeTab === 'works'     && <WorksTab works={works} setWorks={setWorks} formId={rec.id} projectRefId={rec.projectRefId || ''} user={user} files={rec.files || []} />}
        {activeTab === 'furniture' && <FurnitureTab furniture={furnitureFixed} setFurniture={setFurnitureFixed} appliances={appliancesFixed} setAppliances={setAppliancesFixed} custom={furnitureCustom} setCustom={setFurnitureCustom} />}
        {activeTab === 'croquis'   && <CroquisTab croquises={croquises} setCroquises={setCroquises} />}
        {activeTab === 'approval'  && (
          <ApprovalTab
            summary={summary}
            isEngineerStage={isEngineerStage}
            pledge={engineerPledge}         setPledge={setEngineerPledge}
            approvalNotes={approvalNotes}   setApprovalNotes={setApprovalNotes}
          />
        )}
      </div>
      <FormComments formId={rec.id} user={user} />
    </FormShell>
  );
};

// ============================================================================
// TabBar — scroll-snap on mobile, full row on lg+
// ============================================================================
const TabBar: React.FC<{
  activeTab: TabKey;
  onChange: (t: TabKey) => void;
  safetyHazard: boolean;
}> = ({ activeTab, onChange, safetyHazard }) => (
  <div className="flex gap-1 overflow-x-auto snap-x snap-mandatory pb-1 sticky top-0 z-10 bg-surface-up -mx-5 px-5 lg:mx-0 lg:px-0 lg:overflow-visible">
    {TAB_DEFS.map(t => {
      const active = activeTab === t.key;
      return (
        <button
          key={t.key}
          onClick={() => onChange(t.key)}
          className={`shrink-0 snap-start px-3 py-2 rounded-lg text-xs font-medium transition-colors flex items-center gap-1 ${
            active
              ? 'bg-brand text-white shadow-card'
              : 'bg-surface-down text-fg-muted hover:bg-border-subtle'
          }`}
        >
          {t.label}
          {t.key === 'basic' && safetyHazard && (
            <AlertTriangle className={`w-3 h-3 ${active ? 'text-amber-200' : 'text-red-500'}`} />
          )}
        </button>
      );
    })}
  </div>
);

// ============================================================================
// Tab: Works
// ============================================================================
const WorksTab: React.FC<{
  works: WorkItem[]; setWorks: React.Dispatch<React.SetStateAction<WorkItem[]>>;
  formId: string; projectRefId: string; user: RendererProps['user']; files: WorkFiles;
}> = ({ works, setWorks }) => {
  const [expandedId, setExpandedId] = useState<string | null>(null);

  const addWork = () => {
    const id = `w${Date.now()}`;
    setWorks(ws => [...ws, {
      id, name: `العنصر ${ws.length + 1}`,
      civilFixed: {}, civilCustom: [],
      electricalFixed: {}, electricalCustom: [],
      plumbingFixed: {}, plumbingCustom: [],
    }]);
    setExpandedId(id);
  };
  const updateWork = (id: string, patch: Partial<WorkItem>) =>
    setWorks(ws => ws.map(w => w.id === id ? { ...w, ...patch } : w));
  const removeWork = (id: string) =>
    setWorks(ws => ws.filter(w => w.id !== id));

  return (
    <div className="space-y-3">
      <div className="flex items-center justify-between">
        <h3 className="font-semibold text-fg">حصر الأعمال ({works.length})</h3>
        <button onClick={addWork} className="text-xs text-brand-teal hover:underline flex items-center gap-1">
          <Plus className="w-3 h-3" /> إضافة غرفة / منطقة
        </button>
      </div>

      {works.length === 0 ? (
        <p className="text-xs text-fg-faint p-4 text-center rounded-lg bg-surface-down">
          لم يتم إضافة عناصر بعد. ابدأ بإضافة غرفة أو منطقة (صالة، مطبخ، حمام...).
        </p>
      ) : (
        <div className="space-y-2">
          {works.map(work => (
            <WorkCard
              key={work.id}
              work={work}
              expanded={expandedId === work.id}
              onToggle={() => setExpandedId(e => e === work.id ? null : work.id)}
              onChange={patch => updateWork(work.id, patch)}
              onRemove={() => removeWork(work.id)}
            />
          ))}
        </div>
      )}
    </div>
  );
};

// (helper alias to avoid pulling in UploadedFile type at call site)
type WorkFiles = unknown[];

const WorkCard: React.FC<{
  work: WorkItem;
  expanded: boolean;
  onToggle: () => void;
  onChange: (patch: Partial<WorkItem>) => void;
  onRemove: () => void;
}> = ({ work, expanded, onToggle, onChange, onRemove }) => {
  const civil = work.civilFixed || {};
  const electrical = work.electricalFixed || {};
  const plumbing = work.plumbingFixed || {};

  const civilCount      = Object.values(civil).filter(v => v?.needed).length + (work.civilCustom?.length || 0);
  const electricalCount = Object.values(electrical).reduce((s, v) => s + (Number(v) || 0), 0) + (work.electricalCustom?.length || 0);
  const plumbingCount   = Object.values(plumbing).reduce((s, v) => s + (Number(v) || 0), 0) + (work.plumbingCustom?.length || 0);

  const setCivilField = (key: keyof CivilFixed, patch: { needed?: boolean; note?: string }) =>
    onChange({ civilFixed: { ...civil, [key]: { ...(civil[key] || { needed: false }), ...patch } } });
  const setElec = (key: keyof ElectricalFixed, v: number) =>
    onChange({ electricalFixed: { ...electrical, [key]: v } });
  const setPlumb = (key: keyof PlumbingFixed, v: number) =>
    onChange({ plumbingFixed: { ...plumbing, [key]: v } });

  return (
    <div className="rounded-lg border border-border-subtle bg-surface-up overflow-hidden">
      {/* Header */}
      <div className="flex items-center gap-2 p-2">
        <button onClick={onToggle} className="p-1 text-fg-muted hover:bg-surface-down rounded" aria-label="فتح / إغلاق">
          {expanded ? <ChevronUp className="w-4 h-4" /> : <ChevronDown className="w-4 h-4" />}
        </button>
        <input
          type="text"
          value={work.name}
          onChange={(e) => onChange({ name: e.target.value })}
          placeholder="اسم الغرفة / المنطقة"
          className="flex-1 min-w-0 px-2 py-1.5 rounded border border-border-subtle bg-input-bg text-fg text-sm focus:outline-none focus:ring-1 focus:ring-brand-teal"
        />
        <div className="text-[10px] text-fg-faint flex items-center gap-2 shrink-0">
          <span>مدني: {civilCount}</span>
          <span>كهرباء: {electricalCount}</span>
          <span>سباكة: {plumbingCount}</span>
        </div>
        <button onClick={onRemove} aria-label="إزالة" className="p-1 text-fg-faint hover:text-red-600 hover:bg-red-50 dark:hover:bg-red-900/20 rounded">
          <X className="w-4 h-4" />
        </button>
      </div>

      {/* Body */}
      {expanded && (
        <div className="border-t border-border-subtle p-3 space-y-4 bg-app-bg">

          {/* Civil */}
          <Section title="الأعمال المدنية">
            <div className="grid grid-cols-1 sm:grid-cols-2 gap-2">
              {CIVIL_LABELS.map(({ key, label }) => {
                const item = civil[key] || { needed: false };
                return (
                  <div key={key} className="flex items-start gap-2">
                    <label className="flex items-center gap-1.5 cursor-pointer min-w-[100px]">
                      <input
                        type="checkbox"
                        checked={item.needed}
                        onChange={(e) => setCivilField(key, { needed: e.target.checked })}
                        className="w-4 h-4 accent-brand-teal"
                      />
                      <span className="text-sm text-fg">{label}</span>
                    </label>
                    {item.needed && (
                      <input
                        type="text"
                        value={item.note || ''}
                        onChange={(e) => setCivilField(key, { note: e.target.value })}
                        placeholder="ملاحظة"
                        className="flex-1 min-w-0 px-2 py-1 rounded border border-border-subtle bg-input-bg text-fg text-xs focus:outline-none focus:ring-1 focus:ring-brand-teal"
                      />
                    )}
                  </div>
                );
              })}
            </div>
            <CustomList
              items={work.civilCustom || []}
              onChange={(next) => onChange({ civilCustom: next })}
              placeholder="عنصر مدني مخصص"
            />
          </Section>

          {/* Electrical */}
          <Section title="الأعمال الكهربائية">
            <div className="grid grid-cols-2 gap-2">
              {ELECTRICAL_LABELS.map(({ key, label }) => (
                <NumberCounter
                  key={key} label={label}
                  value={electrical[key] || 0}
                  onChange={(v) => setElec(key, v)}
                />
              ))}
            </div>
            <CustomList
              items={work.electricalCustom || []}
              onChange={(next) => onChange({ electricalCustom: next })}
              placeholder="عنصر كهربائي مخصص"
            />
          </Section>

          {/* Plumbing */}
          <Section title="أعمال السباكة">
            <div className="grid grid-cols-2 gap-2">
              {PLUMBING_LABELS.map(({ key, label }) => (
                <NumberCounter
                  key={key} label={label}
                  value={plumbing[key] || 0}
                  onChange={(v) => setPlumb(key, v)}
                />
              ))}
            </div>
            <CustomList
              items={work.plumbingCustom || []}
              onChange={(next) => onChange({ plumbingCustom: next })}
              placeholder="عنصر سباكة مخصص"
            />
          </Section>
        </div>
      )}
    </div>
  );
};

const Section: React.FC<{ title: string; children: React.ReactNode }> = ({ title, children }) => (
  <div>
    <h4 className="text-xs font-semibold text-fg-muted mb-2">{title}</h4>
    <div className="space-y-2">{children}</div>
  </div>
);

const CustomList: React.FC<{
  items: NoteItem[];
  onChange: (next: NoteItem[]) => void;
  placeholder: string;
}> = ({ items, onChange, placeholder }) => {
  const add = () => onChange([...items, { id: `n${Date.now()}`, name: '', note: '' }]);
  const update = (id: string, patch: Partial<NoteItem>) =>
    onChange(items.map(x => x.id === id ? { ...x, ...patch } : x));
  const remove = (id: string) => onChange(items.filter(x => x.id !== id));

  return (
    <div className="space-y-1.5">
      {items.map(it => (
        <div key={it.id} className="flex items-center gap-1.5">
          <input
            type="text" value={it.name}
```

## Notes worth flagging

1. **F-85 manual-entry doesn't auto-clear winner fields when toggling from a bid.** If a user picks bid A then clicks "manual entry," the contractor/price stay populated from bid A — they can edit freely. Intentional: avoids losing typed data on a mis-click. If you'd rather have manual mode start blank, three-line change in `pickManual()`.

2. **F-19's `handoverProgress=100` is the only Approve gate.** That's the spec ("never closes until project closes" essentially means handoverProgress reaches 100). PROCUREMENT_OFFICER can sit on the form for weeks updating progressively; standard SLA still ticks unless the project is held. If the SLA needs to be silenced for permanent trackers like F-19, that's a `slaDays: null` change on the FormDef in `data.ts` — flag it and I'll patch in Batch 7.

3. **F-34's `f20_items` are read-only display.** The engineer reviews but can't edit the F-20 items at transfer time. If a correction is needed, F-20 must be re-opened (rejection workflow). This matches spec — the transfer form is a checkpoint, not a re-edit screen.

4. **All six renderers use the standard action bar** (no `customActions`). The multi-stage forms in Batch 8 will need the `<MultiStageActionBar>` helper you asked for — I'll add it to `_shared.tsx` at the top of Batch 8 so F-15, F-35, and F-23 all share the same stage UI.
