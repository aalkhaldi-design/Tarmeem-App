```ts
// REPLACE the rejectForm method on formsApi. Multi-stage forms (chain length > 1)
// now reset to approvalIndex=0 and stay 'pending' so the first-stage role can re-edit.
// Single-stage forms keep the existing bounce-to-rejected semantic.
rejectForm: async (formId, user, note) => {
  const rec = stateRef.current.forms.find((f) => f.id === formId);
  if (!rec) throw new Error('Form not found');
  if (!formAwaitsUser(rec, user)) throw new Error('Not authorized to reject');

  const isMultiStage = rec.approvalChain.length > 1;
  const approval = {
    role: rec.approvalChain[rec.approvalIndex],
    actorId: user.id, actorName: user.fullName,
    at: new Date().toISOString(), decision: 'rejected' as const, note: note || '',
  };

  if (isMultiStage) {
    // Reset to stage 0; preserve form.data (audit-correct re-edit context).
    await updateDoc(doc(db, 'forms', formId), {
      status: 'pending',
      approvalIndex: 0,
      stepStartedAt: serverTimestamp(),
      approvals: arrayUnion(approval),
      updatedAt: serverTimestamp(),
    });

    const firstStageRole = rec.approvalChain[0];
    const def = ROLE_BY_KEY[firstStageRole as RoleKey];
    const recipients = stateRef.current.users
      .filter(u => u.status === 'active')
      .filter(u => u.role === firstStageRole || (u.isAdmin && def?.department && u.department === def.department))
      .map(u => u.id);

    if (recipients.length) {
      await addDoc(collection(db, 'notifications'), {
        recipients, readBy: [],
        text: `نموذج ${rec.code} عاد لمرحلة البداية — مراجعة من ${def?.name || firstStageRole}: ${note || 'بدون سبب محدد'}`,
        subject: 'مراجعة متعددة المراحل',
        type: 'multi_stage_rejected',
        link: rec.id,
        createdAt: new Date().toISOString(),
      });
    }
  } else {
    // Single-stage: existing behavior.
    await updateDoc(doc(db, 'forms', formId), {
      status: 'rejected',
      approvals: arrayUnion(approval),
      updatedAt: serverTimestamp(),
    });
    if (rec.createdBy && rec.createdBy !== 'system') {
      await addDoc(collection(db, 'notifications'), {
        recipients: [rec.createdBy], readBy: [],
        text: `نموذج ${rec.code} مرفوض للتعديل — السبب: ${note || 'بدون سبب محدد'}`,
        subject: 'مطلوب تعديل',
        type: 'form_rejected',
        link: rec.id,
```

```tsx
// Add this import alongside the existing lucide imports:
import { Check } from 'lucide-react';

// Append at the bottom of _shared.tsx (after CroquisEditor + GenericFallback):

export interface StageDef {
  index: number;
  label: string;
  role: RoleKey;
  /** JSX body for this stage. Parent owns state; closure-captured setters work.
   *  MultiStageActionBar wraps this in a <fieldset disabled> so inputs lock
   *  automatically when this stage isn't the active one. */
  body: React.ReactNode;
  /** Optional Approve gate for this stage. Return null to allow, string to block. */
  canApprove?: () => string | null;
}

export interface MultiStageActionBarProps {
  rec: FormRecord;
  user: UserProfile;
  api: FormsApiLike;
  stages: StageDef[];
}

type MSMode = null | 'approve' | 'reject' | 'decline' | 'defer';

export const MultiStageActionBar: React.FC<MultiStageActionBarProps> = ({ rec, user, api, stages }) => {
  const [mode, setMode] = useState<MSMode>(null);
  const [note, setNote] = useState('');
  const [busy, setBusy] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const awaitsMe    = formAwaitsUser(rec, user);
  const canDecline  = DECLINE_ELIGIBLE_FORMS.includes(rec.code);
  const currentIdx  = rec.approvalIndex;
  const currentStage = stages.find(s => s.index === currentIdx);
  const isPending   = rec.status === 'pending';
  const isApproved  = rec.status === 'approved' || rec.status === 'completed';

  const submit = async () => {
    setError(null);
    if (mode === 'approve' && currentStage?.canApprove) {
      const err = currentStage.canApprove();
      if (err) { setError(err); return; }
    }
    if ((mode === 'decline' || mode === 'reject') && !note.trim()) {
      setError(mode === 'decline' ? 'السبب مطلوب للرفض النهائي' : 'السبب مطلوب');
      return;
    }
    setBusy(true);
    try {
      if      (mode === 'approve') await api.approveForm(rec.id, user, note.trim());
      else if (mode === 'reject')  await api.rejectForm (rec.id, user, note.trim());
      else if (mode === 'decline') await api.declineForm(rec.id, user, note.trim());
      else if (mode === 'defer')   await api.deferForm  (rec.id, user, note.trim());
      setMode(null);
      setNote('');
    } catch (err) {
      setError((err as Error).message || 'تعذّر تنفيذ العملية');
    } finally {
      setBusy(false);
    }
  };

  return (
    <div className="space-y-4">
      {/* Stage chip strip */}
      <div className="flex flex-wrap items-center gap-2">
        {stages.map((stage, i) => {
          const past    = currentIdx > stage.index || isApproved;
          const active  = currentIdx === stage.index && isPending;
          const cls = past
            ? 'bg-green-100 text-green-800 dark:bg-green-900/30 dark:text-green-300'
            : active
              ? 'bg-brand text-white shadow-card'
              : 'bg-surface-down text-fg-faint';
          return (
            <React.Fragment key={stage.index}>
              <div className={`flex items-center gap-1.5 px-3 py-1.5 rounded-full text-xs ${cls}`}>
                {past && <Check className="w-3 h-3" />}
                <span className="font-medium">{stage.label}</span>
                <span className="text-[10px] opacity-70">· {ROLE_BY_KEY[stage.role]?.name || stage.role}</span>
              </div>
              {i < stages.length - 1 && <span className="text-fg-faint text-xs">←</span>}
            </React.Fragment>
          );
        })}
      </div>

      {/* Stage bodies — all stacked, only the active one is editable */}
      <div className="space-y-3">
        {stages.map(stage => {
          const past   = currentIdx > stage.index || isApproved;
          const active = currentIdx === stage.index && isPending;
          const future = currentIdx < stage.index && isPending;
          return (
            <div
              key={stage.index}
              className={`rounded-lg border p-3 ${
                active ? 'border-brand-teal bg-brand-teal/5'
                       : past ? 'border-border-subtle bg-surface-up'
                              : 'border-border-subtle bg-surface-up opacity-70'
              }`}
            >
              <div className="flex items-center gap-2 text-sm mb-2">
                {past && <Check className="w-4 h-4 text-green-600 dark:text-green-400" />}
                {future && <Lock className="w-4 h-4 text-fg-faint" />}
                <span className="font-medium text-fg">{stage.label}</span>
                <span className="text-xs text-fg-faint">· {ROLE_BY_KEY[stage.role]?.name || stage.role}</span>
              </div>
              <fieldset
                disabled={!active || !awaitsMe}
                className="border-0 p-0 m-0 min-w-0"
              >
                {stage.body}
              </fieldset>
            </div>
          );
        })}
      </div>

      {/* Action buttons (only render when this user actually owns the active stage) */}
      {isPending && awaitsMe && !mode && (
        <div className="pt-2 flex flex-wrap gap-2">
          <button onClick={() => setMode('approve')} className="px-4 py-2 rounded-lg bg-emerald-600 text-white text-sm font-medium hover:bg-emerald-700 flex items-center gap-1.5">

      {isPending && awaitsMe && mode && (

              عند الإرجاع، يعود النموذج لمرحلة البداية «{stages[0].label}» لإعادة المعالجة. البيانات المعبأة تُحفظ ولا تُمسح.
            </div>

            }
```

## Notes worth flagging

1. **Multi-stage rejection is a behavioral change to live forms.** F-02, F-08, F-14, F-21 already exist with the old "rejection = status:'rejected'" semantic. After the App.tsx patch, any future rejection on those forms goes to "back to stage 0, status:'pending'" instead. Any forms currently sitting in `status:'rejected'` from prior rejections are unaffected — they stay rejected until an admin manually clears them. Worth noting in your release comms.

2. **`<MultiStageActionBar>` `body` is JSX, not an FC.** Closure-captured state from the parent renderer makes autosave and `canApprove()` straightforward — no prop drilling, no re-mount issues. The fieldset wrapping inside the bar gates editability per stage automatically. If you ever need stage-isolated state (rare), wrap that stage's body in its own small component declared at file top.

3. **Decline button only shows on F-23.** The `DECLINE_ELIGIBLE_FORMS` registry already gates it correctly — F-15 and F-35 inherit the right behavior automatically (no Decline button). F-23's decline cascade (notify originator + supervising engineer, no project state change) was wired in Batch 5.

4. **F-15 `paymentIndex` defaults to 2 if missing.** F-14 trigger always sets it on auto-create, but defensive default keeps the renderer from crashing if a record is corrupt. Same defensive pattern as other renderers with required props.

5. **`PaymentSummary` is duplicated between F-15 and F-35.** Intentional — they're the only two consumers, the component is 12 lines, and adding it to `_shared.tsx` would inflate the shared module for negligible gain. If a third payment form ever lands, extract then.

🔴 Must fix in Batch 9 opening: FileUploader bypasses <fieldset disabled> inside stage bodies
MultiStageActionBar wraps each stage's body in <fieldset disabled={!active || !awaitsMe}>. This correctly locks <input>, <select>, <textarea>, <button> elements via the HTML spec. But FileUploader's drag-and-drop handler is on a <label> — <fieldset disabled> does not propagate to label-level event handlers per the HTML spec. A user viewing F-35 stage 2 as a past stage (or as a non-awaitee) can still drag a receipt into the uploader.
Three options:
(a) Add a stageActive context. Create React.createContext<{ active: boolean; awaitsMe: boolean }> in _shared.tsx, provide it from MultiStageActionBar per-stage, and have FileUploader consume it.
(b) Pass disabled explicitly at every FileUploader call site inside stage bodies. Renderers need to know which stage they're in. The closure-captured pattern doesn't trivially give them that.
(c) Have FileUploader check its parent <fieldset> via a ref. Hacky.
Pick (a). It generalizes — any "drag-drop / pointer / portal" component inside future stages gets stage-gating for free. Patch:
// _shared.tsx — add near the top, alongside other exports:
export const StageGateContext = React.createContext<{ active: boolean; awaitsMe: boolean }>({
  active: true,
  awaitsMe: true,
});

// Inside MultiStageActionBar, wrap each stage body:
<StageGateContext.Provider value={{ active, awaitsMe }}>
  <fieldset disabled={!active || !awaitsMe} className="border-0 p-0 m-0 min-w-0">
    {stage.body}
  </fieldset>
</StageGateContext.Provider>
Then FileUploader reads the context and ORs into its disabled prop:
// FileUploader.tsx
import { StageGateContext } from '../forms/_shared';
// ...
const stageGate = useContext(StageGateContext);
const effectivelyDisabled = disabled || (!stageGate.active || !stageGate.awaitsMe);

## `src/lib/createProject.ts`

```ts
import {
  addDoc, collection, doc, getDocs, query, runTransaction,
  serverTimestamp, where,
} from 'firebase/firestore';
import { db } from './firebase';
import { FORMS } from './data';
import type { FormCode, FormStatus, RoleKey } from './data';
import type { UserProfile } from '../components/Auth';
import type { F02Data } from '../components/forms/F02_SocialResearch';

export interface CreateProjectResult {
  projectRefId: string;
  projectId: string;
  formIds: Partial<Record<FormCode, string>>;
}

/** Atomic project creation: counter increment + project doc + 22 form drafts in one transaction.
 *  Firestore transaction limit (500 writes) is well above the 24 writes this performs. */
export async function createProject(formData: F02Data, user: UserProfile): Promise<CreateProjectResult> {
  if (user.role !== 'SOCIAL_RESEARCHER' && !user.isAdmin) {
    throw new Error('فقط الباحث الاجتماعي أو مسؤول النظام يستطيع إنشاء حالة جديدة');
  }
  const beneficiaryName = formData.fullName?.trim();
  if (!beneficiaryName) throw new Error('الاسم الكامل للمستفيد مطلوب');

  const result = await runTransaction(db, async (tx) => {
    // 1. Increment the project serial atomically.
    const counterRef = doc(db, 'counters', 'projectSerial');
    const counterSnap = await tx.get(counterRef);
    const currentSerial = counterSnap.exists() ? ((counterSnap.data() as { value?: number }).value || 0) : 0;
    const nextSerial = currentSerial + 1;
    tx.set(counterRef, { value: nextSerial, updatedAt: serverTimestamp() }, { merge: true });

    const year = new Date().getFullYear();
    const projectId = `TRM-${year}-${String(nextSerial).padStart(3, '0')}`;

    // 2. Project document.
    const projectRef = doc(collection(db, 'projects'));
    tx.set(projectRef, {
      projectId,
      beneficiaryName,
      phase: 'RESEARCH',
      status: 'active',
      archived: false,
      progressPct: 5,
      city: formData.city || null,
      createdBy: user.id,
      createdByName: user.fullName,
      createdAt: serverTimestamp(),
      updatedAt: serverTimestamp(),
    });

    // 3. All 22 form drafts. F-02 is special: status='pending', approvalIndex=1,
    //    with stage 0 (researcher) auto-completed in approvals[] since the wizard
    //    submission IS the researcher's action.
    const formIds: Partial<Record<FormCode, string>> = {};
    for (const def of FORMS) {
      const formRef = doc(collection(db, 'forms'));
      formIds[def.code] = formRef.id;

      const isF02 = def.code === 'F-02';
      const status: FormStatus = isF02 ? 'pending' : 'draft';
      const approvalIndex = isF02 ? 1 : 0;
      const approvals = isF02 ? [{
        role: 'SOCIAL_RESEARCHER' as RoleKey,
        actorId: user.id,
        actorName: user.fullName,
        at: new Date().toISOString(),
        decision: 'approved' as const,
        note: 'تم تعبئة الاستمارة وتقديمها عبر معالج إنشاء الحالة',
      }] : [];

      tx.set(formRef, {
        code: def.code,
        title: def.title,
        projectId,
        projectRefId: projectRef.id,
        beneficiaryName,
        status,
        approvalIndex,
        approvalChain: def.approvalChain,
        approvals,
        createdBy: user.id,
        createdByName: user.fullName,
        createdByRole: user.role,
        createdAt: serverTimestamp(),
        updatedAt: serverTimestamp(),
        ownerDept: def.ownerDept,
        bridgesTo: def.bridgesTo || [],
        data: isF02 ? formData : {},
        assigneeId: null,
        files: [],
        stepStartedAt: isF02 ? serverTimestamp() : null,
      });
    }

    return { projectRefId: projectRef.id, projectId, formIds };
  });

  // Fire notification asynchronously — not part of the transaction (notifications
  // don't need atomicity with the workflow, and a failed notification shouldn't
  // roll back project creation).
  notifyResearchManagers(result.projectId, beneficiaryName, result.formIds['F-02'])
    .catch(err => console.error('[createProject] notification failed:', err));

  return result;
}

async function notifyResearchManagers(projectId: string, beneficiaryName: string, f02FormId?: string): Promise<void> {
  if (!f02FormId) return;
  const snap = await getDocs(query(collection(db, 'users'), where('status', '==', 'active')));
  const recipients = snap.docs
    .map(d => ({ id: d.id, ...d.data() } as { id: string; role: string; isAdmin?: boolean; department?: string }))
    .filter(u => u.role === 'RESEARCH_MANAGER' || (u.isAdmin && u.department === 'RESEARCH'))
    .map(u => u.id);
  if (!recipients.length) return;
  await addDoc(collection(db, 'notifications'), {
    recipients, readBy: [],
    text: `حالة جديدة بانتظار مراجعتك: ${beneficiaryName} (${projectId})`,
    subject: 'استمارة بحث اجتماعي جديدة',
    type: 'new_f02',
    link: f02FormId,
    createdAt: new Date().toISOString(),
  });
}
```

```tsx
const FieldRenderer: React.FC<{
  field: FieldDef; value: unknown; onChange: (v: unknown) => void; disabled?: boolean;
}> = ({ field, value, onChange, disabled }) => {
  const label = field.label + (field.required ? ' *' : '');
  if (field.type === 'select') {
    return <Select label={label} value={(value as string) || ''} onChange={(e) => onChange(e.target.value)} options={field.options || []} placeholder="اختر..." disabled={disabled} />;
  }
  if (field.type === 'textarea') {
    return <TextArea label={label} value={(value as string) || ''} onChange={(e) => onChange(e.target.value)} rows={3} disabled={disabled} hint={field.hint} />;
  }
  const htmlType = field.type === 'tel' ? 'tel' : field.type === 'number' ? 'number' : field.type === 'date' ? 'date' : 'text';
  return (
    <Input
      label={label} type={htmlType}
      value={(value as string | number) ?? ''}
      onChange={(e) => onChange(field.type === 'number' ? (Number(e.target.value) || 0) : e.target.value)}
      disabled={disabled} hint={field.hint} dir={field.dir}
    />
  );
};

      if (!data.beneficiaryAcceptance)     return 'يجب تأكيد قبول المستفيد للترميم';
      if (!data.documentsCollected)        return 'يجب تأكيد جمع المستندات الأساسية';
    }
    return null;
  };

  const next = () => {
    const err = validateStep(step);
    if (err) { setError(err); return; }
    setError(null);
    setStep(s => Math.min(4, s + 1));
  };
  const prev = () => { setError(null); setStep(s => Math.max(1, s - 1)); };

  const submit = async () => {
    for (let s = 1; s <= 4; s++) {
      const err = validateStep(s);
      if (err) { setStep(s); setError(err); return; }
    }
    setBusy(true);
    setError(null);
    try {
      const result = await createProject(data, user);
      try { localStorage.removeItem(draftKey); } catch { /* ignore */ }
      onComplete?.(result.projectRefId);
      onClose();
    } catch (e) {
      setError((e as Error).message || 'تعذّر إنشاء المشروع');
      setBusy(false);
    }
  };

  return (
    <div className="fixed inset-0 z-50 bg-black/50 flex items-end sm:items-center justify-center p-0 sm:p-4" dir="rtl">
```

## Notes worth flagging

1. **`createProject` is called from one place** — the wizard's submit handler. If you ever expose a programmatic project-creation API (e.g., bulk import), call this same function. The transaction is its single source of truth.

2. **`counters/projectSerial` doc is auto-created on first run.** The `merge: true` setDoc inside the transaction handles the no-doc case. No migration needed.

3. **Year rollover behavior** — the projectId uses `new Date().getFullYear()` at creation time, but the counter is global (not per-year). On Jan 1, project #007 of 2026 becomes TRM-2027-007. If you want per-year counters, swap to `counters/projectSerial-2026`. Not blocking; flag if the operations team objects.

4. **Notification after createProject is fire-and-forget** — wrapped in `.catch(console.error)` so a failed notification doesn't roll back the transaction. The form will still appear in the manager's queue via `formAwaitsUser` even without the notification.

When those pages get rebuilt (per your trajectory list), drop `<NewProjectButton user={profile} variant="primary" />` into Home's quick-action strip, `variant="fab"` floating on Projects, and `variant="menu-item"` inside the Forms "+" dropdown. 6. **Wizard's "Start Over" cleans `localStorage` only** — closing via the X header button preserves the draft for next session. Both behaviors are intentional.

🔴 Must fix in Batch 10 opening: F02Renderer doesn't stage-gate manager input
In F02_SocialResearch.tsx, the F02Renderer (view mode) section that captures the manager's verdict has no formAwaitsUser gate:
<Select label="القرار *" value={verdict || ''} onChange={...} />
<TextArea label="ملاحظات المراجعة" value={managerNotes} onChange={...} />
These inputs are always editable in the DOM. The autosave effect fires on any change. So:
	•	✅ Manager (stage 1, awaitsMe=true) opens F-02 → edits verdict + notes → approves. Correct.
	•	🔴 Researcher (stage 1, awaitsMe=false) opens F-02 → edits verdict + notes → autosave writes their changes to data.managerNotes and data.eligibilityVerdict. They can't approve, but they CAN modify what the manager will see.
	•	🔴 Any non-awaitee with read access to the project → same vulnerability.
The data integrity hole: a researcher could pre-fill the manager's verdict before the manager opens it, and the manager might not notice the prepopulated values were tampered with.
Fix — wrap the manager input section in a fieldset disabled by formAwaitsUser:
